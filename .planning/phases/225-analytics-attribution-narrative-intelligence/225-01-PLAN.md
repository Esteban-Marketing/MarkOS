       -- Canonical view: resolves multi-source disagreement on (metric_id, tenant_id, period) to ONE row

       CREATE OR REPLACE VIEW metric_canonical_view AS
       WITH ranked_sources AS (
         SELECT
           md.metric_id,
           md.tenant_id,
           md.last_computed_at AS period,
           md.source_precedence_chain,
           src.source_kind,
           src.source_value,
           src.commit_sha,
           src.inserted_at,
           -- Rank: lower precedence_index = higher precedence
           array_position(
             COALESCE(md.source_precedence_chain::text[], ARRAY[]::text[]),
             src.source_kind
           ) AS precedence_index
         FROM metric_definitions md
         JOIN LATERAL (
           SELECT 'cdp_events' AS source_kind, NULL::numeric AS source_value, NULL::text AS commit_sha, NULL::timestamptz AS inserted_at WHERE FALSE
           UNION ALL SELECT 'crm_activity', NULL, NULL, NULL WHERE FALSE
           UNION ALL SELECT 'dispatch_events', NULL, NULL, NULL WHERE FALSE
           UNION ALL SELECT 'conversion_events', NULL, NULL, NULL WHERE FALSE
           UNION ALL SELECT 'launch_outcomes', NULL, NULL, NULL WHERE FALSE
         ) src ON TRUE
       ),
       canonical AS (
         SELECT DISTINCT ON (metric_id, tenant_id, period)
           metric_id, tenant_id, period, source_kind, source_value, commit_sha, inserted_at
         FROM ranked_sources
         WHERE precedence_index IS NOT NULL
         ORDER BY metric_id, tenant_id, period,
                  precedence_index ASC,
                  commit_sha ASC NULLS LAST,
                  inserted_at ASC NULLS LAST
       )
       SELECT * FROM canonical;
