       export interface BetaParams { alpha: number; beta: number; }

       export const UNINFORMATIVE_PRIOR: BetaParams = { alpha: 1, beta: 1 };

       export function updatePosterior(prior: BetaParams, conversions: number, trials: number): BetaParams {
         return { alpha: prior.alpha + conversions, beta: prior.beta + (trials - conversions) };
       }

       function sampleBeta(params: BetaParams): number {
         // Sample from Beta(alpha, beta) via two Gamma(alpha,1) and Gamma(beta,1) draws
         // Use Marsaglia & Tsang method for Gamma(>=1)
         const x = sampleGamma(params.alpha);
         const y = sampleGamma(params.beta);
         return x / (x + y);
       }
       function sampleGamma(shape: number): number {
         // Marsaglia & Tsang for shape >= 1
         if (shape < 1) {
           // Boost: Gamma(shape) = Gamma(shape+1) * U^(1/shape)
           return sampleGamma(shape + 1) * Math.pow(Math.random(), 1 / shape);
         }
         const d = shape - 1 / 3;
         const c = 1 / Math.sqrt(9 * d);
         while (true) {
           let x: number;
           let v: number;
           do { x = standardNormal(); v = 1 + c * x; } while (v <= 0);
           v = v * v * v;
           const u = Math.random();
           if (u < 1 - 0.0331 * x ** 4) return d * v;
           if (Math.log(u) < 0.5 * x * x + d * (1 - v + Math.log(v))) return d * v;
         }
       }
       function standardNormal(): number {
         // Box-Muller
         return Math.sqrt(-2 * Math.log(Math.random())) * Math.cos(2 * Math.PI * Math.random());
       }

       export function bayesianProbabilityAWinsB(a: BetaParams, b: BetaParams, samples = 10000): number {
         let aWins = 0;
         for (let i = 0; i < samples; i++) {
           if (sampleBeta(a) > sampleBeta(b)) aWins++;
         }
         return aWins / samples;
       }
       ```

       export interface ZTestResult {
         z: number;
         p_value: number;
         significant_at_0_05: boolean;
       }

       function normCdf(z: number): number {
         // Abramowitz & Stegun approximation
         const t = 1 / (1 + 0.2316419 * Math.abs(z));
         const d = 0.3989422804014327 * Math.exp(-(z * z) / 2);
         const probability = d * t * (0.31938153 + t * (-0.356563782 + t * (1.781477937 + t * (-1.821255978 + t * 1.330274429))));
         return z >= 0 ? 1 - probability : probability;
       }

       export function zTestProportions(p1: number, n1: number, p2: number, n2: number): ZTestResult {
         const pooled = (p1 * n1 + p2 * n2) / (n1 + n2);
         const se = Math.sqrt(pooled * (1 - pooled) * (1 / n1 + 1 / n2));
         if (se === 0) return { z: 0, p_value: 1, significant_at_0_05: false };
         const z = (p1 - p2) / se;
         const p_two_sided = 2 * (1 - normCdf(Math.abs(z)));
         return { z, p_value: p_two_sided, significant_at_0_05: p_two_sided <= 0.05 };
       }
       ```

       export interface GuardrailResult {
         passed: boolean;
         failures: Array<{ rule: string; details: string }>;
       }

       export function enforceStatisticalGuardrails(args: {
         sample_size: number;
         min_sample_size: number;
         duration_days: number;
         min_duration_days: number;
         started_at: string;
         novelty_window_days: number;
       }): GuardrailResult {
         const failures: GuardrailResult['failures'] = [];
         if (args.sample_size < args.min_sample_size) {
           failures.push({ rule: 'min_sample_size', details: `${args.sample_size} < ${args.min_sample_size}` });
         }
         if (args.duration_days < args.min_duration_days) {
           failures.push({ rule: 'min_duration_days', details: `${args.duration_days} < ${args.min_duration_days}` });
         }
         const noveltyEnd = new Date(args.started_at).getTime() + args.novelty_window_days * 86400_000;
         if (Date.now() < noveltyEnd) {
           failures.push({ rule: 'novelty_window', details: `current time still within ${args.novelty_window_days}-day novelty window` });
         }
         return { passed: failures.length === 0, failures };
       }

         const { data: last } = await client.from('pricing_signals')
           .select('emitted_at, payload')
           .eq('tenant_id', args.tenant_id)
           .eq('signal_kind', args.signal_kind)
           .order('emitted_at', { ascending: false })
           .limit(1);

         if (!last || last.length === 0) return { allow: true };

         const lastEmittedMs = new Date(last[0].emitted_at).getTime();
         if (Date.now() - lastEmittedMs < 3600_000) {
           return { allow: false, reason: 'within_1h_debounce' };

         const lastObserved = (last[0].payload as { observed_value: number }).observed_value ?? 0;
         if (lastObserved === 0) return { allow: true };
