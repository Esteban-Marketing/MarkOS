         revenue: 1.0, pipeline: 0.9, conversion: 0.85, retention: 0.85,
         acquisition: 0.7, activation: 0.7, support: 0.55, pricing: 0.95, community: 0.4,

       export function scoreSeverity(deviationSigma: number, metric: MetricDefinition, observedValue: number): AnomalySeverity {
         const score = Math.abs(deviationSigma) * (CATEGORY_IMPORTANCE[metric.category] ?? 0.5);

         if (score < 3.5) return 'medium';
         if (score < 5) return 'high';
         return 'critical';

       export async function detectAnomaliesForMetric(client, args: { metric: MetricDefinition; tenant_id: string; period_start: string; period_end: string }): Promise<AnomalyDetection | null> {
         const cfg = args.metric.anomaly_config as { sigma_threshold: number; min_sample_size: number; sensitivity: 'low' | 'medium' | 'high'; seasonality_aware: boolean };

         // 1. Read historical values
         const history = cfg.seasonality_aware
           ? await readSeasonalHistory(client, args.metric.metric_id, args.tenant_id)
           : await readRollingHistory(client, args.metric.metric_id, args.tenant_id, cfg.min_sample_size);

         if (history.length < cfg.min_sample_size) return null; // not enough data

         const mean = history.reduce((a, b) => a + b.value, 0) / history.length;
         const stddev = Math.sqrt(history.reduce((a, b) => a + (b.value - mean) ** 2, 0) / history.length);
         if (stddev === 0) return null; // no variance

         // 2. Read observed value for the period
         const observedValue = await getMetricValueForPeriod(client, args.metric.metric_id, args.tenant_id, args.period_start, args.period_end);
         const deviationSigma = (observedValue - mean) / stddev;

         if (Math.abs(deviationSigma) < cfg.sigma_threshold) return null;

         const severity = scoreSeverity(deviationSigma, args.metric, observedValue);
         return {
           tenant_id: args.tenant_id, metric_id: args.metric.metric_id,
           period_start: args.period_start, period_end: args.period_end,
           baseline_value: mean, observed_value: observedValue, deviation_sigma: deviationSigma,
           severity, status: 'detected', detected_at: new Date().toISOString(),
           evidence_refs: [], narrative_id: null,
         };
       }
       ```

       export async function transitionAnomaly(client, args: { detection_id: string; new_status: AnomalyStatus; actor: string; reason?: string }): Promise<AnomalyDetection> {
         const { data: current } = await client.from('anomaly_detections').select('*').eq('detection_id', args.detection_id).single();

         if (args.new_status === 'resolved') updates.resolved_at = new Date().toISOString();

           metadata: { detection_id: args.detection_id, from: current.status, to: args.new_status, actor: args.actor, reason: args.reason },

       function hashCode(s: string): number {
         let h = 0;
         for (let i = 0; i < s.length; i++) { h = (h << 5) - h + s.charCodeAt(i); h |= 0; }
         return h;
       }
       export function applyJitterMs(tenantId: string, maxSeconds = 300): number {
         return Math.abs(hashCode(tenantId)) % (maxSeconds * 1000);
       }
       export async function jitterDelay(tenantId: string, maxSeconds = 300): Promise<void> {
         const ms = applyJitterMs(tenantId, maxSeconds);
         return new Promise((r) => setTimeout(r, ms));
       }

       export async function detectAnomalySpike(client, tenantId: string): Promise<{ spiking: boolean; recent_count: number; baseline_mean: number; baseline_stddev: number }> {
         // 1. Count anomalies in last 1h
         const { count: recent } = await client.from('anomaly_detections').select('*', { count: 'exact', head: true })
           .eq('tenant_id', tenantId).gte('detected_at', new Date(Date.now() - 3600_000).toISOString());

         // 2. 7-day rolling baseline (24 hourly buckets × 7 days = 168 samples)
         // Compute hourly buckets via SQL aggregation
         const { data: hourly } = await client.rpc('analytics_hourly_anomaly_counts', { p_tenant_id: tenantId, p_hours: 168 });

       import { detectAnomaliesForMetric } from '../../../lib/markos/analytics/anomaly/detector';
       import { shouldSuppress, setSuppression } from '../../../lib/markos/analytics/anomaly/suppressor';
       import { detectAnomalySpike } from '../../../lib/markos/analytics/anomaly/spike-of-spikes';
       import { evaluateRulesForTenant } from '../../../lib/markos/analytics/decision/engine';
       import { jitterDelay } from '../../../lib/markos/analytics/anomaly/thundering-herd';
       import { createTaskBridge } from '../../../lib/markos/analytics/decision/task-bridge';

       export default async function handler(req: Request) {
         // 1. Verify CRON_SECRET
         const auth = req.headers.get('authorization');
         if (auth !== `Bearer ${process.env.CRON_SECRET}`) return new Response('unauthorized', { status: 401 });

         const url = new URL(req.url);
         const cadence = url.searchParams.get('cadence') ?? 'hourly'; // hourly|daily|weekly

         // 2. Loop tenants (paginated)
         // D-49: inline service-role client (no serviceRoleClient() helper exists; pattern from api/submit.js:39-43)
         const { createClient } = require('@supabase/supabase-js');
         const client = createClient(
           process.env.NEXT_PUBLIC_SUPABASE_URL,
           process.env.SUPABASE_SERVICE_ROLE_KEY,
           { auth: { persistSession: false } }
         );
         const tenantIds = await listActiveTenants(client);
         let totalDetected = 0; let totalRulesFired = 0; let spikesDetected = 0;

         for (const tenantId of tenantIds) {
           await jitterDelay(tenantId, 300);

           // 3. Read metrics for cadence
           const { data: metrics } = await client.from('metric_definitions').select('*')
             .eq('freshness_mode', cadence).eq('status', 'approved')
             .or(`tenant_id.is.null,tenant_id.eq.${tenantId}`);

           const periodStart = computePeriodStart(cadence);
           const periodEnd = new Date().toISOString();

           for (const metric of metrics ?? []) {
             if (await shouldSuppress(client, { metric_id: metric.metric_id, tenant_id: tenantId })) continue;
             const detection = await detectAnomaliesForMetric(client, { metric, tenant_id: tenantId, period_start: periodStart, period_end: periodEnd });
             if (!detection) continue;
             const { data: inserted } = await client.from('anomaly_detections').insert(detection).select().single();
             await setSuppression(client, { detection_id: inserted.detection_id, window_seconds: 3600 });
             totalDetected++;
           }

           // 4. Evaluate decision rules for this tenant
           const ruleResult = await evaluateRulesForTenant(client, { tenant_id: tenantId, since: new Date(periodStart) });
           totalRulesFired += ruleResult.rules_fired;

           // 5. Spike-of-spikes check (D-41)
           const spike = await detectAnomalySpike(client, tenantId);
           if (spike.spiking) {
             await createTaskBridge(client, {
               tenant_id: tenantId, title: `Anomaly burst detected (${spike.recent_count} in 1h vs ${spike.baseline_mean.toFixed(1)} baseline)`,
               priority: 'P1', metadata: { recent_count: spike.recent_count, baseline_mean: spike.baseline_mean, baseline_stddev: spike.baseline_stddev },
             });
             spikesDetected++;
           }
         }

         return Response.json({ tenants: tenantIds.length, detected: totalDetected, rules_fired: totalRulesFired, spikes_detected: spikesDetected });
       }
       ```
