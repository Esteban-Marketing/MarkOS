       export function computeTimeDecay(events: SourcedEvent[], cfg: AttributionModel): AttributionTouchInput[] {
         if (events.length === 0) return [];
         const halfLife = (cfg.weights_strategy as { half_life_days: number }).half_life_days ?? 7;
         const lambda = Math.log(2) / halfLife; // decay constant
         const last = new Date(events[events.length - 1].occurred_at).getTime();
         const rawWeights = events.map(e => Math.exp(-lambda * (last - new Date(e.occurred_at).getTime()) / 86400_000));
         const sum = rawWeights.reduce((a, b) => a + b, 0);
         return events.map((e, i) => ({
           touch_type: i === 0 ? 'first' : i === events.length - 1 ? 'last' : 'assist',
           credit_weight: rawWeights[i] / sum,
           confidence_score: 1.0,
           // ... rest
         }));
       }

         event_id: string;
         tenant_id: string;
         opportunity_id: string;
         profile_id: string | null;
         account_id: string | null;
         campaign_id: string | null;

         occurred_at: string;
         source_table: 'cdp_events' | 'crm_activity' | 'dispatch_events' | 'conversion_events';
       }

       const CHANNEL_BY_DOMAIN: Record<string, AttributionChannel> = {
         'website.organic': 'organic_search',
         'website.paid': 'paid_search',
         'social': 'social',
         'email': 'email',
         'whatsapp': 'messaging', 'sms': 'messaging', 'push': 'messaging',
         'crm.referral': 'referral',
         'crm.partner': 'partner',
         'community': 'community',
         'event': 'event',
         'product': 'product',
       };

       export async function gatherEventsForOpportunity(client, opportunityId: string, tenantId: string, lookbackDays: number): Promise<SourcedEvent[]> {
         const cutoff = new Date(Date.now() - lookbackDays * 86400_000).toISOString();

         // 1. Read opportunity to get profile_id + account_id
         const { data: opp } = await client.from('crm_opportunities').select('profile_id, account_id').eq('opportunity_id', opportunityId).single();
         if (!opp) return [];

         // 2. cdp_events (P221) — partition pruning: explicit profile_id + occurred_at filter
         const { data: cdp } = await client.from('cdp_events')
           .select('event_id, event_domain, occurred_at, properties, profile_id')
           .eq('tenant_id', tenantId).eq('profile_id', opp.profile_id).gte('occurred_at', cutoff);

         // 3. crm_activity (P222)
         // 4. dispatch_events (P223)
         // 5. conversion_events (P224)

         // Merge + map to SourcedEvent
         const events: SourcedEvent[] = [];
         for (const e of cdp ?? []) {
           events.push({
             event_id: e.event_id, tenant_id: tenantId, opportunity_id: opportunityId,
             profile_id: opp.profile_id, account_id: opp.account_id,
             campaign_id: e.properties?.campaign_id ?? null,
             campaign_kind: e.properties?.campaign_kind ?? null,
             channel: CHANNEL_BY_DOMAIN[e.event_domain] ?? 'direct',
             occurred_at: e.occurred_at,
             source_table: 'cdp_events',
           });
         }
         // ... similar for other 3 sources

         events.sort((a, b) => new Date(a.occurred_at).getTime() - new Date(b.occurred_at).getTime());
         return events;
       }

         position_based: computePositionBased,

       export async function computeAttributionForOpportunity(client, opportunityId: string, tenantId: string): Promise<AttributionTouchInput[]> {
         // 1. Read all active models for tenant + global (status='active' only)
         const { data: models } = await client.from('attribution_models').select('*').eq('status', 'active').or(`tenant_id.is.null,tenant_id.eq.${tenantId}`);

         // 2. For each model: gather events (model.lookback_days) + invoke matching compute function
         const touches: AttributionTouchInput[] = [];
         for (const model of models ?? []) {
           const events = await gatherEventsForOpportunity(client, opportunityId, tenantId, model.lookback_days);
           const fn = COMPUTE[model.name];
           if (!fn) continue;
           touches.push(...fn(events, model));
         }
         return touches;
       }
