       ```typescript
       const PRICING_TOKENS = /\$|\bprice\b|\bcost\b|\btier\b|\bplan\b|\bdiscount\b|\bMRR\b|\bARR\b|save offer|payback/gi;

       export async function enforcePricingPlaceholder(client, args: { text: string; narrative_kind: NarrativeKind; tenant_id: string }): Promise<{ output_text: string; placeholders_inserted: number; pricing_records_consumed: string[] }> {
         if (!PRICING_TOKENS.test(args.text)) return { output_text: args.text, placeholders_inserted: 0, pricing_records_consumed: [] };

         // Probe for P205 approved PricingRecommendation
         const { data: approved } = await client.from('pricing_recommendations').select('recommendation_id, scope_metadata').eq('tenant_id', args.tenant_id).eq('status', 'approved').limit(10).catch(() => ({ data: [] }));

         if (!approved || approved.length === 0) {
           // No approved Pricing Engine record → replace pricing-touching sentences with placeholder
           // Strategy: split text by sentence; for each sentence matching PRICING_TOKENS, replace with {{MARKOS_PRICING_ENGINE_PENDING}}
           const sentences = args.text.split(/(?<=[.!?])\s+/);
           let placeholders = 0;
           const result = sentences.map((s) => {
             if (PRICING_TOKENS.test(s)) { placeholders++; return '{{MARKOS_PRICING_ENGINE_PENDING}}'; }
             return s;
           }).join(' ');
           return { output_text: result, placeholders_inserted: placeholders, pricing_records_consumed: [] };
         }

         // Pricing Engine has approved records → use them; mark consumption
         return { output_text: args.text, placeholders_inserted: 0, pricing_records_consumed: approved.map((a) => a.recommendation_id) };
       }
       ```

       ```typescript
       export interface NarrativeClaim {
         claim_text: string;
         span_start: number;
         span_end: number;
         evidence_ref: string | null;
         confidence: 'high' | 'medium' | 'low';
       }

       const NUMERIC_SPAN = /(\$?\d[\d,.]*\s*(%|days?|weeks?|months?|x|MRR|ARR|customers?|opportunities|conversions?|leads?|signups?|seats?)?)/g;

       export function extractClaims(generatedText: string, slotsWithEvidence: Map<string, { value: string; evidence_ref: string }>): NarrativeClaim[] {
         const claims: NarrativeClaim[] = [];
         let match: RegExpExecArray | null;
         while ((match = NUMERIC_SPAN.exec(generatedText)) !== null) {
           const span = match[0];
           // Check if span value matches a slot value
           let evidenceRef: string | null = null;
           let confidence: 'high' | 'medium' | 'low' = 'low';
           for (const [_slotName, slot] of slotsWithEvidence) {
             if (slot.value === span || slot.value.includes(span) || span.includes(slot.value)) {
               evidenceRef = slot.evidence_ref;
               confidence = 'high';
               break;
       ```

       ```typescript
         const events = await gatherEventsForProfile(client, { profile_id: args.profile_id, tenant_id: args.tenant_id, period_start: args.period_start, period_end: args.period_end });
         if (events.length === 0) return { steps_created: 0, steps_updated: 0 };

         // Read prior journey_step to determine starting funnel_stage
         const { data: prior } = await client.from('journey_steps').select('funnel_stage').eq('tenant_id', args.tenant_id).eq('profile_id', args.profile_id).lt('period_start', args.period_start).order('period_start', { ascending: false }).limit(1);
         const startingStage: FunnelStage = (prior?.[0]?.funnel_stage as FunnelStage) ?? 'anonymous';

         const { final_stage, transitions } = cdpEventsToFunnelStage(events, startingStage);

         // Group events by funnel_stage; for each stage period, write a journey_steps row
         const stageEventGroups = groupEventsByStage(events, transitions, startingStage);
         let created = 0, updated = 0;
         for (const [stage, stageEvents] of stageEventGroups) {
           const stepHash = await deterministicStepHash({
             profile_id: args.profile_id, tenant_id: args.tenant_id,
             period_start: stageEvents[0].occurred_at,
             event_refs: stageEvents.map((e) => e.event_id),
           });
           const timeInStage = stageEvents.length > 1
             ? Math.floor((new Date(stageEvents[stageEvents.length - 1].occurred_at).getTime() - new Date(stageEvents[0].occurred_at).getTime()) / 1000)
             : 0;
           const { data: upserted } = await client.from('journey_steps').upsert({
             step_id: stepHash, tenant_id: args.tenant_id, profile_id: args.profile_id,
             account_id: events[0].account_id, period_start: stageEvents[0].occurred_at, period_end: stageEvents[stageEvents.length - 1].occurred_at,
             ordered_event_refs: stageEvents.map((e) => e.event_id),
             funnel_stage: stage as FunnelStage, time_in_stage_seconds: timeInStage,
             transition_drivers: transitions.filter((t) => t.to_stage === stage).map((t) => ({ channel: t.channel, commercial_signal: t.commercial_signal, occurred_at: t.occurred_at })),
             computed_at: new Date().toISOString(),
           }, { onConflict: 'tenant_id,profile_id,period_start,funnel_stage' }).select();
           if (upserted) created++; // Upsert returns row regardless; for accuracy use returning=minimal
         }

         return { steps_created: created, steps_updated: updated };
       }
       ```
