         const available = error === null || error.code !== '42P01'; // 42P01 = undefined_table
         p207AvailableCache = { ts: Date.now(), available };

       export async function wrapInAgentRun<T>(
         fn: () => Promise<T>,
         ctx: { client: ReturnType<typeof createClient>; domain: string; tenant_id: string; metric_id?: string }
       ): Promise<{ result: T; run_id: string | null }> {
         if (await isAgentRunAvailable(ctx.client)) {
           const runId = crypto.randomUUID();
           await ctx.client.from('markos_agent_runs').insert({
             run_id: runId, tenant_id: ctx.tenant_id,
             domain: ctx.domain, status: 'running', metadata: { metric_id: ctx.metric_id },
             started_at: new Date().toISOString()
           });
           try {
             const result = await fn();
             await ctx.client.from('markos_agent_runs').update({
               status: 'completed', completed_at: new Date().toISOString()
             }).eq('run_id', runId);
             return { result, run_id: runId };
           } catch (err) {
             await ctx.client.from('markos_agent_runs').update({
               status: 'failed', completed_at: new Date().toISOString(),
               error_message: String(err)
             }).eq('run_id', runId);
             throw err;

       import jsonLogic from 'json-logic-js';

       export class FormulaEvaluationError extends Error {
         constructor(public code: string, message: string) { super(message); }
       }

       export interface FormulaContext {
         tenant_id: string;
         lookback_days: number;
         grain: 'person' | 'account' | 'campaign' | 'content' | 'channel' | 'tenant';
         events: { event_family: string; occurred_at: string; properties: Record<string, unknown>; profile_id?: string; account_id?: string }[];
         dimension_filter?: Record<string, string>;
       }
       export interface FormulaResult {
         value: number;
         sample_size: number;
         computed_at: string;
         evidence_refs: string[]; // event UUIDs
       }

       const ALLOWED_FORMULA_TYPES = ['count','sum','ratio','sum_weighted','percentage','distinct_count','decay'];

       export function evaluateFormula(formulaDsl: unknown, ctx: FormulaContext): FormulaResult {
         if (!formulaDsl || typeof formulaDsl !== 'object') {
           throw new FormulaEvaluationError('invalid_ast', 'formula_dsl must be an object');
         }
         const ast = formulaDsl as Record<string, unknown>;
         const t = ast.type;
         if (typeof t !== 'string' || !ALLOWED_FORMULA_TYPES.includes(t)) {
           throw new FormulaEvaluationError('unsafe_formula', `unknown formula type: ${t}`);
         }

         // Reject any string field that matches SQL keywords (defense in depth)
         const stringified = JSON.stringify(ast);
         if (/\b(SELECT|INSERT|UPDATE|DELETE|DROP|ALTER|CREATE|TRUNCATE|GRANT|EXEC|UNION|--)\b/i.test(stringified)) {
           throw new FormulaEvaluationError('unsafe_formula', 'formula contains SQL-like tokens');
         }

         // Apply lookback filter
         const cutoff = new Date(Date.now() - ctx.lookback_days * 86400_000);
         const events = ctx.events.filter(e => new Date(e.occurred_at) >= cutoff);

         let value: number; let sample_size: number; let evidence: string[] = [];

         switch (t) {
           case 'count': {
             const fam = ast.event_family;
             const matched = events.filter(e => e.event_family === fam);
             value = matched.length;
             sample_size = matched.length;
             evidence = matched.map(m => (m.properties.event_id as string) ?? '');
             break;
           }
           case 'ratio': {
             const num = evaluateFormula(ast.numerator, ctx);
             const den = evaluateFormula(ast.denominator, ctx);
             value = den.value === 0 ? 0 : num.value / den.value;
             sample_size = den.sample_size;
             evidence = [...num.evidence_refs, ...den.evidence_refs];
             break;
           }
           case 'sum': { /* sum events.properties[field] */ break; }
           case 'percentage': { /* ratio * 100 */ break; }
           case 'distinct_count': { /* unique by field */ break; }
           case 'sum_weighted': { /* time-decay weighted sum */ break; }
           // ... all 7 types
         }

         return { value, sample_size, computed_at: new Date().toISOString(), evidence_refs: evidence };
       }

       export async function recomputeMetricForTenant(client, args: {
         metric_id: string; tenant_id: string; period_start: string; period_end: string;
       }): Promise<{ value: number; sample_size: number; computed_at: string; run_id: string | null }> {
         const { result, run_id } = await wrapInAgentRun(async () => {
           const metric = await resolveMetricForTenant(client, args.metric_id, args.tenant_id);
           if (!metric) throw new Error('metric_not_found');

           // Pre-fetch tenant-scoped events for source_event_families
           const events = await client.from('cdp_events')
             .select('*')
             .eq('tenant_id', args.tenant_id)
             .in('event_domain', metric.source_event_families)
             .gte('occurred_at', args.period_start)
             .lte('occurred_at', args.period_end);

           const fr = evaluateFormula(metric.formula_dsl, {
             tenant_id: args.tenant_id,
             lookback_days: Math.ceil((new Date(args.period_end).getTime() - new Date(args.period_start).getTime()) / 86400_000),
             grain: metric.grain,

           return fr;
         }, { client, domain: 'analytics.recompute', tenant_id: args.tenant_id, metric_id: args.metric_id });
