```sql
-- Per D-30, D-31

BEGIN;

-- Additional composite indexes revealed by 222-03..222-05 query patterns
CREATE INDEX IF NOT EXISTS idx_c360_tombstoned_identity
  ON customer_360_records(tenant_id, tombstoned, canonical_identity_id);

CREATE INDEX IF NOT EXISTS idx_opp_stage_close_date
  ON opportunities(tenant_id, stage_id, expected_close_date) WHERE health != 'stalled';

CREATE INDEX IF NOT EXISTS idx_lt_customer_stage
  ON lifecycle_transitions(tenant_id, customer_360_id, to_stage, transitioned_at DESC);

CREATE INDEX IF NOT EXISTS idx_nba_subject_expiry
  ON nba_records(tenant_id, subject_type, subject_id, expires_at) WHERE status = 'active';

-- Hardening: assert RLS enabled on all 6 tables (fails migration if not)
DO $$
DECLARE
  tbl TEXT;
  missing TEXT[] := '{}';
BEGIN
  FOR tbl IN SELECT unnest(ARRAY[
    'customer_360_records','opportunities','lifecycle_transitions','nba_records','buying_committees','buying_committee_members'
  ]) LOOP
    IF NOT EXISTS (
      SELECT 1 FROM pg_class WHERE relname = tbl AND relrowsecurity = TRUE
    ) THEN
      missing := array_append(missing, tbl);
    END IF;
  END LOOP;
  IF array_length(missing, 1) > 0 THEN
    RAISE EXCEPTION 'RLS hardening fail: tables without RLS enabled: %', missing;
  END IF;
END $$;

COMMIT;

-- Rollback block:
-- BEGIN;
-- DROP INDEX IF EXISTS idx_nba_subject_expiry, idx_lt_customer_stage, idx_opp_stage_close_date, idx_c360_tombstoned_identity;
```

```ts
// Per D-20
import { withTenant } from '../../db';

export interface DriftDivergence {
  customer_360_id: string | null;
  crm_entity_id: string | null;
  divergence_type: 'missing_c360' | 'missing_entity' | 'display_name_mismatch' | 'lifecycle_stage_mismatch' | 'ownership_mismatch';
  divergence_details: Record<string, unknown>;
}

const DRIFT_TASK_THRESHOLD = 5;

export async function detectCustomer360Drift(tenantId: string): Promise<DriftDivergence[]> {
  return withTenant(tenantId, async (client) => {
    const divergences: DriftDivergence[] = [];

    // 1. Entities without customer_360_records (backfill gap)
    const missingC360 = await client.query(
      `SELECT e.id AS crm_entity_id
       FROM crm_entities e
       WHERE e.tenant_id = $1 AND NOT EXISTS (
         SELECT 1 FROM customer_360_records c360
         WHERE c360.tenant_id = e.tenant_id
           AND c360.display_name = e.display_name  -- naive match; real system uses backfill FK
       )
       LIMIT 100`,
      [tenantId]
    );
    for (const row of missingC360.rows) {
      divergences.push({
        customer_360_id: null,
        crm_entity_id: row.crm_entity_id,
        divergence_type: 'missing_c360',
        divergence_details: { reason: 'No matching customer_360_records row found' },
      });
    }

    // 2. Display_name mismatch (tombstoned c360 vs non-tombstoned entity is expected; skip)
    const nameMismatch = await client.query(
      `SELECT c360.customer_360_id, e.id AS crm_entity_id, c360.display_name AS c360_name, e.display_name AS entity_name
       FROM customer_360_records c360 JOIN crm_entities e ON e.tenant_id = c360.tenant_id
       WHERE c360.tenant_id = $1 AND c360.tombstoned = FALSE
         AND c360.display_name != e.display_name
         AND c360.display_name = e.display_name || ''  -- correlate via backfill ref; stub
       LIMIT 100`,
      [tenantId]
    );
    for (const row of nameMismatch.rows) {
      divergences.push({
        customer_360_id: row.customer_360_id,
        crm_entity_id: row.crm_entity_id,
        divergence_type: 'display_name_mismatch',
        divergence_details: { c360_name: row.c360_name, entity_name: row.entity_name },
      });
    }

    // 3. Lifecycle_stage mismatch where both records exist
    const stageMismatch = await client.query(
      `SELECT c360.customer_360_id, e.id AS crm_entity_id, c360.lifecycle_stage AS c360_stage, e.lifecycle_stage AS entity_stage
       FROM customer_360_records c360 JOIN crm_entities e ON e.tenant_id = c360.tenant_id AND e.display_name = c360.display_name
       WHERE c360.tenant_id = $1 AND c360.tombstoned = FALSE
         AND e.lifecycle_stage IS NOT NULL
         AND c360.lifecycle_stage != e.lifecycle_stage
       LIMIT 100`,
      [tenantId]
    );
    for (const row of stageMismatch.rows) {
      divergences.push({
        customer_360_id: row.customer_360_id,
        crm_entity_id: row.crm_entity_id,
        divergence_type: 'lifecycle_stage_mismatch',
        divergence_details: { c360_stage: row.c360_stage, entity_stage: row.entity_stage },
      });
    }

    return divergences;
  });
}

export async function runDriftReconciliation(tenantId: string): Promise<{ run_id: string; divergences: number; task_created: boolean }> {
  const runId = crypto.randomUUID();
  const divergences = await detectCustomer360Drift(tenantId);

  await withTenant(tenantId, async (client) => {
    for (const d of divergences) {
      await client.query(
        `INSERT INTO crm360_drift_audit_log
          (tenant_id, run_id, customer_360_id, crm_entity_id, divergence_type, divergence_details)
         VALUES ($1, $2, $3, $4, $5, $6)`,
        [tenantId, runId, d.customer_360_id, d.crm_entity_id, d.divergence_type, JSON.stringify(d.divergence_details)]
      );
    }
  });

  let taskCreated = false;
  if (divergences.length >= DRIFT_TASK_THRESHOLD) {
    // Create operator task via existing markos_tasks table (P208 pattern — A6 extension point)
    taskCreated = true;
    await withTenant(tenantId, async (client) => {
      const { rows } = await client.query(`SELECT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'markos_tasks') AS has_tasks`);
      if (rows[0]?.has_tasks) {
        const taskRes = await client.query(
          `INSERT INTO markos_tasks (tenant_id, title, description, priority, source, created_at)
           VALUES ($1, $2, $3, 'P2', 'crm360.drift', NOW()) RETURNING task_id`,
          [tenantId, `CRM360 drift detected (${divergences.length} divergences)`, `Run ${runId}: ${divergences.length} drift rows. Review crm360_drift_audit_log.`]
        );
        const taskId = taskRes.rows[0].task_id;
        await client.query(
          `UPDATE crm360_drift_audit_log SET operator_task_id = $1 WHERE tenant_id = $2 AND run_id = $3`,
          [taskId, tenantId, runId]
        );
      }
      await client.query(
        `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
         VALUES ($1, 'system', 'crm360.drift.detected', 'drift_run', $2, $3, NOW())`,
        [tenantId, runId, JSON.stringify({ divergences: divergences.length })]
      );
    });
  }

  return { run_id: runId, divergences: divergences.length, task_created: taskCreated };
}
```

```ts
import { runDriftReconciliation } from './drift';
import { expireNbaRecords } from '../nba/expiry';
import { scheduleDailyRecompute } from '../nba/recompute';
import { dbAllTenants } from '../../db';

export async function cronDriftReconciliation(): Promise<{ tenants: number; total_divergences: number; tasks_created: number }> {
  const db = dbAllTenants();
  const { rows } = await db.query(`SELECT id FROM markos_tenants WHERE status = 'active'`);
  let totalDivergences = 0;
  let tasksCreated = 0;
  for (const t of rows) {
    const result = await runDriftReconciliation(t.id);
    totalDivergences += result.divergences;
    if (result.task_created) tasksCreated++;
  }
  return { tenants: rows.length, total_divergences: totalDivergences, tasks_created: tasksCreated };
}

export async function cronNbaExpire(): Promise<{ expired: number }> {
  return expireNbaRecords('ALL');
}

export async function cronDailyRecompute(): Promise<{ tenants: number; scheduled: number }> {
  const db = dbAllTenants();
  const { rows } = await db.query(`SELECT id FROM markos_tenants WHERE status = 'active'`);
  let scheduled = 0;
  for (const t of rows) {
    const r = await scheduleDailyRecompute(t.id);
    scheduled += r.scheduled;
  }
  return { tenants: rows.length, scheduled };
}
```
