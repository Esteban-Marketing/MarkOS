// Per D-16, RD-01 (application-side derivation — NOT Postgres generated column)
import type { Customer360Record } from '../records/customer360';

const MARKETING_STAGES = new Set(['anonymous', 'known', 'engaged', 'mql']);
const DEAL_STAGES = new Set(['sql', 'opportunity']);
const CS_STAGES = new Set(['customer', 'expansion', 'renewal', 'advocate']);

export function derivePrimaryOwner(c360: Pick<
  Customer360Record,
  'lifecycle_stage' | 'account_owner_user_id' | 'deal_owner_user_id' | 'cs_owner_user_id' | 'marketing_owner_user_id'
>): string | null {
  const stage = c360.lifecycle_stage;
  if (MARKETING_STAGES.has(stage)) {
    return c360.marketing_owner_user_id ?? c360.account_owner_user_id ?? null;
  }
  if (DEAL_STAGES.has(stage)) {
    return c360.deal_owner_user_id ?? c360.account_owner_user_id ?? null;
  }
  if (CS_STAGES.has(stage)) {
    return c360.cs_owner_user_id ?? c360.account_owner_user_id ?? null;
  }
  if (stage === 'lost') {
    return c360.account_owner_user_id ?? null;
  }
  // Unknown stage — fail safe to account_owner
  return c360.account_owner_user_id ?? null;
}

// Per D-11/D-12/D-13, Pitfall 5 (concurrent race)
// D-13 EMIT LOOP (Blocker 3 fix): appendLifecycleTransition ALWAYS writes THREE rows:
//   1) lifecycle_transitions INSERT (event source-of-truth)
//   2) crm_activity_ledger INSERT (activity_family='lifecycle_transition', source_domain='crm',
//      commercial_signal derived from to_stage, source_event_ref shared)
//   3) cdp_events envelope via emitCrmEventEnvelope (fail-closed per RD-10)
// All three share input.source_event_ref. Any failure in 3 throws — fail-closed (RD-10).
import { db, withTenant } from '../../db';  // existing Supabase client helper
import { emitCrmEventEnvelope } from '../events/emit';
export interface LifecycleTransition { /* per interfaces */ }
export interface AppendLifecycleTransitionInput { /* per interfaces */ }
export class ConcurrentTransitionError extends Error { code = 'CONCURRENT_TRANSITION_SUPERSEDED' as const; constructor(msg?: string) { super(msg ?? 'Lifecycle transition superseded by concurrent write'); } }

// D-13 commercial_signal mapping (Blocker 3):
const TO_STAGE_TO_COMMERCIAL_SIGNAL: Readonly<Record<string, string | null>> = Object.freeze({
  mql: 'interest',
  sql: 'interest',
  opportunity: 'interest',
  customer: 'expansion',
  expansion: 'expansion',
  renewal: 'renewal',
  advocate: 'expansion',
  lost: 'risk',
  // anonymous | known | engaged -> null (no commercial signal yet)
});

export async function appendLifecycleTransition(input: AppendLifecycleTransitionInput): Promise<LifecycleTransition> {
  return withTenant(input.tenant_id, async (client) => {
    // 1. First-writer-wins UPDATE on customer_360_records (Pitfall 5).
    //    SKIP this UPDATE when from_stage === to_stage AND reason === 'handoff' (D-17 same-stage handoff path
    //    — stage is unchanged, customer_360_records ownership columns are updated by recordOwnershipHandoff).
    if (!(input.from_stage === input.to_stage && input.reason === 'handoff')) {
      const updateResult = await client.query(
        `UPDATE customer_360_records SET lifecycle_stage = $1, updated_at = NOW()
         WHERE customer_360_id = $2 AND tenant_id = $3 AND lifecycle_stage = $4
         RETURNING customer_360_id, canonical_identity_id`,
        [input.to_stage, input.customer_360_id, input.tenant_id, input.from_stage]
      );
      if (updateResult.rowCount === 0) {
        throw new ConcurrentTransitionError(`Customer360 ${input.customer_360_id} not at from_stage=${input.from_stage}`);
      }
    }
    // 2. Append transition row
    const insertResult = await client.query(
      `INSERT INTO lifecycle_transitions
        (tenant_id, customer_360_id, from_stage, to_stage, actor_id, actor_type, evidence_ref, reason, source_event_ref)
       VALUES ($1,$2,$3,$4,$5,$6,$7,$8,$9) RETURNING *`,
      [input.tenant_id, input.customer_360_id, input.from_stage, input.to_stage,
       input.actor_id, input.actor_type, input.evidence_ref ?? null, input.reason ?? null, input.source_event_ref]
    );
    const transition = insertResult.rows[0];

    // 3. D-13 (Blocker 3): Write crm_activity_ledger row (source_domain='crm')
    const commercialSignal = TO_STAGE_TO_COMMERCIAL_SIGNAL[input.to_stage] ?? null;
    await client.query(
      `INSERT INTO crm_activity_ledger
        (tenant_id, entity_id, activity_family, source_domain, commercial_signal, actor_type, actor_id, thread_id, occurred_at, payload)
       VALUES ($1, $2, 'lifecycle_transition', 'crm', $3, $4, $5, $6, NOW(), $7)`,
      [input.tenant_id, input.customer_360_id, commercialSignal, input.actor_type, input.actor_id,
       input.source_event_ref,
       JSON.stringify({ from_stage: input.from_stage, to_stage: input.to_stage, reason: input.reason, transition_id: transition.transition_id })]
    );

    // 4. D-13 (Blocker 3): Emit cdp_events envelope with shared source_event_ref (fail-closed per RD-10).
    //    Resolve canonical_identity_id from customer_360_records.
    const c360 = await client.query(
      `SELECT canonical_identity_id FROM customer_360_records WHERE tenant_id = $1 AND customer_360_id = $2`,
      [input.tenant_id, input.customer_360_id]
    );
    await emitCrmEventEnvelope({
      source_event_ref: input.source_event_ref,
      tenant_id: input.tenant_id,
      crm_event_name: input.reason === 'handoff' ? 'crm.ownership.handoff' : 'crm.lifecycle.transition',
      occurred_at: transition.transitioned_at,
      canonical_identity_id: c360.rows[0]?.canonical_identity_id ?? null,
      account_id: null,
      mutation_payload: {
        from_stage: input.from_stage,
        to_stage: input.to_stage,
        reason: input.reason,
        transition_id: transition.transition_id,
      },
    });

    return transition;
  });
}

export async function getLifecycleHistory(tenantId: string, customer360Id: string, opts?: { limit?: number }): Promise<LifecycleTransition[]> {
  return withTenant(tenantId, async (client) => {
    const res = await client.query(
      `SELECT * FROM lifecycle_transitions WHERE tenant_id = $1 AND customer_360_id = $2
       ORDER BY transitioned_at DESC LIMIT $3`,
      [tenantId, customer360Id, opts?.limit ?? 100]
    );
    return res.rows;
  });
}

// Per D-07, D-22, RD-10 (fail-closed), P221 D-28 (no raw PII)
import { db, withTenant } from '../../db';

export interface CrmMutationContext { /* per <interfaces> */ }
export class CdpEventEmitError extends Error { code = 'CDP_EMIT_FAILED' as const; constructor(public cause?: unknown) { super('cdp_events write failed; CRM mutation must not silently succeed (RD-10)'); } }

const PII_KEY_REGEX = /^(email|phone|display_name|full_name|first_name|last_name|address|ssn|tax_id|passport|ip_address)$/i;
export function scrubPiiFromPayload(payload: Record<string, unknown>): Record<string, unknown> {
  const result: Record<string, unknown> = {};
  for (const [k, v] of Object.entries(payload)) {
    if (PII_KEY_REGEX.test(k)) continue;
    if (v && typeof v === 'object' && !Array.isArray(v)) {
      result[k] = scrubPiiFromPayload(v as Record<string, unknown>);
    } else {
      result[k] = v;
    }
  }
  return result;
}

export async function emitCrmEventEnvelope(ctx: CrmMutationContext): Promise<void> {
  const cleanPayload = scrubPiiFromPayload(ctx.mutation_payload);
  try {
    await withTenant(ctx.tenant_id, async (client) => {
      // P221 hard prerequisite (D-38) — cdp_events table guaranteed present by preflight gate. NO A3 stub branch.
      await client.query(
        `INSERT INTO cdp_events (event_id, tenant_id, event_name, event_domain, occurred_at, profile_id, account_id, properties)
         VALUES ($1, $2, $3, 'crm', $4, $5, $6, $7)`,
        [ctx.source_event_ref, ctx.tenant_id, ctx.crm_event_name, ctx.occurred_at,
         ctx.canonical_identity_id, ctx.account_id, JSON.stringify(cleanPayload)]
      );
    });
  } catch (err) {
    throw new CdpEventEmitError(err);
  }
}
