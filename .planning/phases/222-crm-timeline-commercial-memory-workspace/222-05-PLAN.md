// Per D-14, D-30
import { withTenant } from '../../db';
export interface BuyingCommittee { /* per <interfaces> */ }

const ALL_ROLES = ['champion','economic_buyer','technical_buyer','end_user','blocker','legal','finance','unknown'] as const;

export async function getCommitteeBySubject(tenantId: string, subjectType: BuyingCommittee['subject_type'], subjectId: string): Promise<BuyingCommittee | null> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT * FROM buying_committees WHERE tenant_id = $1 AND subject_type = $2 AND subject_id = $3`,
      [tenantId, subjectType, subjectId]
    );
    return rows[0] ?? null;
  });
}

export async function upsertCommittee(tenantId: string, input: Omit<BuyingCommittee,'committee_id'|'created_at'|'updated_at'|'tenant_id'>): Promise<BuyingCommittee> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `INSERT INTO buying_committees (tenant_id, subject_type, subject_id, coverage_score, missing_roles, last_assessed_at, assessment_actor_id)
       VALUES ($1,$2,$3,$4,$5,NOW(),$6)
       ON CONFLICT (tenant_id, subject_type, subject_id) DO UPDATE SET
         coverage_score = EXCLUDED.coverage_score,
         missing_roles = EXCLUDED.missing_roles,
         last_assessed_at = NOW(),
         assessment_actor_id = EXCLUDED.assessment_actor_id,
         updated_at = NOW()
       RETURNING *`,
      [tenantId, input.subject_type, input.subject_id, input.coverage_score, input.missing_roles, input.assessment_actor_id]
    );
    return rows[0];
  });
}

export async function recomputeCommitteeCoverage(tenantId: string, committeeId: string): Promise<BuyingCommittee> {
  return withTenant(tenantId, async (client) => {
    const { rows: members } = await client.query(
      `SELECT role FROM buying_committee_members WHERE tenant_id = $1 AND committee_id = $2 AND valid_to IS NULL`,
      [tenantId, committeeId]
    );
    const filledRoles = new Set(members.map((r: { role: string }) => r.role).filter(role => role !== 'unknown'));
    const missingRoles = ALL_ROLES.filter(r => r !== 'unknown' && !filledRoles.has(r));
    const coverage = ((ALL_ROLES.length - 1 - missingRoles.length) / (ALL_ROLES.length - 1)) * 100;
    const { rows } = await client.query(
      `UPDATE buying_committees SET coverage_score = $1, missing_roles = $2, last_assessed_at = NOW(), updated_at = NOW()
       WHERE tenant_id = $3 AND committee_id = $4 RETURNING *`,
      [coverage, missingRoles, tenantId, committeeId]
    );
    return rows[0];
  });
}

// Human-string helper for operator UI (D-25 BuyingCommitteePanel copy)
export function formatMissingRolesForOperator(missingRoles: string[]): string[] {
  const messages: string[] = [];
  if (missingRoles.includes('finance')) messages.push('No finance stakeholder engaged');
  if (missingRoles.includes('economic_buyer')) messages.push('Champion active, economic buyer absent');
  if (missingRoles.includes('technical_buyer')) messages.push('Technical buyer asked for security artifact');
  if (missingRoles.includes('legal')) messages.push('Legal review likely next blocker');
  for (const r of missingRoles) {
    if (!['finance','economic_buyer','technical_buyer','legal'].includes(r)) {
      messages.push(`Missing ${r.replace(/_/g, ' ')}`);
    }
  }
  return messages;
}

import { withTenant } from '../../db';
import { recomputeCommitteeCoverage } from './committee';

export class ConcurrentRoleChangeError extends Error { code = 'CONCURRENT_ROLE_CHANGE' as const; constructor() { super('Role change superseded by concurrent write'); } }

export interface BuyingCommitteeMember { /* per <interfaces> */ }

export async function listActiveMembers(tenantId: string, committeeId: string): Promise<BuyingCommitteeMember[]> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT * FROM buying_committee_members WHERE tenant_id = $1 AND committee_id = $2 AND valid_to IS NULL ORDER BY role`,
      [tenantId, committeeId]
    );
    return rows;
  });
}

export async function getMemberHistory(tenantId: string, committeeId: string, personId: string): Promise<BuyingCommitteeMember[]> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT * FROM buying_committee_members WHERE tenant_id = $1 AND committee_id = $2 AND person_id = $3 ORDER BY valid_from DESC`,
      [tenantId, committeeId, personId]
    );
    return rows;
  });
}

export async function addCommitteeMember(input: { tenant_id: string; committee_id: string; person_id: string; role: BuyingCommitteeMember['role']; actor_id: string; }): Promise<BuyingCommitteeMember> {
  return withTenant(input.tenant_id, async (client) => {
    const { rows } = await client.query(
      `INSERT INTO buying_committee_members (tenant_id, committee_id, person_id, role, valid_from, valid_to)
       VALUES ($1, $2, $3, $4, NOW(), NULL) RETURNING *`,
      [input.tenant_id, input.committee_id, input.person_id, input.role]
    );
    await client.query(
      `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
       VALUES ($1, $2, 'crm360.committee_member.add', 'buying_committee_member', $3, $4, NOW())`,
      [input.tenant_id, input.actor_id, rows[0].member_id, JSON.stringify({ committee_id: input.committee_id, role: input.role })]
    );
    await recomputeCommitteeCoverage(input.tenant_id, input.committee_id);
    return rows[0];
  });
}

export async function changeCommitteeMemberRole(input: { tenant_id: string; committee_id: string; person_id: string; new_role: BuyingCommitteeMember['role']; actor_id: string; }): Promise<BuyingCommitteeMember> {
  return withTenant(input.tenant_id, async (client) => {
    // Atomic close+insert (Pitfall 4)
    const closeRes = await client.query(
      `UPDATE buying_committee_members SET valid_to = NOW()
       WHERE tenant_id = $1 AND committee_id = $2 AND person_id = $3 AND valid_to IS NULL
       RETURNING member_id`,
      [input.tenant_id, input.committee_id, input.person_id]
    );
    if (closeRes.rowCount === 0) {
      throw new ConcurrentRoleChangeError();
    }
    const { rows } = await client.query(
      `INSERT INTO buying_committee_members (tenant_id, committee_id, person_id, role, valid_from, valid_to)
       VALUES ($1, $2, $3, $4, NOW(), NULL) RETURNING *`,
      [input.tenant_id, input.committee_id, input.person_id, input.new_role]
    );
    await client.query(
      `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
       VALUES ($1, $2, 'crm360.committee_member.role_change', 'buying_committee_member', $3, $4, NOW())`,
      [input.tenant_id, input.actor_id, rows[0].member_id,
       JSON.stringify({ committee_id: input.committee_id, new_role: input.new_role, prior_member_id: closeRes.rows[0].member_id })]
    );
    await recomputeCommitteeCoverage(input.tenant_id, input.committee_id);
    return rows[0];

function mapError(err: unknown): NextResponse {
  if (err instanceof TombstonedRecordError) return NextResponse.json({ error: 'tombstoned' }, { status: 410 });
  if (err instanceof ApprovalRequiredError || err instanceof NbaApprovalRequiredError) {
    return NextResponse.json({ error: 'approval_required', code: err.code }, { status: 422 });
  }
  if (err instanceof ConcurrentRoleChangeError) return NextResponse.json({ error: 'concurrent_role_change' }, { status: 409 });
  console.error(err);
  return NextResponse.json({ error: 'internal' }, { status: 500 });
}

// One handler function per endpoint — 10 total.
// Each exported as handleGET<Endpoint> / handlePATCH<Endpoint> etc.

export async function handleCustomer360Get(req: NextRequest, id: string): Promise<NextResponse> {
  try {
    const { tenantId } = await authenticateTenant(req);
    const record = await getCustomer360(tenantId, id);
    if (!record) return NextResponse.json({ error: 'not_found' }, { status: 404 });
    return NextResponse.json(record);
  } catch (err) { return mapError(err); }
}

export async function handleCustomer360Patch(req: NextRequest, id: string): Promise<NextResponse> {
  try {
    const { tenantId, actorId } = await authenticateTenant(req);
    const body = await req.json();
    const updated = await patchCustomer360(tenantId, id, body, { actor_id: actorId, approval_ref: body.approval_ref });
    await emitCrmEventEnvelope({
      source_event_ref: crypto.randomUUID(),
      tenant_id: tenantId,
      crm_event_name: 'crm.customer360.patch',
      occurred_at: new Date().toISOString(),
      canonical_identity_id: updated.canonical_identity_id,
      account_id: null,
      mutation_payload: { patch: body },
    });
    return NextResponse.json(updated);
  } catch (err) { return mapError(err); }
}

// Per D-25
import React, { useState, useEffect } from 'react';
import styles from './BuyingCommitteePanel.module.css';

const ROLES_ALL = ['champion','economic_buyer','technical_buyer','end_user','blocker','legal','finance'] as const;

export interface BuyingCommitteePanelProps {
  subjectType: 'opportunity' | 'account';
  subjectId: string;
  tenantId: string;
  onInviteRole?: (role: string) => void;
}

export function BuyingCommitteePanel({ subjectType, subjectId, tenantId, onInviteRole }: BuyingCommitteePanelProps) {
  const [committee, setCommittee] = useState<any>(null);
  const [members, setMembers] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    setLoading(true);
    fetch(`/api/v1/crm/committees?subject_type=${subjectType}&subject_id=${encodeURIComponent(subjectId)}`)
      .then(r => r.ok ? r.json() : null)
      .then(data => { if (data) { setCommittee(data.committee); setMembers(data.members ?? []); } })
      .finally(() => setLoading(false));
  }, [subjectType, subjectId]);

  if (loading) return <div className={styles.loading}>Loading committee...</div>;
  if (!committee) return <div className={styles.empty}>No buying committee recorded</div>;

  const filledRoles = new Set(members.map(m => m.role));
  const missingRolesFormatted = formatMissingRolesForOperator(committee.missing_roles);

  return (
    <section className={styles.panel} data-testid="buying-committee-panel">
      <header className={styles.header}>
        <h3>Buying committee</h3>
        <span className={styles.coverage}>coverage {committee.coverage_score.toFixed(0)}%</span>
      </header>
      <div className={styles.missingRoles}>
        {missingRolesFormatted.map((msg, i) => (
          <div key={i} className={styles.missingRoleAlert}>{msg}</div>
        ))}
      </div>
      <ul className={styles.memberList}>
        {members.map(m => (
          <li key={m.member_id} className={styles.member}>
            <span className={styles.role}>{m.role}</span>
            <span className={styles.influence}>influence {(m.influence_score ?? 0).toFixed(0)}</span>
            <span className={styles.engagement}>engagement {(m.engagement_score ?? 0).toFixed(0)}</span>
            {m.last_touch_at && <time>last touch {m.last_touch_at}</time>}
          </li>
        ))}
      </ul>
      <footer className={styles.actions}>
        {ROLES_ALL.filter(r => !filledRoles.has(r)).map(role => (
          <button key={role} onClick={() => onInviteRole?.(role)}>Invite {role.replace(/_/g, ' ')}</button>
        ))}
      </footer>
    </section>
  );
}

// Inline copy of formatMissingRolesForOperator from lib/markos/crm360/committees/committee.ts
// (server function not directly importable in client — mirror logic):
function formatMissingRolesForOperator(missingRoles: string[]): string[] {
  const messages: string[] = [];
  if (missingRoles.includes('finance')) messages.push('No finance stakeholder engaged');
  if (missingRoles.includes('economic_buyer')) messages.push('Champion active, economic buyer absent');
  if (missingRoles.includes('technical_buyer')) messages.push('Technical buyer asked for security artifact');
  if (missingRoles.includes('legal')) messages.push('Legal review likely next blocker');
  for (const r of missingRoles) {
    if (!['finance','economic_buyer','technical_buyer','legal'].includes(r)) {
      messages.push(`Missing ${r.replace(/_/g, ' ')}`);
    }
  }
  return messages;
}
