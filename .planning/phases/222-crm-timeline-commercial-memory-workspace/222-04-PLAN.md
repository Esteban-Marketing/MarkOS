// Per RD-08 jitter; Pitfall 2 thundering herd guard
export const ACTION_TYPE_TTL_HOURS: Readonly<Record<string, number>> = Object.freeze({
  send_followup: 72,
  propose_expansion: 168,
  flag_risk: 48,
  engage_committee_member: 120,
  schedule_meeting: 96,
  send_renewal_reminder: 240,
});
const DEFAULT_TTL_HOURS = 72;

export function computeExpiresAt(actionType: string, opts?: { jitter?: boolean }): string {
  const base = ACTION_TYPE_TTL_HOURS[actionType] ?? DEFAULT_TTL_HOURS;
  // +/-12h spread per RD-08 (range: base-12 .. base+12 capped at min base*0.5)
  const jitterHours = opts?.jitter ? (Math.random() * 24 - 12) : 0;
  const effective = Math.max(base + jitterHours, base * 0.5);
  return new Date(Date.now() + effective * 3600_000).toISOString();
}

import { withTenant, dbAllTenants } from '../../db';

export async function expireNbaRecords(tenantId: string | 'ALL'): Promise<{ expired: number }> {
  const client = tenantId === 'ALL' ? dbAllTenants() : await withTenant(tenantId, async (c) => c);
  const res = await client.query(
    `UPDATE nba_records SET status = 'expired'
     WHERE status = 'active' AND expires_at < NOW()
     ${tenantId === 'ALL' ? '' : 'AND tenant_id = $1'}
     RETURNING nba_id`,
    tenantId === 'ALL' ? [] : [tenantId]
  );
  return { expired: res.rowCount ?? 0 };
}

import { withTenant } from '../../db';
import { computeExpiresAt } from './expiry';

export const ACTION_TYPES_REQUIRING_APPROVAL: ReadonlyArray<string> = Object.freeze([
  'send_followup',
  'propose_expansion',
  'send_renewal_reminder',
  'draft_outreach',
]);

export class NbaApprovalRequiredError extends Error {
  code = 'NBA_APPROVAL_REQUIRED' as const;
  status = 422;
  constructor(public actionType: string) { super(`NBA action_type=${actionType} requires approval_ref to execute (D-28)`); }
}

export interface NbaRecord { /* per <interfaces> */ }
export interface NbaComputeInput { /* per <interfaces> */ }

export async function upsertNbaRecord(input: NbaComputeInput): Promise<NbaRecord> {
  if (input.evidence_refs.length === 0) throw new Error('NBA evidence_refs must be non-empty (no black-box NBA)');
  if (input.confidence < 0 || input.confidence > 1) throw new Error('confidence must be in [0,1]');
  return withTenant(input.tenant_id, async (client) => {
    // Blocker 5 fix: wrap INSERT + supersedence UPDATE in a transaction.
    // Order: INSERT new row first -> obtain real UUID -> UPDATE old row with real superseded_by FK.
    // Never write a placeholder UUID like 'pending-new-id' (data-integrity bug).
    await client.query('BEGIN');
    try {
      const expiresAt = computeExpiresAt(input.action_type, { jitter: true });
      // 1. INSERT new NBA row first to obtain real UUID
      const insertRes = await client.query(
        `INSERT INTO nba_records
          (tenant_id, subject_type, subject_id, action_type, rationale, evidence_refs,
           expires_at, confidence, approval_ref, status)
         VALUES ($1,$2,$3,$4,$5,$6,$7,$8,$9,'active') RETURNING *`,
        [input.tenant_id, input.subject_type, input.subject_id, input.action_type,
         input.rationale, input.evidence_refs, expiresAt, input.confidence, input.approval_ref ?? null]
      );
      const inserted = insertRes.rows[0];
      const newNbaId = inserted.nba_id;

      // 2. If supersedes, UPDATE old row with real UUID (single statement, no placeholder)
      if (input.supersedes_nba_id) {
        await client.query(
          `UPDATE nba_records SET status = 'superseded', superseded_by = $1
           WHERE nba_id = $2 AND tenant_id = $3 AND status = 'active'`,
          [newNbaId, input.supersedes_nba_id, input.tenant_id]
        );
      }

      // 3. Update customer_360_records.next_best_action_id if subject is customer_360
      if (input.subject_type === 'customer_360') {
        await client.query(
          `UPDATE customer_360_records SET next_best_action_id = $1, updated_at = NOW()
           WHERE customer_360_id = $2 AND tenant_id = $3`,
          [newNbaId, input.subject_id, input.tenant_id]
        );
      }

      await client.query('COMMIT');
      return inserted;
    } catch (err) {
      await client.query('ROLLBACK');
      // Blocker 5 invariant: on INSERT failure, old superseded_by row remains status='active'; no placeholder UUID ever persisted.
      throw err;
    }
  });
}

export async function getActiveNbaForSubject(tenantId: string, subjectType: NbaRecord['subject_type'], subjectId: string): Promise<NbaRecord | null> {
  return withTenant(tenantId, async (client) => {
    const res = await client.query(
      `SELECT * FROM nba_records WHERE tenant_id = $1 AND subject_type = $2 AND subject_id = $3 AND status = 'active'
       ORDER BY computed_at DESC LIMIT 1`,
      [tenantId, subjectType, subjectId]
    );
    return res.rows[0] ?? null;
  });
}

export async function listActiveNbaByOwner(tenantId: string, ownerUserId: string, opts?: { limit?: number }): Promise<NbaRecord[]> {
  return withTenant(tenantId, async (client) => {
    const res = await client.query(
      `SELECT n.* FROM nba_records n
       JOIN customer_360_records c ON c.customer_360_id = n.subject_id AND n.subject_type = 'customer_360' AND c.tenant_id = n.tenant_id
       WHERE n.tenant_id = $1 AND n.status = 'active' AND c.primary_owner_user_id = $2
       ORDER BY n.computed_at DESC LIMIT $3`,
      [tenantId, ownerUserId, opts?.limit ?? 50]
    );
    return res.rows;
  });
}

export async function markNbaDismissed(tenantId: string, nbaId: string, actorId: string): Promise<void> {
  await withTenant(tenantId, async (client) => {
    await client.query(
      `UPDATE nba_records SET status = 'dismissed'
       WHERE tenant_id = $1 AND nba_id = $2 AND status = 'active'`,
      [tenantId, nbaId]
    );
    await client.query(
      `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
       VALUES ($1, $2, 'crm360.nba.dismiss', 'nba_record', $3, $4, NOW())`,
      [tenantId, actorId, nbaId, JSON.stringify({})]
    );
  });
}

export async function markNbaExecuted(tenantId: string, nbaId: string, actorId: string, approvalRef: string | null): Promise<void> {
  await withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT action_type, approval_ref FROM nba_records WHERE tenant_id = $1 AND nba_id = $2`,
      [tenantId, nbaId]
    );
    if (rows.length === 0) throw new Error('NBA not found');
    const { action_type } = rows[0];
    if (ACTION_TYPES_REQUIRING_APPROVAL.includes(action_type) && !approvalRef) {
      throw new NbaApprovalRequiredError(action_type);
    }
    await client.query(
      `UPDATE nba_records SET status = 'executed', approval_ref = COALESCE($1, approval_ref)
       WHERE tenant_id = $2 AND nba_id = $3 AND status = 'active'`,
      [approvalRef, tenantId, nbaId]
    );
    await client.query(
      `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
       VALUES ($1, $2, 'crm360.nba.execute', 'nba_record', $3, $4, NOW())`,
      [tenantId, actorId, nbaId, JSON.stringify({ approval_ref: approvalRef })]
    );
  });
}

export async function markNbaSuperseded(tenantId: string, supersededNbaId: string, newNbaId: string): Promise<void> {

       WHERE tenant_id = $2 AND nba_id = $3 AND status = 'active'`,
      [newNbaId, tenantId, supersededNbaId]
    );
  });
}

function deriveNbaForCustomer(c360: { lifecycle_stage: string; engagement_score: number | null; risk_score: number | null }): { action_type: string; rationale: string; confidence: number; evidence_refs: string[] } | null {
  if (c360.risk_score !== null && c360.risk_score >= 0.7) {
    return { action_type: 'flag_risk', rationale: `Risk score ${c360.risk_score} exceeds 0.7 threshold`, confidence: c360.risk_score, evidence_refs: ['score-provenance-risk'] };
  }
  if (['customer','expansion','renewal'].includes(c360.lifecycle_stage) && c360.engagement_score !== null && c360.engagement_score >= 0.6) {
    return { action_type: 'propose_expansion', rationale: `Engagement ${c360.engagement_score} supports expansion outreach`, confidence: c360.engagement_score, evidence_refs: ['score-provenance-engagement'] };
  }
  if (['mql','sql'].includes(c360.lifecycle_stage)) {
    return { action_type: 'send_followup', rationale: `${c360.lifecycle_stage} requires follow-up`, confidence: 0.6, evidence_refs: ['score-provenance-engagement'] };
  }
  return null;
}

async function logAgentRun(tenantId: string, runType: string, payload: Record<string, unknown>): Promise<void> {
  await withTenant(tenantId, async (client) => {
    // A5 guard
    const { rows } = await client.query(`SELECT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'markos_agent_runs') AS has_runs`);
    if (rows[0]?.has_runs) {
      await client.query(
        `INSERT INTO markos_agent_runs (tenant_id, run_type, status, input, created_at)
         VALUES ($1, $2, 'completed', $3, NOW())`,
        [tenantId, runType, JSON.stringify(payload)]
      );
    } else {
      await client.query(
        `INSERT INTO markos_audit_log (tenant_id, actor_id, action, target_type, target_id, payload, occurred_at)
         VALUES ($1, 'system', 'crm360.nba.recompute_fallback', $2, 'agent_run_absent', $3, NOW())`,
        [tenantId, runType, JSON.stringify(payload)]
      );
    }
  });
}

export async function recomputeNbaForCustomer(tenantId: string, customer360Id: string, triggerReason: string): Promise<NbaRecord | null> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT lifecycle_stage, engagement_score, risk_score FROM customer_360_records
       WHERE tenant_id = $1 AND customer_360_id = $2 AND tombstoned = FALSE`,
      [tenantId, customer360Id]
    );
    if (rows.length === 0) return null;
    const nbaInput = deriveNbaForCustomer(rows[0]);
    if (!nbaInput) return null;
    const existing = await getActiveNbaForSubject(tenantId, 'customer_360', customer360Id);
    const result = await upsertNbaRecord({
      tenant_id: tenantId,
      subject_type: 'customer_360',
      subject_id: customer360Id,
      action_type: nbaInput.action_type,
      rationale: nbaInput.rationale,
      evidence_refs: nbaInput.evidence_refs,
      confidence: nbaInput.confidence,
      supersedes_nba_id: existing?.nba_id,
    });
    await logAgentRun(tenantId, 'nba_recompute_customer', { customer_360_id: customer360Id, trigger_reason: triggerReason, nba_id: result.nba_id });
    return result;
  });
}

export async function recomputeNbaForOpportunity(tenantId: string, opportunityId: string, triggerReason: string): Promise<NbaRecord | null> {
  // Symmetric logic for opportunities; use health + evidence_gap_score
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT health, evidence_gap_score, next_required_action FROM opportunities
       WHERE tenant_id = $1 AND opportunity_id = $2`,
      [tenantId, opportunityId]
    );
    if (rows.length === 0) return null;
    const o = rows[0];
    let input: { action_type: string; rationale: string; confidence: number; evidence_refs: string[] } | null = null;
    if (o.health === 'at_risk' || o.health === 'stalled') {
      input = { action_type: 'flag_risk', rationale: `Opportunity health=${o.health}`, confidence: 0.8, evidence_refs: ['opportunity-health'] };
    } else if ((o.evidence_gap_score ?? 0) > 0.5) {
      input = { action_type: 'engage_committee_member', rationale: `Evidence gap score ${o.evidence_gap_score}`, confidence: o.evidence_gap_score, evidence_refs: ['opportunity-evidence-gap'] };
    }
    if (!input) return null;
    const existing = await getActiveNbaForSubject(tenantId, 'opportunity', opportunityId);
    const result = await upsertNbaRecord({
      tenant_id: tenantId,
      subject_type: 'opportunity',
      subject_id: opportunityId,
      action_type: input.action_type,
      rationale: input.rationale,
      evidence_refs: input.evidence_refs,
      confidence: input.confidence,
      supersedes_nba_id: existing?.nba_id,
    });
    await logAgentRun(tenantId, 'nba_recompute_opportunity', { opportunity_id: opportunityId, trigger_reason: triggerReason, nba_id: result.nba_id });
    return result;
  });
}

export async function scheduleDailyRecompute(tenantId: string): Promise<{ scheduled: number }> {
  return withTenant(tenantId, async (client) => {
    const { rows } = await client.query(
      `SELECT customer_360_id FROM customer_360_records
       WHERE tenant_id = $1 AND tombstoned = FALSE
         AND (next_best_action_id IS NULL OR EXISTS (
           SELECT 1 FROM nba_records n WHERE n.nba_id = customer_360_records.next_best_action_id AND n.expires_at < NOW() + INTERVAL '24 hours'
         ))`,
      [tenantId]
    );
    let scheduled = 0;
    for (const r of rows) {
      await recomputeNbaForCustomer(tenantId, r.customer_360_id, 'daily_cron');
      scheduled++;
    }
    return { scheduled };
  });
}
