// Extended buildCrmTimeline — additive opts arg; legacy two-arg call still works
export function buildCrmTimeline(
  entityId: string,
  activities: Array<CrmActivityRow>,
  opts?: { highSignalOnly?: boolean; source_domain?: string; commercial_signal?: string }
): CrmTimelineEventExtended[] {
  let projected = applyHighSignalFilter(activities, opts?.highSignalOnly);
  if (opts?.source_domain) {
    projected = projected.filter(r =>
      r.source_domain === opts.source_domain ||
      (r.source_domain === null && ACTIVITY_FAMILY_TO_SOURCE_DOMAIN[r.activity_family] === opts.source_domain)
    );
  }
  if (opts?.commercial_signal) {
    projected = projected.filter(r => r.commercial_signal === opts.commercial_signal);
  }
  return projected.map(r => ({
    event_id: r.event_id,
    occurred_at: r.occurred_at,
    activity_family: r.activity_family,
    source_domain: r.source_domain ?? ACTIVITY_FAMILY_TO_SOURCE_DOMAIN[r.activity_family] ?? null,
    commercial_signal: r.commercial_signal ?? null,
    actor_type: r.actor_type ?? null,
    actor_id: r.actor_id ?? null,
    opportunity_id: r.opportunity_id ?? null,
    sentiment: r.sentiment ?? null,
    thread_id: r.thread_id ?? null,
    entity_id: r.entity_id,
    tenant_id: r.tenant_id,
    payload: r.payload,
  }));
}

import { withTenant } from '../../db';
import { buildCrmTimeline, type CrmTimelineEventExtended } from '../../crm/timeline';

export interface TimelineQueryParams { /* per <interfaces> */ }

export async function handleTimelineQuery(params: TimelineQueryParams): Promise<CrmTimelineEventExtended[]> {
  return withTenant(params.tenant_id, async (client) => {
    const conditions: string[] = ['tenant_id = $1', 'entity_id = $2'];
    const values: unknown[] = [params.tenant_id, params.subject_id];
    if (params.source_domain) { conditions.push(`source_domain = $${values.length + 1}`); values.push(params.source_domain); }
    if (params.commercial_signal) { conditions.push(`commercial_signal = $${values.length + 1}`); values.push(params.commercial_signal); }
    if (params.opportunity_id) { conditions.push(`opportunity_id = $${values.length + 1}`); values.push(params.opportunity_id); }
    if (params.before_at) { conditions.push(`occurred_at < $${values.length + 1}`); values.push(params.before_at); }
    const limit = Math.min(params.limit ?? 100, 500);
    values.push(limit);
    const sql = `SELECT * FROM crm_activity_ledger WHERE ${conditions.join(' AND ')}
                 ORDER BY occurred_at DESC LIMIT $${values.length}`;
    const res = await client.query(sql, values);
    return buildCrmTimeline(params.subject_id, res.rows, { highSignalOnly: true });
  });
}

// Per D-24 (default record detail view), D-25 (source_domain + commercial_signal filters)
import React, { useState, useEffect } from 'react';
import styles from './TimelineDetailView.module.css';

const SOURCE_DOMAINS = ['website','email','messaging','meeting','crm','billing','support','product','social','research','agent'] as const;
const COMMERCIAL_SIGNALS = ['interest','risk','expansion','renewal','support','pricing','silence'] as const;

export interface TimelineDetailViewProps {
  subjectId: string;
  tenantId: string;
  initialFilters?: { source_domain?: string; commercial_signal?: string };
}

export function TimelineDetailView({ subjectId, tenantId, initialFilters }: TimelineDetailViewProps) {
  const [sourceDomain, setSourceDomain] = useState<string | null>(initialFilters?.source_domain ?? null);
  const [commercialSignal, setCommercialSignal] = useState<string | null>(initialFilters?.commercial_signal ?? null);
  const [events, setEvents] = useState<unknown[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    setLoading(true);
    const qs = new URLSearchParams({ subject_id: subjectId });
    if (sourceDomain) qs.set('source_domain', sourceDomain);
    if (commercialSignal) qs.set('commercial_signal', commercialSignal);
    fetch(`/api/v1/crm/timeline?${qs.toString()}`)
      .then(r => r.ok ? r.json() : Promise.reject(`HTTP ${r.status}`))
      .then(data => { setEvents(data); setError(null); })
      .catch(err => setError(String(err)))
      .finally(() => setLoading(false));
  }, [subjectId, sourceDomain, commercialSignal]);

  return (
    <div className={styles.root} data-testid="timeline-detail-view">
      <header className={styles.header}>
        <h2>Timeline</h2>
        <div className={styles.filters}>
          <fieldset>
            <legend>Source</legend>
            {SOURCE_DOMAINS.map(d => (
              <button key={d} className={sourceDomain === d ? styles.chipActive : styles.chip}
                      onClick={() => setSourceDomain(sourceDomain === d ? null : d)}>{d}</button>
            ))}
          </fieldset>
          <fieldset>
            <legend>Signal</legend>
            {COMMERCIAL_SIGNALS.map(s => (
              <button key={s} className={commercialSignal === s ? styles.chipActive : styles.chip}
                      onClick={() => setCommercialSignal(commercialSignal === s ? null : s)}>{s}</button>
            ))}
          </fieldset>
        </div>
      </header>
      <main className={styles.events}>
        {loading && <div className={styles.loading}>Loading...</div>}
        {error && <div className={styles.error} role="alert">{error}</div>}
        {!loading && !error && events.length === 0 && <div className={styles.empty}>No events match filter</div>}
        {!loading && !error && events.length > 0 && (
          <ul className={styles.eventList}>
            {events.map((e: any) => (
              <li key={e.event_id} className={styles.event}>
                <time>{e.occurred_at}</time>
                <span className={styles.sourceBadge}>{e.source_domain ?? e.activity_family}</span>
                {e.commercial_signal && <span className={`${styles.signalBadge} ${styles[e.commercial_signal]}`}>{e.commercial_signal}</span>}
                {e.actor_type && <span className={styles.actorBadge}>{e.actor_type}:{e.actor_id}</span>}
              </li>
            ))}
          </ul>
        )}
      </main>
    </div>
  );
}
