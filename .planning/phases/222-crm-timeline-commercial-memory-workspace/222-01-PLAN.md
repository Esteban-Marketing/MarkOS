  { pattern: /may not exist at execute time/, reason: 'D-38: hard prerequisite', replacement: 'assertUpstreamReady throws' },

const NO_X_COMMENT_REGEX = /<!--\s*NO-X[\s\S]*?-->/g;
// Paired NO-X marker block: <!-- NO-X: rationale --> ... <!-- /NO-X -->
const NO_X_BLOCK_REGEX = /<!--\s*NO-X[\s\S]*?-->[\s\S]*?<!--\s*\/NO-X\s*-->/g;
const FENCED_CODE_BLOCK_REGEX = /```[\s\S]*?```/g;

export function scanFileForForbiddenPatterns(filePath: string): Array<{ pattern: string; line: number; context: string; reason: string }> {
  const src = readFileSync(filePath, 'utf8');
  // Strip NO-X comment blocks AND fenced code blocks (allowed mentions in either) before scanning.
  // Rationale: fenced code blocks contain implementation samples (incl. detector source itself, doctrinal NO-X examples, regex literals);
  //            plan-level prose + frontmatter is the contract surface. Only flag forbidden patterns there.
  // Strip paired NO-X blocks first (greediest), then any standalone NO-X comments, then fenced code blocks.
  const stripped = src
    .replace(NO_X_BLOCK_REGEX, '')
    .replace(NO_X_COMMENT_REGEX, '')
    .replace(FENCED_CODE_BLOCK_REGEX, '');
  const lines = stripped.split('\n');
  const findings: Array<{ pattern: string; line: number; context: string; reason: string }> = [];
  for (let i = 0; i < lines.length; i++) {
    for (const { pattern, reason } of FORBIDDEN_PATTERNS) {
      if (pattern.test(lines[i])) {
        findings.push({ pattern: String(pattern), line: i + 1, context: lines[i].trim().slice(0, 200), reason });
      }
    }
  }
  return findings;
}

export async function scanPhasePlans(phaseGlob: string): Promise<Array<{ file: string; findings: ReturnType<typeof scanFileForForbiddenPatterns> }>> {
  const files = await glob(phaseGlob);
  return files.map(file => ({ file, findings: scanFileForForbiddenPatterns(file) }));
}
