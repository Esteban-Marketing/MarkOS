       ```typescript
       import { getIdentityProfileByContactId, getIdentityProfile } from '../identity/profile';
       import { getConsentState } from '../consent/state';
       import { getTraitSnapshot } from '../traits/snapshot';

       export async function getProfileForContact(tenantId: string, contactId: string): Promise<CDPProjection | null> {
         if (!tenantId) throw new Error('CDP_TENANT_REQUIRED');
         const profile = await getIdentityProfileByContactId(tenantId, contactId);
         if (!profile) return null;
         return projectFromIdentityProfile(tenantId, profile);
       }
       export async function getProfileById(tenantId: string, profileId: string): Promise<CDPProjection | null> {
         if (!tenantId) throw new Error('CDP_TENANT_REQUIRED');
         const profile = await getIdentityProfile(tenantId, profileId);
         if (!profile) return null;
         return projectFromIdentityProfile(tenantId, profile);
       }
       async function projectFromIdentityProfile(tenantId: string, profile: IdentityProfile): Promise<CDPProjection> {
         if (profile.tombstoned) {
           return {
             canonical_identity_id: profile.profile_id,
             profile_type: profile.profile_type,
             lifecycle_state: profile.lifecycle_state,
             mode: profile.mode,
             consent_state: null,
             traits: [],
             tombstoned: true,
           };
         }
         const [consent, traits] = await Promise.all([
           getConsentState(tenantId, profile.profile_id),
           getTraitSnapshot(tenantId, profile.profile_id),
         ]);
         return {
           canonical_identity_id: profile.profile_id,
           profile_type: profile.profile_type,
           lifecycle_state: profile.lifecycle_state,
           mode: profile.mode,
           consent_state: consent ? projectConsentSummary(consent) : null,
           traits: traits.map(projectTraitSummary),
           tombstoned: false,
         };
       }
       // projectConsentSummary redacts deletion_evidence_ref (operator-only)
       // projectTraitSummary preserves trait_name, value, computed_at, source_event_refs, freshness_mode, confidence
       ```
