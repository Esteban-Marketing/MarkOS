       ```typescript
       export async function cascadeDeleteProfileData(tenantId: string, profileId: string) {
         if (!tenantId || !profileId) throw new Error('CDP_TENANT_REQUIRED');
         // Events: partition-scoped DELETE (Postgres handles partition pruning via WHERE on partition key)
         const { count: events_purged } = await supabase.from('cdp_events')
           .delete({ count: 'exact' })
           .eq('tenant_id', tenantId)
           .eq('profile_id', profileId);
         // Traits: ON DELETE CASCADE from FK auto-fires when profile is eventually deleted; but since we TOMBSTONE (not delete),
         // we must explicitly delete trait rows here.
         const { count: traits_purged } = await supabase.from('cdp_trait_snapshots')
           .delete({ count: 'exact' })
           .eq('tenant_id', tenantId)
           .eq('profile_id', profileId);
         // Audience memberships: explicit delete (same reason as traits — tombstone != DELETE on profile row)
         const { count: memberships_purged } = await supabase.from('cdp_audience_snapshot_memberships')
           .delete({ count: 'exact' })
           .eq('tenant_id', tenantId)
           .eq('profile_id', profileId);
         return { events_purged: events_purged ?? 0, traits_purged: traits_purged ?? 0, memberships_purged: memberships_purged ?? 0 };
       }
       ```

       ```typescript
       export interface ReconciliationResult {
         tenant_id: string;
         window_hours: number;
         crm_activity_count: number;
         cdp_events_count: number;
         drift_count: number;
         drift_ratio: number;
         threshold: number;
         breach: boolean;
         tasks_emitted: number;
       }
       export async function reconcileDualWriteDrift(scope: { tenant_id: string; window_hours?: number }): Promise<ReconciliationResult> {
         const threshold = Number(process.env.CDP_DUAL_WRITE_DRIFT_THRESHOLD ?? '0.001');
         const windowHours = scope.window_hours ?? 24;
         const since = new Date(Date.now() - windowHours * 3600_000).toISOString();
         // Count crm_activity rows in window
         const crmCount = await countCrmActivityRows(scope.tenant_id, since);
         // Count cdp_events rows in window
         const cdpCount = await countCdpEventsRows(scope.tenant_id, since);
         const driftCount = Math.abs(crmCount - cdpCount);
         const driftRatio = crmCount === 0 ? 0 : driftCount / crmCount;
         const breach = driftRatio >= threshold;
         let tasksEmitted = 0;
         if (breach) {
           await writeAuditLog({
             tenant_id: scope.tenant_id,
             event_type: 'cdp_dual_write_drift_detected',
             payload: { crm_activity_count: crmCount, cdp_events_count: cdpCount, drift_count: driftCount, drift_ratio: driftRatio, threshold, window_hours: windowHours },
           });
       ```
