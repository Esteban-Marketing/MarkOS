       export const ALLOWED_OPERATORS = new Set([
         'if', '==', '!=', '<', '<=', '>', '>=',
         'in', 'and', 'or', 'not',
         '+', '-', '*', '/',
         'var', 'missing', 'missing_some',

         readonly code = 'CDP_DSL_OPERATOR_REJECTED';
         readonly operator: string;
         constructor(operator: string) { super(`JSON Logic operator not allowed: ${operator}`); this.operator = operator; }
       }
       export function validateLogicJson(node: unknown): void {
         if (node === null || typeof node !== 'object') return;
         if (Array.isArray(node)) { for (const child of node) validateLogicJson(child); return; }
         const keys = Object.keys(node as Record<string, unknown>);
         // JSON Logic AST: a single-key object whose key is the operator.
         if (keys.length !== 1) {
           // Not a strict operator node; still walk children.
           for (const k of keys) validateLogicJson((node as Record<string, unknown>)[k]);
           return;
         }
         const op = keys[0];
         if (!ALLOWED_OPERATORS.has(op)) throw new DslOperatorRejected(op);
         validateLogicJson((node as Record<string, unknown>)[op]);
       }
