       -- Single-writer enforcement on outboundConsentRecords during the P221 shim window.

       CREATE OR REPLACE FUNCTION cdp_consent_single_writer_check() RETURNS trigger AS $$
       DECLARE
         writer_source TEXT;
       BEGIN
         BEGIN
           writer_source := current_setting('app.consent_writer_source', true);
         EXCEPTION WHEN others THEN
           writer_source := NULL;
         END;
         IF writer_source IS DISTINCT FROM 'cdp_setConsentState' THEN

         END IF;
         RETURN NEW;
       END;
