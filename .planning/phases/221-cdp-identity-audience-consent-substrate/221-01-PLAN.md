       // Heuristic: a phase has landed if .planning/phases/{NNN}-*/SUMMARY.md exists OR .planning/STATE.md marks it complete.
       export async function assertUpstreamReady({ phase }: { phase: number }): Promise<void> {
         const fs = require('node:fs');
         const path = require('node:path');
         const phasesDir = path.resolve(__dirname, '../../../../.planning/phases');
         const missing: string[] = [];
         for (const upstream of REQUIRED_UPSTREAM) {
           const matches = fs.readdirSync(phasesDir).filter((d: string) => d.startsWith(`${upstream}-`));
           if (matches.length === 0) { missing.push(`P${upstream}`); continue; }
           const summary = path.join(phasesDir, matches[0], `${upstream}-SUMMARY.md`);
           if (!fs.existsSync(summary)) missing.push(`P${upstream}`);
         }
         if (missing.length > 0) throw new UpstreamPhaseNotLandedError(missing);

         const fs = require('node:fs');
         const path = require('node:path');
         const violations: Array<{ pattern: string; file: string; line: number; snippet: string }> = [];
         for (const rel of paths) {
           const abs = path.resolve(rootDir, rel);
           if (!fs.existsSync(abs)) continue;
           const lines = fs.readFileSync(abs, 'utf8').split('\n');
