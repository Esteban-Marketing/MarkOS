    ```js
    // Phase 204.1 Plan 01 â€” D-01 CLI audit trail primitive.
    //
    // Contract: dispatch-layer wrapper records {command, args_redacted, profile,
    // exit_code, duration_ms, cli_version} via POST /api/tenant/cli-audit AFTER
    // every operator subcommand completes (success OR failure). Failures NEVER
    // change exit code (D-01 line 39 + D-12 lines 108-110).
    //
    // CLI-side ONLY â€” server-side audit_log/markos_audit emission stays at the
    // /api/tenant/* endpoint layer (lib/markos/cli/*.cjs uses enqueueAuditStaging).

    const { request } = require('./http.cjs');

    const SECRET_FLAG_NAMES = new Set(['password', 'token', 'key', '--password', '--token', '--key']);
    const SECRET_POSITIONAL_TRIGGERS = new Set(['--password', '--token', '--key']);
    const ENV_PUT_TRIGGER = new Set(['put', 'set']);  // `markos env put|set KEY VALUE`
    const REDACTED = '[REDACTED]';

    function redactArgs(parsed) {
      const out = {};
      for (const [k, v] of Object.entries(parsed)) {
        if (SECRET_FLAG_NAMES.has(k)) {
          out[k] = (v == null || v === false || v === '') ? v : REDACTED;
          continue;
        }
        if (k === 'positional' && Array.isArray(v)) {
          // Walk positional: if a value follows --token/--key/--password OR
          // follows env-write trigger ('put'|'set' KEY <value>), redact value.
          // Heuristic: redact every 3rd positional after env put/set trigger.
          out.positional = redactPositional(v);
          continue;
        }
        out[k] = v;
      }
      return out;
    }

    function redactPositional(arr) {
      const out = [];
      for (let i = 0; i < arr.length; i++) {
        const tok = arr[i];
        out.push(tok);
        if (ENV_PUT_TRIGGER.has(tok) && i + 2 < arr.length) {
          out.push(arr[++i]);            // KEY (kept)
          out.push(REDACTED); ++i;       // VALUE (redacted)
        }
      }
      return out;
    }

    async function recordCommandAudit({command, parsed, profile, exit_code, duration_ms, cli_version, debug = false}) {
      const args_redacted = redactArgs(parsed || {});
      const body = { command, args_redacted, profile: profile || null, exit_code, duration_ms, cli_version: cli_version || 'unknown' };
      try {
        const res = await request({ method: 'POST', path: '/api/tenant/cli-audit', body, profile, timeoutMs: 5000 });
        if (res && res.status === 202) return { recorded: true, status: 202 };
        if (res && res.status === 404) {
          if (debug) process.stderr.write('[audit] endpoint not yet available â€” skipping (pending P204 server-side wire-up)\n');
          return { recorded: false, reason: 'endpoint_not_available' };
        }
        if (debug) process.stderr.write(`[audit] unexpected status ${res && res.status} â€” skipping\n`);
        return { recorded: false, reason: `unexpected_status_${res && res.status}` };
      } catch (err) {
        if (debug) process.stderr.write(`[audit] transport error â€” skipping (${err && err.message})\n`);
        return { recorded: false, reason: 'transport_error' };
      }
    }

    module.exports = { recordCommandAudit, redactArgs };
    ```
