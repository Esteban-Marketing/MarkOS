    ```js
    'use strict';

    // Phase 204.1 Plan 02 Task 1 — D-02 spinner primitive.
    //
    // Renders to stderr (default) with locale-aware frames + suppression matrix.
    // Zero deps (D-08 zero-dep posture from P204).

    const { shouldUseJson, shouldUseColor } = require('./output.cjs');

    const FRAMES_UTF8 = ['⠋','⠙','⠹','⠸','⠼','⠴','⠦','⠧','⠇','⠏'];
    const FRAMES_ASCII = ['\\','|','/','-'];
    const INTERVAL_UTF8_MS = 80;
    const INTERVAL_ASCII_MS = 100;
    const CLEAR_LINE = '\r\x1b[K';

    function isUtf8Locale(env = process.env) {
      const sample = env.LC_ALL || env.LANG || '';
      if (!sample) return true; // assume UTF-8 in absence of locale (modern terminals)
      return /UTF-?8/i.test(sample);
    }

    function createSpinner({ label = '', stream = 'stderr', opts = {} } = {}) {
      const out = stream === 'stdout' ? process.stdout : process.stderr;
      const utf8 = isUtf8Locale();
      const frames = utf8 ? FRAMES_UTF8 : FRAMES_ASCII;
      const interval = utf8 ? INTERVAL_UTF8_MS : INTERVAL_ASCII_MS;

      // Suppression matrix (UI-SPEC §Suppression Matrix).
      const suppress = !out.isTTY || shouldUseJson(opts) || !shouldUseColor(opts);

      let i = 0;
      let timer = null;

      function render() {
        if (suppress) return;
        out.write(CLEAR_LINE + frames[i % frames.length] + ' ' + label);
        i++;
      }

      function tick() {
        // Manual tick (used for first paint or in tests). Auto-tick happens via timer.
        render();
      }

      function start() {
        if (suppress) return;
        render();
        timer = setInterval(render, interval);
        if (timer.unref) timer.unref();
      }

      function stop() {
        if (timer) { clearInterval(timer); timer = null; }
        if (!suppress) out.write(CLEAR_LINE);
      }

      // Auto-start on construction (matches caller idiom: const sp = createSpinner(...); ... sp.stop();).
      start();

      return { tick, stop };
    }

    module.exports = { createSpinner, isUtf8Locale, FRAMES_UTF8, FRAMES_ASCII };
    ```
