```typescript
import { createHmac, randomUUID } from 'node:crypto';

export function generateShareToken(tenantId: string, signingKey: string): string {
  const nonce = randomUUID();
  const payload = `${tenantId.slice(0, 8)}:${nonce}`;
  const sig = createHmac('sha256', signingKey).update(payload).digest('hex').slice(0, 16);
  return `${payload}:${sig}`;
}

export function verifyShareToken(token: string, signingKey: string): boolean {
  const parts = token.split(':');
  if (parts.length !== 3) return false;
  const [prefix, nonce, sig] = parts;
  const expected = createHmac('sha256', signingKey).update(`${prefix}:${nonce}`).digest('hex').slice(0, 16);
  // Constant-time compare via Buffer
  return sig.length === expected.length &&
    Buffer.from(sig).every((b, i) => b === Buffer.from(expected)[i]);
}
```

       ```typescript
       export function verifyShareToken(token: string, signing_key: string): { valid: boolean; tenant_prefix?: string; nonce?: string } {
         const parts = token.split(':');
         if (parts.length !== 3) return { valid: false };
         const [prefix, nonce, sig] = parts;
         if (!prefix || !nonce || !sig || sig.length !== 16) return { valid: false };
         const expected = createHmac('sha256', signing_key).update(`${prefix}:${nonce}`).digest('hex').slice(0, 16);
         try {
           const sigBuf = Buffer.from(sig, 'utf-8');
           const expBuf = Buffer.from(expected, 'utf-8');
           if (sigBuf.length !== expBuf.length) return { valid: false };
           const match = timingSafeEqual(sigBuf, expBuf);  // constant-time compare via Node built-in
       ```
