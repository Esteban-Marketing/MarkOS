export function resolveApprovalClass(artifactKind: SalesArtifactKind): ApprovalClass {
  const INTERNAL_ONLY = new Set(['battlecard', 'objection_library', 'internal_deal_brief', 'deal_health_signal']);
  return INTERNAL_ONLY.has(artifactKind) ? 'internal_auto' : 'customer_facing_required';
}

export async function gateApproval(
  artifactKind: SalesArtifactKind,
  mutationPayload: unknown,
  actorId: string,
): Promise<{ approved: boolean; approval_ref: string | null }> {
  const cls = resolveApprovalClass(artifactKind);
  if (cls === 'internal_auto') return { approved: true, approval_ref: null };
  const pkg = await createApprovalPackage({ kind: artifactKind, payload: mutationPayload, requested_by: actorId });
  return { approved: false, approval_ref: pkg.approval_id };
}

       export class QuoteImmutabilityError extends Error {
         constructor(public readonly quote_id: string, public readonly current_status: QuoteStatus, public readonly attempted_fields: string[]) {
           super(`Quote ${quote_id} status='${current_status}'; cannot modify fields: ${attempted_fields.join(', ')} (D-21 Quote-as-Snapshot immutability)`);
           this.name = 'QuoteImmutabilityError';
         }
       }
       export class QuoteStateError extends Error { /* expected status mismatch */ }
       export class QuoteContentClassifierError extends Error { /* unbound pricing var */ }
       
       const FROZEN_FIELDS = ['pricing_recommendation_snapshot', 'pricing_recommendation_id', 'snapshot_at'] as const;
       
       export function assertQuoteMutable(quote: { quote_id: string; status: QuoteStatus }, intendedFields: string[]): void {
         if (IMMUTABLE_QUOTE_STATUSES.includes(quote.status)) {
           const frozenTouched = intendedFields.filter(f => FROZEN_FIELDS.includes(f as any));

         draft: ['pending_approval', 'expired'],
         pending_approval: ['approved', 'draft'],  // approver may send back to draft for edits
         approved: ['sent', 'draft'],  // operator can also revert to draft if approval revoked
         sent: ['accepted', 'rejected', 'expired'],
         accepted: [],  // terminal
         rejected: [],  // terminal
         expired: [],  // terminal
       };

         constructor(public readonly from: ProposalStatus, public readonly to: ProposalStatus) {
           super(`Illegal proposal transition: ${from} → ${to} (allowed: ${PROPOSAL_TRANSITIONS[from].join(', ') || 'none — terminal state'})`);
           this.name = 'IllegalProposalTransitionError';
         }
       }

       }

         artifactKind: SalesArtifactKind,
         mutationPayload: unknown,
         actorId: string,

         const baseClass = resolveApprovalClass(artifactKind);

           ? String((mutationPayload as { tenant_id?: string }).tenant_id ?? '')
           : '';
         
         if (effectiveClass === 'internal_auto') {
           if (tenant_id) {
             await emitSalesAuditLog({
               tenant_id,
               event_type: 'approval.auto_approved',
               actor_user_id: actorId,
               target_kind: artifactKind,
               target_id: typeof mutationPayload === 'object' && mutationPayload && 'id' in mutationPayload ? String((mutationPayload as { id?: string }).id ?? '') : '',
               payload: { artifactKind, baseClass, effectiveClass, force_customer_facing_applied: false },
             });
           }

         }
         
         // Customer-facing path: route through P105
         const pkg = await createApprovalPackage({
           kind: artifactKind,
           payload: mutationPayload,
           requested_by: actorId,
         });

             payload: { artifactKind, baseClass, effectiveClass, force_customer_facing_applied: force_applied, approval_id: pkg.approval_id },
           });
         }
         return { approved: false, approval_ref: pkg.approval_id, class: 'customer_facing_required', force_customer_facing_applied: force_applied };
       }
