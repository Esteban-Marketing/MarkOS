       export async function auditDealBriefClaims(input: { tenant_id: string; clauses: NarrativeClause[]; pricing_context_id?: string | null }): Promise<{ passed: boolean; unbound_count: number; warnings: string[] }> {
         const unbound: { clause_id: string; reason: string }[] = [];
         const warnings: string[] = [];
         
         for (const clause of input.clauses) {
           // Check pricing references
           const hasPricingMention = /\{\{[a-z_]*price|monthly|annual|cost|MARKOS_PRICING/i.test(clause.content);
           if (hasPricingMention) {
             const isPlaceholder = clause.content.includes('{{MARKOS_PRICING_ENGINE_PENDING}}');
             const hasPricingRef = clause.pricing_refs.length > 0 || input.pricing_context_id;
             if (!isPlaceholder && !hasPricingRef) {
               unbound.push({ clause_id: clause.clause_id, reason: 'unbound_pricing_variable' });
             }
           }
           
           // Check claim references for freshness
           if (clause.claim_refs.length > 0) {
             const freshness = await getClaimFreshness(input.tenant_id, clause.claim_refs);
             const hasStale = freshness.some(f => f.status === 'stale' || f.status === 'retired');
             if (hasStale) warnings.push(`claim_stale_in_${clause.clause_id}`);
           }
         }
         
         if (unbound.length > 0) throw new DealBriefClaimAuditError(unbound);
         return { passed: true, unbound_count: 0, warnings };
       }
