    'use strict';

    function normalizeToken(value) {
      return String(value || '').trim();
    }

    /**
     * Apply discipline and/or audience_tags filters to a set of audit entries.
     * Both filters use AND semantics when supplied together.
     * Empty or absent filter returns all entries unchanged.
     *
     * @param {object[]} entries  Array of audit-store entries (already tenant-scoped)
     * @param {object}   [filter]
     * @param {string}   [filter.discipline]    Exact match (case-insensitive)
     * @param {string[]} [filter.audience_tags] AND match — all tags must be present in entry.audience
     * @returns {object[]}
     */
    function applyFilter(entries, filter) {
      const safeEntries = Array.isArray(entries) ? entries : [];
      const safeFilter = filter && typeof filter === 'object' ? filter : {};

      const { discipline, audience_tags } = safeFilter;

      let results = safeEntries;

      if (discipline && normalizeToken(discipline)) {
        const normDiscipline = normalizeToken(discipline).toLowerCase();
        results = results.filter(
          (entry) => String((entry && entry.discipline) || '').trim().toLowerCase() === normDiscipline
        );
      }

      if (Array.isArray(audience_tags) && audience_tags.length > 0) {
        const normTags = audience_tags.map((t) => normalizeToken(t)).filter((t) => t.length > 0);
        if (normTags.length > 0) {
          results = results.filter((entry) => {
            const entryAudience = Array.isArray(entry && entry.audience) ? entry.audience : [];
            // AND: every required tag must be present in entry.audience (exact string match)
            return normTags.every((tag) => entryAudience.includes(tag));
          });
        }
      }

      return results;
    }

    module.exports = { applyFilter };
    ```

    Note: discipline filter is case-insensitive (`.toLowerCase()` on both sides). Audience tag filter is case-sensitive exact match (`.includes(tag)` with no lowercase). This matches the existing `audience-schema.cjs` canonical tag format where prefix is uppercase (ICP:, SEGMENT:, etc.).

    /**
     * Allowed roles for vault artifact retrieval (Phase 86+ retrieval layer).
     * Extends Phase 85's ALLOWED_VISIBILITY_ROLES (ingestion-adjacent) with 'agent' role.
     * Note: 'audit-viewer' is intentionally excluded — read-only audit access ≠ retrieval.
     */
    const ALLOWED_RETRIEVAL_ROLES = new Set(['operator', 'admin', 'agent']);

    /**
     * Check whether a claims object has sufficient tenant + role authority
     * to perform vault artifact retrieval.
     *
     * Mirrors checkVisibilityScope shape but uses ALLOWED_RETRIEVAL_ROLES.
     *
     * @param {object} claims           { tenantId: string, role: string }
     * @param {object} resourceContext  { tenantId: string }
     * @returns {{ allowed: boolean, code: string|null, reason: string|null }}
     */
    function checkRetrievalScope(claims, resourceContext) {
      const claimTenantId = String((claims && claims.tenantId) || '').trim();
      const claimRole = String((claims && claims.role) || '').trim();

      if (!claimTenantId || !claimRole) {
        return {
          allowed: false,
          code: 'E_RETRIEVAL_SCOPE_CLAIMS_MISSING',
          reason: 'Valid tenant and role claims are required for vault retrieval access.',
        };
      }

      if (!ALLOWED_RETRIEVAL_ROLES.has(claimRole)) {
        return {
          allowed: false,
          code: 'E_RETRIEVAL_SCOPE_ROLE_DENIED',
          reason: `Role '${claimRole}' is not permitted to access vault retrieval.`,
        };
      }

      const resourceTenantId = String((resourceContext && resourceContext.tenantId) || '').trim();
      if (!resourceTenantId || claimTenantId !== resourceTenantId) {
        return {
          allowed: false,
          code: 'E_RETRIEVAL_SCOPE_TENANT_MISMATCH',
          reason: 'Claims tenant does not match the resource tenant.',
        };
      }

      return {
        allowed: true,
        code: null,
        reason: null,
      };
    }
    ```
