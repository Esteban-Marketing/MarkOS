    ```javascript
    'use strict';

    const { projectAuditLineage, checkRetrievalScope } = require('./visibility-scope.cjs');
    const { applyFilter } = require('./retrieval-filter.cjs');
    const { buildHandoffPack } = require('./handoff-pack.cjs');

    function createError(code, message) {
      const error = new Error(message);
      error.code = code;
      return error;
    }

    function asFunction(value, code, message) {
      if (typeof value !== 'function') {
        throw createError(code, message);
      }
      return value;
    }

    /**
     * createVaultRetriever — factory for the Phase 86 read-path (CQRS read side).
     *
     * Mirrors createIngestApply (write-path) factory pattern:
     * injected dependencies, returned method set, fail-fast validation.
     *
     * @param {object} options
     * @param {function} options.getArtifacts           async ({tenantId}) => entry[]
     * @param {function} [options.checkScope]           defaults to checkRetrievalScope
     * @param {function} [options.filterEntries]        defaults to applyFilter
     * @param {function} [options.packHandoff]          defaults to buildHandoffPack
     */
    function createVaultRetriever(options = {}) {
      const getArtifacts = asFunction(
        options.getArtifacts,
        'E_RETRIEVER_ARTIFACTS_REQUIRED',
        'getArtifacts({ tenantId }) is required for vault retriever.'
      );

      const checkScope = typeof options.checkScope === 'function' ? options.checkScope : checkRetrievalScope;
      const filterEntries = typeof options.filterEntries === 'function' ? options.filterEntries : applyFilter;
      const packHandoff = typeof options.packHandoff === 'function' ? options.packHandoff : buildHandoffPack;

      async function retrieve({ tenantId, claims, filter, mode }) {
        // 1. Scope check — fails closed, throws on denial
        const scope = checkScope(claims || {}, { tenantId });
        if (!scope.allowed) {
          throw createError(scope.code || 'E_RETRIEVAL_SCOPE_DENIED', scope.reason || 'Retrieval access denied.');
        }

        // 2. Fetch all artifacts for this tenant from the store
        const all = await getArtifacts({ tenantId });

        // 3. Tenant isolation FIRST (before any filter)
        const scoped = projectAuditLineage(claims || {}, all);

        // 4. Apply discipline + audience filter (AND semantics)
        const safeFilter = filter && typeof filter === 'object' ? filter : {};
        const filtered = filterEntries(scoped, safeFilter);

        // 5. Build audience context for handoff packs
        const filterApplied = !!(
          safeFilter.discipline ||
          (Array.isArray(safeFilter.audience_tags) && safeFilter.audience_tags.length > 0)
        );
        const audienceContext = { ...safeFilter, filter_applied: filterApplied };

        // 6. Pack each artifact into a mode-specific handoff payload
        return filtered.map((artifact) => packHandoff({ mode, artifact, audienceContext, claims }));
      }

      return {
        retrieveReason:  (args) => retrieve({ ...(args || {}), mode: 'reason' }),
        retrieveApply:   (args) => retrieve({ ...(args || {}), mode: 'apply' }),
        retrieveIterate: (args) => retrieve({ ...(args || {}), mode: 'iterate' }),
      };
    }

    module.exports = { createVaultRetriever };
    ```
