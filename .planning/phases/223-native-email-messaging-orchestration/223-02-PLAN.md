<!-- D-50 frequency-cap trigger (migration 121) -->
```sql
CREATE OR REPLACE FUNCTION enforce_frequency_cap()
RETURNS TRIGGER AS $$
DECLARE
  v_cap INTEGER;
  v_count INTEGER;
  v_window_days INTEGER := 7;
  v_marketing_classes TEXT[] := ARRAY['lifecycle', 'broadcast', 'revenue'];
BEGIN
  -- Only enforce on marketing-class events; transactional unlimited
  IF NEW.event_type NOT IN ('queued', 'delivered') OR NEW.campaign_class NOT IN (SELECT unnest(v_marketing_classes)) THEN
    RETURN NEW;
  END IF;
  -- Read tenant cap from tenant_config.frequency_caps[channel]; default 2
  SELECT COALESCE((config->'frequency_caps'->NEW.channel)::INTEGER, 2) INTO v_cap
  FROM markos_tenants WHERE id = NEW.tenant_id;
  -- Count rolling-window dispatches
  SELECT COUNT(*) INTO v_count FROM dispatch_events
  WHERE tenant_id = NEW.tenant_id AND profile_id = NEW.profile_id AND channel = NEW.channel
    AND event_type IN ('queued', 'delivered')
    AND campaign_class IN (SELECT unnest(v_marketing_classes))
    AND occurred_at > NOW() - (v_window_days || ' days')::INTERVAL;
  IF v_count >= v_cap THEN
    RAISE EXCEPTION 'frequency_cap_exceeded: profile=% channel=% count=% cap=%', NEW.profile_id, NEW.channel, v_count, v_cap;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_dispatch_events_freq_cap BEFORE INSERT ON dispatch_events
  FOR EACH ROW EXECUTE FUNCTION enforce_frequency_cap();
```
