**Step 1. Create 10 marketing handler files. Use `plan-campaign.cjs` as scaffold. Each file follows this template (example for remix-draft; repeat with per-tool input/output fields for the other 9):**

`lib/markos/mcp/tools/marketing/remix-draft.cjs`:

```javascript
'use strict';

const { COST_TABLE } = require('../../cost-table.cjs');

const inputSchema = {
  type: 'object', required: ['draft', 'directive'], additionalProperties: false,
  properties: {
    draft: { type: 'string', minLength: 1, maxLength: 10000 },
    directive: { type: 'string', minLength: 1, maxLength: 500 }, // e.g. "shorter", "more formal"
    variants: { type: 'integer', minimum: 1, maximum: 5 },
  },
};

const outputSchema = {
  type: 'object', required: ['content', '_usage'], additionalProperties: true,
  properties: {
    content: { type: 'array', items: { type: 'object', required: ['type', 'text'], properties: { type: { enum: ['text'] }, text: { type: 'string' } } } },
    _usage: { type: 'object', required: ['input_tokens', 'output_tokens'], properties: { input_tokens: { type: 'integer', minimum: 0 }, output_tokens: { type: 'integer', minimum: 0 } } },
  },
};

async function invokeLLM(draft, directive, variants, deps) {
  const llm = deps.llm;
  if (!llm) return { text: JSON.stringify({ variants: [{ remixed: draft }] }), input_tokens: 0, output_tokens: 0 };
  const resp = await llm.messages.create({
    model: COST_TABLE.remix_draft.model,
    max_tokens: 1500,
    system: `You remix a marketing draft according to a directive. Return strict JSON { variants: [{ remixed: string, why: string }] }.`,
    messages: [{ role: 'user', content: `Directive: ${directive}\nVariants: ${variants || 3}\nDraft:\n${draft}` }],
  });
  return {
    text: (resp.content && resp.content[0] && resp.content[0].text) || '{"variants":[]}',
    input_tokens: resp.usage?.input_tokens || 0,
    output_tokens: resp.usage?.output_tokens || 0,
  };
}

async function handler(ctx) {
  const { args, session } = ctx;
  const deps = ctx.deps || {};
  const { text, input_tokens, output_tokens } = await invokeLLM(args.draft, args.directive, args.variants, deps);
  let parsed;
  try { parsed = JSON.parse(text); } catch { parsed = { variants: [{ remixed: args.draft, why: 'parse_failed' }] }; }
  return {
    content: [{
      type: 'text',
      text: JSON.stringify({ tenant_id: session.tenant_id, directive: args.directive, variants: Array.isArray(parsed.variants) ? parsed.variants : [] }, null, 2),
    }],
    _usage: { input_tokens, output_tokens },
  };
}

const descriptor = {
  name: 'remix_draft',
  description: 'Remix a marketing draft according to a directive (e.g. "shorter", "more formal", "different angle"). Returns N variants.',
  latency_tier: 'llm',
  mutating: false,
  cost_model: COST_TABLE.remix_draft,
  inputSchema, outputSchema, handler,
};

module.exports = { descriptor, inputSchema, outputSchema, handler };
```

Repeat the pattern with per-tool adjustments for the other 9:

- `rank-draft-variants.cjs`: input `{ variants: array(2..10) of string, brief?: object }`; output has `content + _usage`; handler asks LLM to score each variant → returns ranked list `[{ variant, score, reasons }]`. Cost = `COST_TABLE.rank_draft_variants` (Haiku).
- `brief-to-plan.cjs`: input `{ brief: object required { channel, audience, pain, promise, brand } }`; output `{ content + _usage }`; handler expands brief into 5-step plan (research → pain → promise → drafts → schedule). Cost = Sonnet.
- `generate-channel-copy.cjs`: input `{ channel: enum[email, x, linkedin, sms], brief: object }`; output channel blocks `{ subject?, preview?, body, cta }`. Sonnet.
- `expand-claim-evidence.cjs`: input `{ claim: string }`; output `{ canon_evidence: [...], strengthening_variants: [...] }`. Sonnet, uses pack-loader.
- `clone-persona-voice.cjs`: input `{ persona_slug: string, topic: string }`; output `{ cloned_draft: string, voice_markers: [...] }`. Sonnet.
- `generate-subject-lines.cjs`: input `{ body: string }`; output `{ subjects: array(10) of { text, score, reasoning } }`. Haiku.
- `optimize-cta.cjs`: input `{ draft: string }`; output `{ alternatives: array of { cta, predicted_lift, source } }`. Haiku.
- `generate-preview-text.cjs`: input `{ subject: string, body: string }`; output `{ previews: array(5) of { text, score } }`. Haiku.

**Step 1. Create `scripts/openapi/build-mcp-schemas.mjs`:**

```javascript
#!/usr/bin/env node
// Phase 202 Plan 07: YAML → _generated/tool-schemas.json codegen for AJV runtime validation.
// Reads F-90..F-93 + retains F-71 shape compatibility. Run at build time + in CI.

import { readFileSync, writeFileSync, mkdirSync } from 'node:fs';
import { resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import YAML from 'yaml';

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = resolve(__dirname, '..', '..');
const CONTRACTS = [
  'contracts/F-90-mcp-tools-marketing-v1.yaml',
  'contracts/F-91-mcp-tools-crm-v1.yaml',
  'contracts/F-92-mcp-tools-literacy-v1.yaml',
  'contracts/F-93-mcp-tools-tenancy-v1.yaml',
];
const OUT = 'lib/markos/mcp/_generated/tool-schemas.json';

function resolveRef(doc, ref) {
  // minimal $ref resolver: "#/shared/llm_content_plus_usage"
  if (typeof ref !== 'string' || !ref.startsWith('#/')) return null;
  const parts = ref.slice(2).split('/');
  let cur = doc;
  for (const p of parts) {
    if (!cur || typeof cur !== 'object' || !(p in cur)) return null;
    cur = cur[p];
  }
  return cur;
}

function dereference(schema, doc) {
  if (!schema || typeof schema !== 'object') return schema;
  if (schema.$ref) {
    const resolved = resolveRef(doc, schema.$ref);
    if (!resolved) throw new Error(`unresolved_ref:${schema.$ref}`);
    return dereference(resolved, doc);
  }
  if (Array.isArray(schema)) return schema.map(s => dereference(s, doc));
  const out = {};
  for (const [k, v] of Object.entries(schema)) out[k] = dereference(v, doc);
  return out;
}

const registry = {};
for (const rel of CONTRACTS) {
  const abs = resolve(ROOT, rel);
  const src = readFileSync(abs, 'utf8');
  const doc = YAML.parse(src);
  if (!doc.tools || typeof doc.tools !== 'object') {
    throw new Error(`contract ${rel} missing tools map`);
  }
  for (const [tool_id, spec] of Object.entries(doc.tools)) {
    if (!spec.input || !spec.output) throw new Error(`${rel}:${tool_id} missing input/output`);
    const input = dereference(spec.input, doc);
    const output = dereference(spec.output, doc);
    registry[tool_id] = { input, output };
  }
}

const outAbs = resolve(ROOT, OUT);
mkdirSync(dirname(outAbs), { recursive: true });
writeFileSync(outAbs, JSON.stringify(registry, null, 2) + '\n');
console.log(`[build-mcp-schemas] wrote ${Object.keys(registry).length} tool schemas → ${OUT}`);
```
