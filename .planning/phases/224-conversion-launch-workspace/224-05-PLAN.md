       import { createHash } from 'node:crypto';
       let xxhasher: { h32: (s: string) => number } | null = null;
       async function getXxHasher() {
         if (xxhasher) return xxhasher;
         try {
           const mod = await import('xxhash-wasm');
           const h = await mod.default();
           xxhasher = { h32: (s: string) => h.h32(s) };
         } catch {
           xxhasher = { h32: (s: string) => createHash('sha256').update(s).digest().readUInt32BE(0) };
         }
         return xxhasher;
       }
       export async function hashToBucket(experimentId: string, identityRef: string, totalWeight: number): Promise<number> {
         const h = await getXxHasher();
         return h.h32(`${experimentId}:${identityRef}`) % totalWeight;
       }
       ```

       import type { ExperimentVariant } from '../contracts/experiment';
       export function resolveVariantByTrafficSplit(variants: ExperimentVariant[], bucket: number): ExperimentVariant {
         const sorted = [...variants].sort((a, b) => a.variant_id.localeCompare(b.variant_id));
         let cumulative = 0;
         for (const v of sorted) {
           cumulative += v.traffic_weight;
           if (bucket < cumulative) return v;
         }
         return sorted[sorted.length - 1];
       }
       export function totalWeight(variants: ExperimentVariant[]): number {
         return variants.reduce((sum, v) => sum + v.traffic_weight, 0);
       }
       ```

       import { hashToBucket } from './sticky-hash';
       import { resolveVariantByTrafficSplit, totalWeight } from './traffic-split';
       import type { ExperimentVariant } from '../contracts/experiment';
       export async function getOrAssignVariant(input: {
         supabase; tenant_id: string; experiment_id: string; identity_ref: string;
       }): Promise<ExperimentVariant | null> {
         const { data: experiment } = await input.supabase
           .from('conversion_experiments').select('status, traffic_split')
           .eq('experiment_id', input.experiment_id).eq('tenant_id', input.tenant_id).single();
         if (!experiment) return null;
         const { data: variants } = await input.supabase
           .from('experiment_variants').select('*')
           .eq('experiment_id', input.experiment_id).eq('tenant_id', input.tenant_id);
         if (!variants || variants.length === 0) return null;

         if (experiment.status !== 'active') {
           return variants.find(v => v.is_control) ?? variants[0];
         }
         const { data: existing } = await input.supabase
           .from('experiment_assignments').select('variant_id')
           .eq('experiment_id', input.experiment_id).eq('identity_ref', input.identity_ref).maybeSingle();
         if (existing) {
           return variants.find(v => v.variant_id === existing.variant_id) ?? null;
         }
         const total = totalWeight(variants);
         const bucket = await hashToBucket(input.experiment_id, input.identity_ref, total);
         const chosen = resolveVariantByTrafficSplit(variants, bucket);
         await input.supabase.from('experiment_assignments').upsert({
           tenant_id: input.tenant_id, experiment_id: input.experiment_id,
           identity_ref: input.identity_ref, variant_id: chosen.variant_id, sticky: true,
         }, { onConflict: 'experiment_id,identity_ref', ignoreDuplicates: true });
         return chosen;
       }
       ```

       import { appendAuditRow } from '../../audit';
       import type { ExperimentStatus } from '../contracts/experiment';
       const TRANSITIONS: Record<ExperimentStatus, ExperimentStatus[]> = {
         draft: ['active'],
         active: ['paused', 'completed'],
         paused: ['active', 'completed'],
         completed: [],
       };
       export async function transitionExperiment(input: {
         supabase; tenant_id: string; experiment_id: string; to: ExperimentStatus;
       }): Promise<{ ok: true } | never> {
         const { data: exp } = await input.supabase.from('conversion_experiments').select('status').eq('experiment_id', input.experiment_id).single();
         if (!exp) throw new Error('experiment_not_found');
         if (!TRANSITIONS[exp.status]?.includes(input.to)) throw new Error(`invalid_transition:${exp.status}->${input.to}`);
         const updates: Record<string, unknown> = { status: input.to, updated_at: new Date().toISOString() };
         if (input.to === 'active' && exp.status === 'draft') updates.started_at = new Date().toISOString();
         if (input.to === 'completed') updates.ended_at = new Date().toISOString();
         await input.supabase.from('conversion_experiments').update(updates).eq('experiment_id', input.experiment_id);
         await appendAuditRow({ tenant_id: input.tenant_id, event_type: `experiment_${input.to}`, payload: { experiment_id: input.experiment_id, from: exp.status, to: input.to } });
         return { ok: true };
       }
       ```
