       ```ts
       /**
        * Hash (experiment_id, identity_ref) → bucket in [0, total_weight).
        * Uses SHA-256 truncated to 16 hex chars (64-bit) per D-70.

         experiment_id: string,
         identity_ref: string,
         total_weight = 10000
       ): number {
         const digest = createHash('sha256')
           .update(`${experiment_id}::${identity_ref}`)
           .digest('hex')
           .substring(0, 16);
         // parseInt of 16 hex chars = up to 64-bit number; safe for modulo
         return parseInt(digest, 16) % total_weight;
       }
       ```

       ```ts
       import type { ExperimentVariant } from '@/lib/markos/conversion/contracts';

       export function resolveVariantByTrafficSplit(
         variants: ExperimentVariant[],
         bucket: number,
         total_weight = 10000
       ): ExperimentVariant {
         // Stable sort by variant_id for deterministic boundary mapping
         const sorted = [...variants].sort((a, b) => a.variant_id.localeCompare(b.variant_id));
         const totalAssignedWeight = sorted.reduce((sum, v) => sum + v.traffic_weight, 0);
         if (totalAssignedWeight === 0) {
           return sorted.find(v => v.is_control) ?? sorted[0];
         }
         const scaled = (bucket / total_weight) * totalAssignedWeight;
         let cumulative = 0;
         for (const v of sorted) {
           cumulative += v.traffic_weight;
           if (scaled < cumulative) return v;
         }
         return sorted[sorted.length - 1]; // boundary fallback
       }
       ```

       ```ts
       import { hashIdentityToBucket } from './hash';
       import { resolveVariantByTrafficSplit } from './traffic-split';

       export async function getOrAssignVariant(input: {
         experiment_id: string;
         tenant_id: string;
         identity_ref: string;
         variants: ExperimentVariant[];
         supabase: SupabaseClient;
       }): Promise<{ variant_id: string; assigned_now: boolean }> {
         // 1. Sticky lookup
         const { data: existing } = await input.supabase
           .from('experiment_assignments')
           .select('variant_id')
           .eq('experiment_id', input.experiment_id)
           .eq('identity_ref', input.identity_ref)
           .maybeSingle();
         if (existing) return { variant_id: existing.variant_id, assigned_now: false };

         // 2. Compute new assignment
         const bucket = hashIdentityToBucket(input.experiment_id, input.identity_ref);
         const variant = resolveVariantByTrafficSplit(input.variants, bucket);

         // 3. Persist via INSERT ON CONFLICT — DB unique index (Plan 01 migration 124) enforces sticky guarantee
         const { data, error } = await input.supabase
           .from('experiment_assignments')
           .upsert({
             tenant_id: input.tenant_id,
             experiment_id: input.experiment_id,
             identity_ref: input.identity_ref,
             variant_id: variant.variant_id,
             assigned_at: new Date().toISOString(),
           }, { onConflict: 'experiment_id,identity_ref', ignoreDuplicates: false })
           .select('variant_id')
           .single();
         if (error) throw error;
         return { variant_id: data.variant_id, assigned_now: true };
       }
       ```
