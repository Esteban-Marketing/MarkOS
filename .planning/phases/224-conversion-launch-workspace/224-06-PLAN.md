<objective>
Ship the LaunchRunbook + AgentRun execution + reverse-runbook rollback + 4 cron handlers + F-142 contract. After this plan, an operator can author a runbook with ordered publish + dispatch + notify steps with paired rollback_steps; arm it (gates re-evaluated); execute it (each step idempotent, AgentRun-wrapped or bridged); rollback at any time (reverse order, non-reversible steps emit operator tasks). Cron jobs run surface health audit (D-48), gate evaluation poll (D-17), outcome computation (D-39), bounce/spike alert (D-50).

Purpose: satisfy LCH-01 (rollback posture is first-class), LCH-02 (multi-channel orchestration via step kinds covering 6 channels), LCH-04 (outcomes feed analytics + cron-driven), LCH-05 (no external launch mutation without evidence/approval/audit — every step writes audit row), CNV-04 (D-50 bounce-spike + D-48 surface health observability), QA-01 (F-142 active + F-145/F-146 MCP stubs), QA-02 (RLS on launch_runbooks + per-step audit + cron heartbeat).

Output: 2 migrations (128 + 132) + 13 lib files (5 runbook + 4 cron + 4 indexes) + 5 API routes + 4 cron handlers + vercel.json update + F-142 contract + 11 test files. Plan 06 runs in Wave 4 — depends on plans 01..05.
</objective>

       import { startAgentRunWrapper, completeAgentRunWrapper } from './agentrun-bridge';
       import { checkStepIdempotency, logStepStart, logStepComplete } from './idempotency';
       import { getStepHandler } from './step-handlers';
       import { appendAuditRow } from '../../audit';

       export async function executeRunbook(input: { supabase; tenant_id: string; runbook_id: string }): Promise<{ state: 'executed' | 'failed'; failed_step?: string }> {
         const { data: runbook } = await input.supabase.from('launch_runbooks').select('*').eq('runbook_id', input.runbook_id).single();
         if (!runbook) throw new Error('runbook_not_found');
         if (!['armed', 'executing'].includes(runbook.state)) throw new Error(`invalid_state_for_execute:${runbook.state}`);

         await input.supabase.from('launch_runbooks').update({ state: 'executing', started_at: runbook.started_at ?? new Date().toISOString() }).eq('runbook_id', input.runbook_id);
         const agentrun_id = await startAgentRunWrapper({ supabase: input.supabase, tenant_id: input.tenant_id, runbook_id: input.runbook_id });

         const steps = runbook.steps as RunbookStep[];
         try {
           // Resolve dependency layers (steps with same depends_on level run in parallel)
           const completed = new Set<string>();
           const skippedStepIds: string[] = [];
           // Resume from runbook.current_step (allows re-entrant execution)
           for (let i = runbook.current_step; i < steps.length; i++) {
             const step = steps[i];
             // Wait for depends_on
             for (const dep of step.depends_on) {
               if (!completed.has(dep)) throw new Error(`unmet_dependency:${step.step_id}->${dep}`);
             }
             // Idempotency check
             const idem = await checkStepIdempotency({ supabase: input.supabase, tenant_id: input.tenant_id, runbook_id: input.runbook_id, step_id: step.step_id, idempotency_key: step.idempotency_key, direction: 'forward' });
             if (idem.already_executed) {
               completed.add(step.step_id);
               continue;
             }
             await logStepStart({ supabase: input.supabase, tenant_id: input.tenant_id, runbook_id: input.runbook_id, step_id: step.step_id, step_kind: step.kind, idempotency_key: step.idempotency_key, direction: 'forward' });
             const handler = getStepHandler(step.kind);
             if (!handler) {
               await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'forward', status: 'failed', payload: { error: `unknown_step_kind:${step.kind}` } });
               throw new Error(`unknown_step_kind:${step.kind}`);
             }
             const result = await handler(step, { supabase: input.supabase, tenant_id: input.tenant_id, launch_id: runbook.launch_id, direction: 'forward' });
             // Graceful-skip pattern: handlers signaling step_handler_unavailable (e.g., post_social v1
             // pending P227, custom v1 not implemented) emit an operator task and CONTINUE the runbook
             // rather than aborting. Mirrors the non-reversible rollback pattern from D-36 / Pitfall 6.
             const stepHandlerUnavailable = !result.ok && (
               result.error.startsWith('step_handler_unavailable:') ||
               result.error === 'custom_step_handler_not_implemented_v1'
             );
             if (stepHandlerUnavailable) {
               await input.supabase.from('markos_tasks').insert({
                 tenant_id: input.tenant_id,
                 kind: 'operator_task_step_skipped',
                 title: `Manual handling required: ${step.name} (${step.kind})`,
                 payload: { runbook_id: input.runbook_id, step_id: step.step_id, kind: step.kind, target_ref: step.target_ref, reason: result.error },
               });
               await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'forward', status: 'operator_task_emitted', payload: { reason: result.error } });
               skippedStepIds.push(step.step_id);
               await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'launch_runbook_step_skipped', payload: { runbook_id: input.runbook_id, step_id: step.step_id, kind: step.kind, reason: result.error } });
               completed.add(step.step_id);
               await input.supabase.from('launch_runbooks').update({ current_step: i + 1, updated_at: new Date().toISOString() }).eq('runbook_id', input.runbook_id);
               continue;
             }
             await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'forward', status: result.ok ? 'succeeded' : 'failed', payload: result.ok ? result.payload : { error: result.error } });
             if (!result.ok) throw new Error(`step_failed:${step.step_id}:${result.error}`);
             completed.add(step.step_id);
             await input.supabase.from('launch_runbooks').update({ current_step: i + 1, updated_at: new Date().toISOString() }).eq('runbook_id', input.runbook_id);
           }
           await input.supabase.from('launch_runbooks').update({ state: 'executed', completed_at: new Date().toISOString() }).eq('runbook_id', input.runbook_id);
           await completeAgentRunWrapper({ supabase: input.supabase, agentrun_id, status: 'succeeded' });
           await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'launch_runbook_executed', payload: { runbook_id: input.runbook_id, launch_id: runbook.launch_id, skipped_step_ids: skippedStepIds } });
           return { state: 'executed' };
         } catch (err) {
           await input.supabase.from('launch_runbooks').update({ state: 'failed', completed_at: new Date().toISOString() }).eq('runbook_id', input.runbook_id);
           await completeAgentRunWrapper({ supabase: input.supabase, agentrun_id, status: 'failed' });
           await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'launch_runbook_failed', payload: { runbook_id: input.runbook_id, error: (err as Error).message } });
           return { state: 'failed', failed_step: (err as Error).message };
         }
       }

       export async function rollbackRunbook(input: { supabase; tenant_id: string; runbook_id: string }): Promise<{ state: 'rolled_back'; partial_rollback_blockers: string[] }> {
         const { data: runbook } = await input.supabase.from('launch_runbooks').select('*').eq('runbook_id', input.runbook_id).single();
         if (!runbook) throw new Error('runbook_not_found');

         await input.supabase.from('launch_runbooks').update({ state: 'rolling_back', updated_at: new Date().toISOString() }).eq('runbook_id', input.runbook_id);

         const rollbackSteps = (runbook.rollback_steps as RunbookStep[]).slice().reverse(); // REVERSE order
         const partial_blockers: string[] = [];

         for (const step of rollbackSteps) {
           const idem = await checkStepIdempotency({ supabase: input.supabase, tenant_id: input.tenant_id, runbook_id: input.runbook_id, step_id: step.step_id, idempotency_key: step.idempotency_key, direction: 'rollback' });
           if (idem.already_executed) continue;

           await logStepStart({ supabase: input.supabase, tenant_id: input.tenant_id, runbook_id: input.runbook_id, step_id: step.step_id, step_kind: step.kind, idempotency_key: step.idempotency_key, direction: 'rollback' });

           // Pitfall 6: non-reversible steps emit operator task + CONTINUE
           if (step.reversible === false) {
             await input.supabase.from('markos_tasks').insert({
               tenant_id: input.tenant_id, kind: 'rollback_non_reversible',
               title: `Manual rollback required: ${step.name}`,
               payload: { runbook_id: input.runbook_id, step_id: step.step_id, target_ref: step.target_ref, kind: step.kind },
             });
             await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'rollback', status: 'operator_task_emitted', payload: { reason: 'non_reversible' } });
             partial_blockers.push(step.step_id);
             await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'runbook_rollback_step', payload: { runbook_id: input.runbook_id, step_id: step.step_id, reversible: false, operator_task_emitted: true } });
             continue;
           }

           const handler = getStepHandler(step.kind);
           if (!handler) {
             await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'rollback', status: 'failed', payload: { error: `unknown_step_kind:${step.kind}` } });
             partial_blockers.push(step.step_id);
             continue; // Pitfall 6 — never abort
           }
           try {
             const result = await handler(step, { supabase: input.supabase, tenant_id: input.tenant_id, launch_id: runbook.launch_id, direction: 'rollback' });
             await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'rollback', status: result.ok ? 'succeeded' : 'failed', payload: result.ok ? result.payload : { error: result.error } });
             if (!result.ok) {
               partial_blockers.push(step.step_id);
               // If the step error indicates non-reversibility (caught at handler level), emit task too
               if (result.error.startsWith('non_reversible')) {
                 await input.supabase.from('markos_tasks').insert({
                   tenant_id: input.tenant_id, kind: 'rollback_non_reversible',
                   title: `Manual rollback required: ${step.name}`,
                   payload: { runbook_id: input.runbook_id, step_id: step.step_id, target_ref: step.target_ref, kind: step.kind },
                 });
               }
             }
             await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'runbook_rollback_step', payload: { runbook_id: input.runbook_id, step_id: step.step_id, reversible: true, succeeded: result.ok } });
           } catch (err) {
             await logStepComplete({ supabase: input.supabase, ...step, runbook_id: input.runbook_id, idempotency_key: step.idempotency_key, direction: 'rollback', status: 'failed', payload: { error: (err as Error).message } });
             partial_blockers.push(step.step_id);
             // CONTINUE — Pitfall 6
           }
         }

         await input.supabase.from('launch_runbooks').update({
           state: 'rolled_back', completed_at: new Date().toISOString(),
           partial_rollback_blockers: partial_blockers,
         }).eq('runbook_id', input.runbook_id);
         await appendAuditRow({ tenant_id: input.tenant_id, event_type: 'launch_runbook_rolled_back', payload: { runbook_id: input.runbook_id, partial_blockers } });

         return { state: 'rolled_back', partial_rollback_blockers: partial_blockers };
       }

         // Per tenant: compute 24h conversion rate; compare to baseline_mean ± 2σ
         const { data: tenants } = await supabase.from('markos_tenants').select('id');
         let alerts = 0;
         for (const t of tenants ?? []) {
           // Conversion rate (form submits / page views)
           const { count: pageViews } = await supabase.from('conversion_events').select('*', { count: 'exact', head: true })
             .eq('tenant_id', t.id).eq('surface_kind', 'page').gte('occurred_at', new Date(Date.now() - 24 * 3600 * 1000).toISOString());
           const { count: formSubmits } = await supabase.from('conversion_events').select('*', { count: 'exact', head: true })
             .eq('tenant_id', t.id).eq('surface_kind', 'form').gte('occurred_at', new Date(Date.now() - 24 * 3600 * 1000).toISOString());
           const rate24h = pageViews ? (formSubmits ?? 0) / pageViews : 0;
           const { data: baseline } = await supabase.from('bounce_alert_baselines').select('baseline_mean, baseline_stddev').eq('tenant_id', t.id).eq('metric', 'conversion_rate').maybeSingle();
           if (baseline && baseline.baseline_mean - rate24h > 2 * baseline.baseline_stddev) {
             await supabase.from('markos_tasks').insert({
               tenant_id: t.id, kind: 'conversion_rate_drop',
               title: `Conversion rate dropped >2σ in last 24h`,
               payload: { rate_24h: rate24h, baseline_mean: baseline.baseline_mean, stddev: baseline.baseline_stddev },
             });
             alerts++;
           }
           // BotID abuse spike
           const { count: botBlocks24h } = await supabase.from('markos_audit_log').select('*', { count: 'exact', head: true })
             .eq('tenant_id', t.id).eq('event_type', 'bot_id_block').gte('occurred_at', new Date(Date.now() - 24 * 3600 * 1000).toISOString());
           const { data: botBaseline } = await supabase.from('bounce_alert_baselines').select('baseline_mean').eq('tenant_id', t.id).eq('metric', 'bot_id_block_rate').maybeSingle();
           if (botBaseline && (botBlocks24h ?? 0) > 10 * botBaseline.baseline_mean) {
             await supabase.from('markos_tasks').insert({
               tenant_id: t.id, kind: 'bot_id_abuse_spike',
               title: `BotID block rate >10× baseline in last 24h`,
               payload: { blocks_24h: botBlocks24h, baseline_mean: botBaseline.baseline_mean },
             });
             alerts++;
           }
         }
         return { alerts_raised: alerts };
       }
