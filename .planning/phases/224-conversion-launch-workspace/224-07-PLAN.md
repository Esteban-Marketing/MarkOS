         const plans = readdirSync(PLAN_DIR).filter(f => /^224-\d+-PLAN\.md$/.test(f));
         const slotMap = new Map();  // slot → [plan_id]
         const SLOT_PATTERN = /supabase\/migrations\/(\d{3})_/g;

         for (const plan of plans) {
           const planId = plan.match(/^(224-\d+)/)[1];
           const body = readFileSync(`${PLAN_DIR}/${plan}`, 'utf8');
           // Extract files_modified block; look for migration paths
           let match;
           while ((match = SLOT_PATTERN.exec(body)) !== null) {
             const slot = parseInt(match[1], 10);
             if (slot < 121 || slot > 140) continue;  // P224 range
             if (!slotMap.has(slot)) slotMap.set(slot, []);
             const owners = slotMap.get(slot);
             if (!owners.includes(planId)) owners.push(planId);
