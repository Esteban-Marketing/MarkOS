       ```sql
       -- Phase 224 plan 02: emit_conversion_event_tx atomic fan-out + indexes + audit registrations

       -- Indexes for fast joining via shared source_event_ref
       CREATE INDEX IF NOT EXISTS idx_cdp_events_source_event_ref ON cdp_events(source_event_ref);
       CREATE INDEX IF NOT EXISTS idx_crm_activity_source_event_ref ON crm_activity(source_event_ref);

       -- Idempotency cache table (60s TTL via cron in plan 06)
       CREATE TABLE IF NOT EXISTS conversion_submit_idempotency (
         idempotency_key  TEXT PRIMARY KEY,
         tenant_id        UUID NOT NULL REFERENCES markos_tenants(id),
         form_id          UUID NOT NULL,
         response_payload JSONB NOT NULL,
         created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
       );
       CREATE INDEX IF NOT EXISTS idx_idem_created ON conversion_submit_idempotency(created_at);

       CREATE POLICY conv_idem_tenant ON conversion_submit_idempotency

       -- Rate-limit counters (IP + email dimensions; rolling 60s window via Postgres)
       CREATE TABLE IF NOT EXISTS conversion_rate_limit_counters (
         counter_id   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
         tenant_id    UUID NOT NULL REFERENCES markos_tenants(id),
         form_id      UUID NOT NULL,
         dimension    TEXT NOT NULL CHECK (dimension IN ('ip', 'email')),
         dimension_value TEXT NOT NULL,
         window_start TIMESTAMPTZ NOT NULL,

         CONSTRAINT conv_rl_unique UNIQUE (tenant_id, form_id, dimension, dimension_value, window_start)
       );

       ALTER TABLE conversion_rate_limit_counters ENABLE ROW LEVEL SECURITY;

         USING (tenant_id = current_setting('app.active_tenant_id', true)::UUID);

       -- Atomic fan-out function — 7 sinks transactional
       CREATE OR REPLACE FUNCTION emit_conversion_event_tx(
         p_tenant_id UUID,
         p_surface_id UUID,
         p_surface_kind TEXT,
         p_form_id UUID,
         p_experiment_variant_id UUID,
         p_objective TEXT,
         p_identity_ref TEXT,
         p_pricing_context_id UUID,
         p_evidence_pack_id UUID,
         p_launch_id UUID,
         p_source_event_ref UUID,
         p_event_domain TEXT,           -- 'website' | 'product' (cdp_events sink)
         p_commercial_signal TEXT,      -- mapped from objective per D-10 (e.g., 'interest', 'expansion')
         p_consent_capture JSONB        -- { channel, status } if surface_kind='form' and consent_capture_block present
       ) RETURNS JSONB
       LANGUAGE plpgsql
       SECURITY DEFINER
       AS $$
       DECLARE
         v_event_id UUID := gen_random_uuid();
         v_source_event_ref UUID := COALESCE(p_source_event_ref, gen_random_uuid());
         v_now TIMESTAMPTZ := NOW();
         v_sinks_written TEXT[] := ARRAY[]::TEXT[];
       BEGIN
         -- Tenant guard: function asserts caller's app.active_tenant_id matches p_tenant_id
         IF current_setting('app.active_tenant_id', true)::UUID != p_tenant_id THEN
           RAISE EXCEPTION 'tenant_mismatch' USING ERRCODE = '42501';
         END IF;

         -- Sink 1: conversion_events
         INSERT INTO conversion_events(
           event_id, tenant_id, surface_id, surface_kind, form_id, experiment_variant_id,
           objective, identity_ref, pricing_context_id, evidence_pack_id, launch_id,
           occurred_at, source_event_ref
         ) VALUES (
           v_event_id, p_tenant_id, p_surface_id, p_surface_kind, p_form_id, p_experiment_variant_id,
           p_objective, p_identity_ref, p_pricing_context_id, p_evidence_pack_id, p_launch_id,
           v_now, v_source_event_ref
         );
         v_sinks_written := array_append(v_sinks_written, 'conversion_events');

         -- Sink 2: cdp_events envelope (P221)
         INSERT INTO cdp_events(
           event_id, tenant_id, profile_id, anonymous_identity_id, event_domain, event_name,
           occurred_at, source_event_ref, payload
         ) VALUES (
           gen_random_uuid(), p_tenant_id,
           CASE WHEN p_identity_ref ~ '^[0-9a-f-]{36}$' THEN p_identity_ref::UUID ELSE NULL END,
           CASE WHEN p_identity_ref !~ '^[0-9a-f-]{36}$' THEN p_identity_ref ELSE NULL END,
           p_event_domain,
           CASE p_surface_kind
             WHEN 'page' THEN 'page_view'
             WHEN 'form' THEN 'form_submit'
             WHEN 'cta' THEN 'cta_click'
           END,
           v_now, v_source_event_ref,
           jsonb_build_object(
             'surface_id', p_surface_id, 'objective', p_objective,
             'experiment_variant_id', p_experiment_variant_id,
             'launch_id', p_launch_id
           )
         );
         v_sinks_written := array_append(v_sinks_written, 'cdp_events');

         -- Sink 3: crm_activity (commercial_signal projection)
         INSERT INTO crm_activity(
           activity_id, tenant_id, source_domain, commercial_signal, identity_ref,
           occurred_at, source_event_ref, payload
         ) VALUES (
           gen_random_uuid(), p_tenant_id,
           CASE p_event_domain WHEN 'product' THEN 'product' ELSE 'website' END,
           p_commercial_signal,
           p_identity_ref, v_now, v_source_event_ref,
       ```

       ```typescript
       export interface RateLimitResult { allowed: boolean; retry_after_seconds?: number; dimension_blocked?: 'ip' | 'email' }
       const DEFAULT_IP_PER_60S = 10;
       const DEFAULT_EMAIL_PER_60S = 3;
       const WINDOW_SECONDS = 60;

       export async function checkRateLimit(input: {
         supabase: ReturnType<typeof createClient>;
         tenant_id: string;
         form_id: string;
         ip: string;
         email?: string;
         overrides?: { ip_per_60s?: number; email_per_60s?: number };
       }): Promise<RateLimitResult> {
         const ipLimit = input.overrides?.ip_per_60s ?? DEFAULT_IP_PER_60S;
         const emailLimit = input.overrides?.email_per_60s ?? DEFAULT_EMAIL_PER_60S;
         const windowStart = new Date(Math.floor(Date.now() / (WINDOW_SECONDS * 1000)) * WINDOW_SECONDS * 1000);

         // IP dimension
         const { data: ipCounter } = await input.supabase
           .from('conversion_rate_limit_counters')
           .upsert(
             { tenant_id: input.tenant_id, form_id: input.form_id, dimension: 'ip', dimension_value: input.ip, window_start: windowStart, submit_count: 1 },
             { onConflict: 'tenant_id,form_id,dimension,dimension_value,window_start', ignoreDuplicates: false }
           )
           .select('submit_count')
           .single();
         // Increment via RPC: SELECT submit_count = submit_count + 1; recommended via dedicated RPC; for simplicity use atomic upsert
         // Implementation note: in Postgres, use INSERT ... ON CONFLICT DO UPDATE SET submit_count = submit_count + 1
         if ((ipCounter?.submit_count ?? 0) > ipLimit) {
           const retryAfter = WINDOW_SECONDS - Math.floor((Date.now() - windowStart.getTime()) / 1000);
           return { allowed: false, retry_after_seconds: retryAfter, dimension_blocked: 'ip' };
         }

         // Email dimension (if provided)
         if (input.email) {
           const { data: emailCounter } = await input.supabase
             .from('conversion_rate_limit_counters')
             .upsert(
               { tenant_id: input.tenant_id, form_id: input.form_id, dimension: 'email', dimension_value: input.email.toLowerCase(), window_start: windowStart, submit_count: 1 },
               { onConflict: 'tenant_id,form_id,dimension,dimension_value,window_start', ignoreDuplicates: false }
             )
             .select('submit_count')
             .single();
           if ((emailCounter?.submit_count ?? 0) > emailLimit) {
             const retryAfter = WINDOW_SECONDS - Math.floor((Date.now() - windowStart.getTime()) / 1000);
             return { allowed: false, retry_after_seconds: retryAfter, dimension_blocked: 'email' };
           }
         }

         return { allowed: true };
       }
       ```

       ```typescript
       export async function handlePublicFormSubmit(input: PublicFormSubmitRequest): Promise<PublicFormSubmitResponse> {
         const { form_id, tenant_id, supabase, req, body, ip, submit_idempotency_key } = input;

         // 1. Load ConversionForm (server-side; bypass RLS via service role for public endpoint)

         if (formError || !form || form.tenant_id !== tenant_id || form.status !== 'published') {
           await appendAuditRow({ tenant_id, event_type: 'public_form_submit_invalid', payload: { form_id, reason: 'form_not_found_or_unpublished' } });
           return { status: 404, body: { error: 'form_not_found' } };

         const botID = await verifyBotID(req);
         if (!botID.allowed) {
           await appendAuditRow({ tenant_id, event_type: 'bot_id_block', payload: { form_id, ip, decision_reason: botID.decision_reason, score: botID.score } });
           return { status: 403, body: { error: 'bot_blocked' } };

         const rateOverrides = form.metadata?.rate_limit;

             body: { error: 'rate_limit_exceeded' },

         const { [honeypotFieldNameForForm(form_id)]: _, ...cleanBody } = body;
         if (!validateInputs(cleanBody)) {
           await appendAuditRow({ tenant_id, event_type: 'public_form_validation_failed', payload: { form_id, errors: validateInputs.errors } });
       ```
