Outputs: jsonb with event_id, source_event_ref, sinks_written. Body performs INSERT into conversion_events (1), cdp_events (2), crm_activity (3) inline; the remaining 4 (identity stitch, ConsentState write, Customer360 update, NBA recompute) are app-side (called by `emit.ts` BEFORE the rpc call OR after, per acceptable transactional boundary):

       ```sql
       CREATE OR REPLACE FUNCTION emit_conversion_event_tx(payload jsonb)
       RETURNS jsonb
       LANGUAGE plpgsql
       SECURITY INVOKER
       AS $$
       DECLARE
         v_event_id        uuid := gen_random_uuid();
         v_source_event_ref uuid := COALESCE((payload->>'source_event_ref')::uuid, gen_random_uuid());
         v_tenant_id       uuid := (payload->>'tenant_id')::uuid;
       BEGIN
         -- Sink 1: conversion_events (D-9)
         INSERT INTO conversion_events (event_id, tenant_id, surface_id, surface_kind, form_id, experiment_variant_id, objective, identity_ref, pricing_context_id, evidence_pack_id, launch_id, occurred_at, source_event_ref)
         VALUES (v_event_id, v_tenant_id, (payload->>'surface_id')::uuid, payload->>'surface_kind', NULLIF(payload->>'form_id','')::uuid, NULLIF(payload->>'experiment_variant_id','')::uuid, payload->>'objective', payload->>'identity_ref', NULLIF(payload->>'pricing_context_id','')::uuid, NULLIF(payload->>'evidence_pack_id','')::uuid, NULLIF(payload->>'launch_id','')::uuid, COALESCE((payload->>'occurred_at')::timestamptz, NOW()), v_source_event_ref);

         -- Sink 2: cdp_events envelope (D-10) — assumes P221 cdp_events table exists; trigger fails in same tx if absent
         INSERT INTO cdp_events (tenant_id, event_domain, event_type, source_event_ref, identity_ref, payload, occurred_at)
         VALUES (v_tenant_id, COALESCE(payload->>'event_domain','website'), 'conversion', v_source_event_ref, payload->>'identity_ref', payload, NOW());

         -- Sink 3: crm_activity (D-22 P222 pattern)
         INSERT INTO crm_activity (tenant_id, source_event_ref, source_domain, commercial_signal, occurred_at, payload)
         VALUES (v_tenant_id, v_source_event_ref, 'website', payload->>'commercial_signal', NOW(), payload);

         -- Audit log (D-49 P201 hash chain)
         INSERT INTO markos_audit_log (tenant_id, event_type, payload_hash, current_hash, occurred_at)
         VALUES (v_tenant_id, 'conversion_event_emit', md5(payload::text), md5(payload::text), NOW());

         RETURN jsonb_build_object(
           'event_id', v_event_id,
           'source_event_ref', v_source_event_ref,
           'sinks_written', jsonb_build_array('conversion_events','cdp_events','crm_activity','markos_audit_log')

       END;
       $$;
       ```

       (c) **D-67 ConsentState DB trigger** — BEFORE INSERT on conversion_events. Rejects when consent_capture_block_id IS NOT NULL AND no matching consent_state row in same tx:
       ```sql
       CREATE OR REPLACE FUNCTION conversion_event_requires_consent_state()
       RETURNS TRIGGER
       LANGUAGE plpgsql
       AS $$
       DECLARE
         v_consent_count int;
       BEGIN
         IF NEW.consent_capture_block_id IS NULL THEN
           RETURN NEW;
         END IF;
         -- Check matching consent_state row exists in same tx (visibility via current xact)
         SELECT COUNT(*) INTO v_consent_count
         FROM consent_state
         WHERE tenant_id = NEW.tenant_id
           AND identity_ref = NEW.identity_ref
           -- channel match implicit per consent_capture_block content
         ;
         IF v_consent_count = 0 THEN
           RAISE EXCEPTION 'conversion_event_requires_consent_state: no consent_state row found for identity_ref % in tenant %', NEW.identity_ref, NEW.tenant_id
             USING ERRCODE = '23P01';
         END IF;
         RETURN NEW;
       END;
       $$;

       CREATE TRIGGER conversion_event_consent_check
         BEFORE INSERT ON conversion_events
         FOR EACH ROW EXECUTE FUNCTION conversion_event_requires_consent_state();
       ```
