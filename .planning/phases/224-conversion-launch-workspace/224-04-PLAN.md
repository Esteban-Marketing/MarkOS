       CREATE OR REPLACE FUNCTION evaluate_gate_persist_audit(
         p_tenant_id UUID,
         p_launch_id UUID,
         p_gate_kind TEXT,
         p_status TEXT,
         p_blocking_reasons TEXT[],
         p_evidence_refs UUID[]
       ) RETURNS UUID
       LANGUAGE plpgsql SECURITY DEFINER AS $$
       DECLARE
         v_gate_id UUID;
       BEGIN
         IF current_setting('app.active_tenant_id', true)::UUID != p_tenant_id THEN
           RAISE EXCEPTION 'tenant_mismatch' USING ERRCODE = '42501';
         END IF;
         INSERT INTO launch_gates(tenant_id, launch_id, gate_kind, status, blocking_reasons, evidence_refs, evaluated_at, updated_at)

       CREATE TABLE IF NOT EXISTS launch_outcomes (
         launch_id          UUID NOT NULL REFERENCES launch_briefs(launch_id),
         tenant_id          UUID NOT NULL REFERENCES markos_tenants(id),
         period_days        INTEGER NOT NULL CHECK (period_days IN (7, 14, 30)),
         reach              INTEGER,

         PRIMARY KEY (launch_id, period_days)
       );
       ALTER TABLE launch_outcomes ENABLE ROW LEVEL SECURITY;
       CREATE POLICY launch_outcomes_tenant_isolation ON launch_outcomes
         USING (tenant_id = current_setting('app.active_tenant_id', true)::UUID);
       CREATE INDEX IF NOT EXISTS idx_launch_outcomes_tenant ON launch_outcomes (tenant_id, computed_at DESC);

       -- Backfill function (Pitfall 8 mitigation)
       CREATE OR REPLACE FUNCTION backfill_launch_id_in_conversion_events(
         p_launch_id UUID,
         p_window_start TIMESTAMPTZ,
         p_window_end TIMESTAMPTZ
       ) RETURNS INTEGER
       LANGUAGE plpgsql SECURITY DEFINER AS $$
       DECLARE
         v_rows_updated INTEGER;
         v_tenant_id UUID;
       BEGIN
         SELECT tenant_id INTO v_tenant_id FROM launch_briefs WHERE launch_id = p_launch_id;
         IF v_tenant_id IS NULL THEN RETURN 0; END IF;
         IF current_setting('app.active_tenant_id', true)::UUID != v_tenant_id THEN
           RAISE EXCEPTION 'tenant_mismatch' USING ERRCODE = '42501';
         END IF;

         -- Identify conversion_events linked to launch_surfaces by surface_id
         WITH launch_surface_ids AS (
           SELECT surface_id FROM launch_surfaces
           WHERE launch_id = p_launch_id AND tenant_id = v_tenant_id
         ),
         eligible AS (
           SELECT ce.event_id
           FROM conversion_events ce
           WHERE ce.tenant_id = v_tenant_id
             AND ce.launch_id IS NULL
             AND ce.occurred_at >= p_window_start AND ce.occurred_at <= p_window_end
             AND ce.surface_id IN (SELECT surface_id FROM launch_surface_ids)
         )
         UPDATE conversion_events SET launch_id = p_launch_id
         WHERE event_id IN (SELECT event_id FROM eligible);
         GET DIAGNOSTICS v_rows_updated = ROW_COUNT;

         INSERT INTO markos_audit_log(tenant_id, event_type, payload)
         VALUES (v_tenant_id, 'launch_outcome_backfill', jsonb_build_object(
           'launch_id', p_launch_id, 'rows_updated', v_rows_updated,
           'window_start', p_window_start, 'window_end', p_window_end
         ));
         RETURN v_rows_updated;
       END;
       $$;

       -- Rollback (operator reference):
       -- DROP FUNCTION IF EXISTS backfill_launch_id_in_conversion_events;
