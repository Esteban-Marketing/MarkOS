       RETURNS TRIGGER
       LANGUAGE plpgsql
       AS $$
       DECLARE
         v_blocking_count int;
       BEGIN
         IF NEW.status = 'published' AND OLD.status <> 'published' THEN
           SELECT COUNT(*) INTO v_blocking_count
           FROM launch_gates
           WHERE launch_id = NEW.launch_id
             AND status = 'blocking';
           IF v_blocking_count > 0 THEN
             RAISE EXCEPTION 'launch_surface_readiness_required: cannot publish surface % when % blocking gate(s) exist for launch %',
               NEW.surface_id, v_blocking_count, NEW.launch_id
               USING ERRCODE = '23P01';
           END IF;
         END IF;
         RETURN NEW;
       END;
       $$;

       CREATE TRIGGER launch_surface_readiness_check
         BEFORE UPDATE ON launch_surfaces
         FOR EACH ROW EXECUTE FUNCTION launch_surface_readiness_required();
