       ```typescript
       export class RenderFailClosedError extends Error {
         constructor(public readonly code: string, message: string) {
           super(message);
         }
       }

       const PRICING_VAR_RE = /\{\{pricing\.([^}]+)\}\}/g;
       const EVIDENCE_VAR_RE = /\{\{evidence\.([^}]+)\}\}/g;
       const PRICING_PLACEHOLDER = '{{MARKOS_PRICING_ENGINE_PENDING}}';
       const EVIDENCE_PLACEHOLDER = '{{MARKOS_EVIDENCE_PENDING}}';

       export interface PricingContext {
         get(path: string): string | null; // path like 'plan_starter.price'
         status: 'approved' | 'draft' | 'archived';
       }
       export interface EvidencePack {
         getClaim(path: string): { value: string; freshness_ok: boolean } | null;
       }

       export function resolveBindings(input: {
         text: string;
         pricing_context: PricingContext | null;
         evidence_pack: EvidencePack | null;
       }): string {
         let out = input.text;

         // Pricing pass
         out = out.replace(PRICING_VAR_RE, (match, path) => {
           if (match === PRICING_PLACEHOLDER) return match;
           if (!input.pricing_context) {
             throw new RenderFailClosedError(
               'unresolved_pricing_binding',
               `pricing binding ${match} requires a pricing_context (received none)`
             );
           }
           if (input.pricing_context.status !== 'approved') {
             throw new RenderFailClosedError(
               'pricing_context_not_approved',
               `pricing binding ${match} requires approved PricingRecommendation; status=${input.pricing_context.status}`
             );
           }
           const value = input.pricing_context.get(path);
           if (value === null) {
             throw new RenderFailClosedError(
               'unresolved_pricing_binding',
               `pricing path ${path} not found in pricing_context`
             );
           }
           return value;
         });

         // Evidence pass
         out = out.replace(EVIDENCE_VAR_RE, (match, path) => {
           if (match === EVIDENCE_PLACEHOLDER) return match;
           if (!input.evidence_pack) {
             throw new RenderFailClosedError(
               'unresolved_evidence_binding',
               `evidence binding ${match} requires an evidence_pack`
             );
           }
           const claim = input.evidence_pack.getClaim(path);
           if (!claim) {
             throw new RenderFailClosedError(
               'unresolved_evidence_binding',
               `evidence path ${path} not found in evidence_pack`
             );
           }
           if (!claim.freshness_ok) {
             throw new RenderFailClosedError(
               'stale_evidence',
               `evidence claim ${path} exceeded TTL`
             );
           }
           return claim.value;
         });

         // Verify no other {{...}} unbound patterns remain (defensive)
         const stillBound = out.match(/\{\{[^}]+\}\}/g);
         if (stillBound) {
           const offenders = stillBound.filter(s => s !== PRICING_PLACEHOLDER && s !== EVIDENCE_PLACEHOLDER);
           if (offenders.length > 0) {
             throw new RenderFailClosedError(
               'unknown_template_variable',
               `unrecognized template variables: ${offenders.join(', ')}`
             );
           }
         }

         return out;
       }

       export function resolveBlockBindings(
         block: { [k: string]: any; block_type: string },
         pricing_context: PricingContext | null,
         evidence_pack: EvidencePack | null
       ): typeof block {
         const out: any = { ...block };
         for (const key of Object.keys(out)) {
           if (typeof out[key] === 'string' && (PRICING_VAR_RE.test(out[key]) || EVIDENCE_VAR_RE.test(out[key]) || out[key].includes('{{'))) {
             out[key] = resolveBindings({ text: out[key], pricing_context, evidence_pack });
           }
         }
         return out;
       }
       ```

       ```typescript
       export function pageCacheTag(tenant_id: string, page_id: string): string {
         return `tenant:${tenant_id}:page:${page_id}`;
       }

       /**
        * Synchronously invalidate the ISR cache for a page (Pitfall 3).
        * MUST be called from publish/archive route handlers BEFORE returning 200.
        * Falls back to revalidatePath if Next.js < 13.5 (A11/A16).
        */
       export async function invalidatePageCache(tenant_id: string, page_id: string, slug: string): Promise<void> {
         try {
           const cacheModule = await import('next/cache');
           if ('updateTag' in cacheModule) {
             // @ts-expect-error — Next.js 13.5+ API
             await cacheModule.updateTag(pageCacheTag(tenant_id, page_id));
           } else if ('revalidateTag' in cacheModule) {
             // @ts-expect-error — Next.js 13.4 API
             cacheModule.revalidateTag(pageCacheTag(tenant_id, page_id));
           } else if ('revalidatePath' in cacheModule) {
             // @ts-expect-error — fallback for very old Next.js
             cacheModule.revalidatePath(`/${slug}`);
           }
         } catch (err) {
           // Fail-closed: if cache cannot be invalidated, propagate error so route handler returns 500
           throw new Error(`isr_cache_invalidation_failed: ${(err as Error).message}`);
         }
       }
       ```

       ```typescript
         pattern: 'currency' | 'claim';
         match: string;
       }

       export function scanContentBlocksForUnboundClaims(blocks: Array<{ block_type: string; [k: string]: any }>): ClassifierMatch[] {
         const matches: ClassifierMatch[] = [];
         for (let i = 0; i < blocks.length; i++) {
           const block = blocks[i];
           for (const [field, value] of Object.entries(block)) {
             if (typeof value !== 'string') continue;
             if (field.endsWith('_binding')) continue; // Bindings handled by binding-resolver
             // If the text already contains a pricing/evidence binding token, treat as bound
             const hasPricingBinding = PRICING_BINDING_RE.test(value);
             const hasEvidenceBinding = EVIDENCE_BINDING_RE.test(value);

             // Currency check (skip if pricing binding present in same text)
             if (!hasPricingBinding) {
               for (const re of CURRENCY_PATTERNS) {
                 const m = value.match(re);
                 if (m) {
                   matches.push({ block_index: i, block_type: block.block_type, field, snippet: value.slice(0, 200), pattern: 'currency', match: m[0] });
                 }
               }
             }
             // Claim check (skip if evidence binding present)
             if (!hasEvidenceBinding) {
               for (const re of CLAIM_PATTERNS) {
                 const m = value.match(re);
                 if (m) {
                   matches.push({ block_index: i, block_type: block.block_type, field, snippet: value.slice(0, 200), pattern: 'claim', match: m[0] });
                 }
               }
             }
           }
         }
         return matches;
       }

       export function scanFormFieldsForUnboundCopy(form: { fields: Array<{ name: string; label?: string; help_text?: string }> }): ClassifierMatch[] {
         const matches: ClassifierMatch[] = [];
         for (let i = 0; i < form.fields.length; i++) {
           const f = form.fields[i];
           const checkFields = [f.label, f.help_text].filter((s): s is string => typeof s === 'string');
           for (const text of checkFields) {
             const hasPricing = PRICING_BINDING_RE.test(text);
             if (!hasPricing) {
               for (const re of CURRENCY_PATTERNS) {
                 const m = text.match(re);
                 if (m) matches.push({ block_index: i, block_type: 'form_field', field: f.name, snippet: text, pattern: 'currency', match: m[0] });
               }
             }
           }
         }
         return matches;
       }
       ```

       ```typescript
         const tenant_id = req.headers.get('x-markos-tenant-id') ?? ''; // resolved by middleware

       }
       ```
