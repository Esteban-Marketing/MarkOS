       ```ts
       const PRICING_PLACEHOLDER = '{{MARKOS_PRICING_ENGINE_PENDING}}';
       const EVIDENCE_PLACEHOLDER = '{{MARKOS_EVIDENCE_PENDING}}';
       const VAR_PATTERN = /\{\{(pricing|evidence)\.([a-z0-9_]+)\}\}/gi;

       export interface BindingContext {
         pricing_context?: Record<string, string>;
         evidence_map?: Record<string, string>;
       }
       export interface BindingResult {
         resolved: string;
         unresolved_variables: string[];
       }

       export function resolveBindings(text: string, ctx: BindingContext): BindingResult {
         if (text === PRICING_PLACEHOLDER || text === EVIDENCE_PLACEHOLDER) {
           return { resolved: text, unresolved_variables: [] };
         }
         const unresolved: string[] = [];
         const out = text.replace(VAR_PATTERN, (match, kind, key) => {
           const value = kind === 'pricing' ? ctx.pricing_context?.[key] : ctx.evidence_map?.[key];
           if (value === undefined) {
             unresolved.push(match);
             return match;
           }
           return value;
         });
         return { resolved: out, unresolved_variables: unresolved };
       }
       ```

       ```ts
       export class StalePricingError extends Error { constructor(public pricing_context_id: string) { super(`stale pricing_context_id: ${pricing_context_id}`); } }
       export class StaleEvidenceError extends Error { constructor(public evidence_pack_id: string) { super(`stale evidence_pack_id: ${evidence_pack_id}`); } }

       export async function assertPricingFresh(pricing_context_id: string | null | undefined, supabase: SupabaseClient): Promise<void> {
         if (!pricing_context_id) return;
         const { data, error } = await supabase
           .from('pricing_recommendations')
           .select('status, archived_at, superseded_at')
           .eq('pricing_context_id', pricing_context_id)
           .single();
         if (error || !data || data.status === 'archived' || data.archived_at || data.superseded_at) {
           throw new StalePricingError(pricing_context_id);
         }
       }
       export async function assertEvidenceFresh(evidence_pack_id: string | null | undefined, supabase: SupabaseClient): Promise<void> {
         if (!evidence_pack_id) return;
         const { data, error } = await supabase
           .from('evidence_packs')
           .select('claims_status, ttl_expires_at')
           .eq('evidence_pack_id', evidence_pack_id)
           .single();
         if (error || !data || data.claims_status !== 'fresh' || (data.ttl_expires_at && new Date(data.ttl_expires_at) < new Date())) {
           throw new StaleEvidenceError(evidence_pack_id);
         }
       }
       ```
