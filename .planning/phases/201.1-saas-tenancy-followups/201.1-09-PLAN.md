```javascript
'use strict';

// Phase 201.1 D-107 (closes M2): BYOD soft-failure grace window helpers.

const BYOD_GRACE_WINDOW_MS = 86_400_000; // 24 hours

function gracePathEnabled() {
  return process.env.MARKOS_BYOD_GRACE_DISABLED !== '1';
}

function isWithinByodGraceWindow(row, now) {
  if (!gracePathEnabled()) return false;
  if (!row || row.status !== 'failed') return false;
  if (!row.last_verified_at) return false;
  const lastVerifiedMs = typeof row.last_verified_at === 'string'
    ? new Date(row.last_verified_at).getTime()
    : (row.last_verified_at && typeof row.last_verified_at.getTime === 'function')
      ? row.last_verified_at.getTime()
      : 0;
  if (!Number.isFinite(lastVerifiedMs) || lastVerifiedMs <= 0) return false;
  const delta = (typeof now === 'number' ? now : Date.now()) - lastVerifiedMs;
  return delta < BYOD_GRACE_WINDOW_MS;
}

module.exports = { BYOD_GRACE_WINDOW_MS, gracePathEnabled, isWithinByodGraceWindow };
```

```javascript
async function resolveTenantByDomain(client, host, opts = {}) {
  if (!host) return null;
  const allowGrace = opts && opts.allowGrace === true;

  let query = client.from('markos_custom_domains')
    .select('domain, tenant_id, org_id, status, verified_at, last_verified_at')
    .eq('domain', host);

  if (allowGrace) {
    // Match verified OR failed-within-24h grace.
    query = query.or(`status.eq.verified,and(status.eq.failed,last_verified_at.gte.${new Date(Date.now() - 86400000).toISOString()})`);
  } else {
    query = query.eq('status', 'verified');
  }

  const { data } = await query.maybeSingle();
  if (!data) return null;

  const withinGrace = data.status === 'failed' && data.last_verified_at &&
    (Date.now() - new Date(data.last_verified_at).getTime()) < 86400000;

  return {
    tenant_id: data.tenant_id,
    org_id: data.org_id,
    domain: data.domain,
    withinGrace,
    status: data.status,
  };
}
```

```javascript
'use strict';

// Phase 201.1 D-107 (closes M2): BYOD verification-lost alert via existing webhook engine.
//
// W-4 dependency note: the rows inserted into markos_webhook_deliveries with
// status='queued' below are consumed by a Phase 203 dispatch worker. If that
// worker is not yet deployed when 201.1 ships, alerts queue silently. The
// runbook (docs/byod-failure-alert.md or this plan's runbook section) surfaces
// the dependency. End-to-end delivery is verified by the v4.0.0 staging-smoke
// harness (Plan 11 / D-108).

const { enqueueAuditStaging } = require('../audit/writer.cjs');

const EVENT_TYPE = 'byod.verification_lost';

async function fireByodFailureAlert(client, input) {
  const { domain, tenant_id, org_id, last_verified_at, reason } = input || {};
  if (!domain || !tenant_id) {
    throw new Error('fireByodFailureAlert: domain + tenant_id required');
  }

  // 1. Audit emit (always — even if no webhook subscriptions exist).
  try {
    await enqueueAuditStaging(client, {
      tenant_id, org_id: org_id || null,
      source_domain: 'tenancy',
      action: EVENT_TYPE,
      actor_id: 'system',
      actor_role: 'system',
      payload: { domain, last_verified_at, reason: reason || 'unknown' },
    });
  } catch { /* fail-soft */ }

  // 2. Look up subscribed webhook endpoints for this tenant + event.
  const { data: subs } = await client
    .from('markos_webhook_subscriptions')
    .select('id, url, secret, events')
    .eq('tenant_id', tenant_id)
    .eq('active', true);

  const matching = (subs || []).filter((s) => Array.isArray(s.events) && s.events.includes(EVENT_TYPE));
  if (matching.length === 0) return { delivered: 0, audited: true };

  // 3. Enqueue delivery via the existing Phase 203 webhook delivery substrate.
  // This plan REUSES the existing markos_webhook_deliveries table or queue path —
  // it does NOT re-implement delivery infrastructure.
  // *** W-4 ***: the resulting rows have status='queued'. Actual delivery requires
  // Phase 203 dispatch worker to be deployed. If not, rows queue silently.
```
