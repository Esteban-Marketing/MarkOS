<objective>
Harden the edge-config slug cache against (a) cache stampede on TTL expiry, (b) cross-region propagation race on rename, and (c) thundering-herd on cache miss. Closes review concern **H5** ("60s TTL: at expiry, N concurrent middleware invocations miss → all hit DB. Worse: rename 'synchronous invalidate' — Vercel Edge Config propagation is not instant across regions; race window allows old slug to resolve to new tenant.").

Purpose: Phase 201 ships `lib/markos/tenant/slug-cache.cjs` (94 LoC) with a fixed 60s TTL and a synchronous edge-config invalidate on tenant slug rename. Per CONTEXT D-104:
- **Jittered TTL** (write-time, not read-time): each entry gets `45 + random(0, 30)` seconds so cohort-expiry stampedes are eliminated.
- **Single-flight per instance**: a Vercel function instance kept warm for multiple requests should not issue N parallel Supabase queries for the same slug; coalesce them into one Promise.
- **Transitional 410 on rename**: between rename time and edge-config global propagation (~tens of seconds), `markos:slug:{old}` is pinned to the literal `__renamed:{new}` for 90s. Middleware sees this and returns HTTP 410 with `Location: <new-canonical-host>` + `Sunset` header.
- **Read-after-write verification**: rename polls edge-config 3 times (50/100/200ms backoff) before returning, ensuring the transitional pin is visible to the calling region.

Output: (1) Jittered-TTL helper added to `slug-cache.cjs`; (2) new `single-flight.cjs` helper; (3) `renameTenantSlug` augmented in `tenants.cjs`; (4) `middleware.ts` patched to detect `__renamed:` prefix and return 410; (5) 3 dedicated test files covering jitter, single-flight, and rename-410 flows.
</objective>

```javascript
// Phase 201.1 D-104 (closes H5): jittered TTL + transitional-rename helpers.

const SLUG_CACHE_TTL_BASE_SECONDS = 45;
const SLUG_CACHE_TTL_JITTER_SECONDS = 30;
const TRANSITIONAL_RENAME_TTL_SECONDS = 90;
const TRANSITIONAL_PREFIX = '__renamed:';

function computeJitteredTtl() {
  return SLUG_CACHE_TTL_BASE_SECONDS + Math.floor(Math.random() * (SLUG_CACHE_TTL_JITTER_SECONDS + 1));
}

async function writeSlugToEdgeJittered(slug, tenantId, deps) {
  const ttl = computeJitteredTtl();
  return writeSlugToEdgeRaw(slug, tenantId, ttl, deps);
}

async function writeSlugToEdgeRaw(slug, value, ttlSeconds, deps) {
  const token = (deps && deps.VERCEL_API_TOKEN) || process.env.VERCEL_API_TOKEN;
  const edgeConfigId = (deps && deps.EDGE_CONFIG_ID) || process.env.EDGE_CONFIG_ID;
  if (!token || !edgeConfigId) return;
  try {
    const fetchImpl = (deps && deps.fetch) || global.fetch;
    if (typeof fetchImpl !== 'function') return;
    const item = { operation: 'upsert', key: keyFor(slug), value };
    if (typeof ttlSeconds === 'number' && ttlSeconds > 0) item.ttl = ttlSeconds;
    await fetchImpl(`https://api.vercel.com/v1/edge-config/${edgeConfigId}/items`, {
      method: 'PATCH',
      headers: {
        Authorization: `Bearer ${token}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ items: [item] }),
    });
  } catch { /* fail soft */ }
}

async function writeTransitionalRename(oldSlug, newSlug, deps) {
  if (!oldSlug || !newSlug) return;
  return writeSlugToEdgeRaw(oldSlug, `${TRANSITIONAL_PREFIX}${newSlug}`, TRANSITIONAL_RENAME_TTL_SECONDS, deps);
}

module.exports.SLUG_CACHE_TTL_BASE_SECONDS = SLUG_CACHE_TTL_BASE_SECONDS;
module.exports.SLUG_CACHE_TTL_JITTER_SECONDS = SLUG_CACHE_TTL_JITTER_SECONDS;
module.exports.TRANSITIONAL_RENAME_TTL_SECONDS = TRANSITIONAL_RENAME_TTL_SECONDS;
module.exports.TRANSITIONAL_PREFIX = TRANSITIONAL_PREFIX;
module.exports.computeJitteredTtl = computeJitteredTtl;
module.exports.writeSlugToEdgeJittered = writeSlugToEdgeJittered;
module.exports.writeTransitionalRename = writeTransitionalRename;
```

```javascript
'use strict';

// Phase 201.1 D-104: per-instance Promise coalescing for slug→tenant lookups.
// Eliminates thundering-herd on cache miss within one warm Vercel function invocation.

function createSingleFlight() {
  const inflight = new Map();
  return {
    async coalesce(key, fn) {
      if (inflight.has(key)) return inflight.get(key);
      const promise = Promise.resolve()
        .then(() => fn())
        .finally(() => { inflight.delete(key); });
      inflight.set(key, promise);
      return promise;
    },
    inflightSize() { return inflight.size; },
    clear() { inflight.clear(); },
  };
}

const _default = createSingleFlight();

module.exports = {
  createSingleFlight,
  singleFlight: _default,
};
```

2. Open `lib/markos/orgs/tenants.cjs`. If `renameTenantSlug` already exists, augment its body with the 4-step D-104 flow. If it does NOT exist, add it:

```javascript
const { writeTransitionalRename, writeSlugToEdgeJittered, readSlugFromEdge, invalidateSlug, TRANSITIONAL_PREFIX } = require('../tenant/slug-cache.cjs');
const { enqueueAuditStaging } = require('../audit/writer.cjs');

const RENAME_POLL_DELAYS_MS = [50, 100, 200];

async function renameTenantSlug(client, input, deps = {}) {
  const { tenant_id, old_slug, new_slug, actor_id, org_id } = input || {};
  if (!tenant_id || !old_slug || !new_slug || !actor_id) {
    throw new Error('renameTenantSlug: tenant_id + old_slug + new_slug + actor_id required');
  }
  if (old_slug === new_slug) return { ok: true, no_op: true };

  // 1. UPDATE the canonical row.
  const { error: upErr } = await client
    .from('markos_tenants')
    .update({ slug: new_slug, updated_at: new Date().toISOString() })
    .eq('id', tenant_id);
  if (upErr) throw new Error(`renameTenantSlug: update failed: ${upErr.message}`);

  // 2. Pin transitional 410 marker on old slug.
  await writeTransitionalRename(old_slug, new_slug, deps);

  // 3. Backfill jittered cache entry for new slug.
  await writeSlugToEdgeJittered(new_slug, tenant_id, deps);

  // 4. Read-after-write verification: poll until the transitional pin is visible.
  let confirmed = false;
  for (const delay of RENAME_POLL_DELAYS_MS) {
    await new Promise((r) => setTimeout(r, delay));
    const v = await readSlugFromEdge(old_slug, deps);
    if (typeof v === 'string' && v.startsWith(TRANSITIONAL_PREFIX)) {
      confirmed = true;
      break;
    }
  }

  // 5. Emit audit row.
  try {
    await enqueueAuditStaging(client, {
      tenant_id, org_id: org_id || null,
      source_domain: 'tenancy',
      action: 'tenant_slug.renamed',
      actor_id, actor_role: 'owner',
      payload: { old_slug, new_slug, transitional_pin_confirmed: confirmed },
    });
  } catch { /* noop — audit drain catches staged rows */ }

  return { ok: true, transitional_pin_confirmed: confirmed };
}

module.exports.renameTenantSlug = renameTenantSlug;
```
