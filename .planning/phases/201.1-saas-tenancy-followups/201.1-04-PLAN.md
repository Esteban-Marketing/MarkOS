<objective>
Move audit emit for `api/approve.js` and `api/submit.js` from the post-`res.end` wrapper hook to inline-before-response, AND make the emit fail-CLOSED. Closes review concern **H1** ("Audit emit AFTER res.end — if emit throws/network-fails, client got 200 but audit row missing. Hash chain silently skips an event. For approval/submit flows this is regulatory-grade evidence loss").

Purpose: Phase 201's `api/approve.js` (61 LoC) and `api/submit.js` (57 LoC) hook `res.end` to capture status + body, then call `enqueueAuditStaging` AFTER the response is already written. Per CONTEXT D-101: this footgun must be eliminated. The audit emit must happen BEFORE the response so that if staging insert fails, the handler returns 500 and the client knows the action wasn't fully recorded. Webhook engine emit (`engine.cjs::emitWebhooksAudit`) is already inline but currently fail-soft via try/catch — D-101 adds an opt-in fail-closed mode and a regression test that asserts an injected staging failure fails the business write.

Output: (1) New shared helper `lib/markos/audit/inline-emit.cjs::emitInlineApprovalAudit` that calls `enqueueAuditStaging` and re-throws on error; (2) rewritten `api/approve.js` + `api/submit.js` that call handleApprove/handleSubmit, capture the return value, then emit AND return 500 if emit fails; (3) fail-closed mode added to `webhooks/engine.cjs::emitWebhooksAudit`; (4) regression test injecting a faulty audit client and asserting 500 response.

**Same-txn caveat:** As 201.1-CONTEXT.md `<specifics>` notes, the underlying `handleApprove`/`handleSubmit` (in `onboarding/backend/handlers.cjs`, ~3000 LoC, NOT touched by D-101) write to potentially different schemas/Supabase clients than the audit staging table. A truly atomic "business + audit in one txn" requires refactoring that handler — out of scope for this delta phase. D-101 instead implements the **next-best fail-closed semantics**: emit-before-response with explicit error path, plus a documented "compensation marker" payload field on the staging row noting that the business write SUCCEEDED but its audit emit was queued separately. The regression test makes this contract explicit.
</objective>

const { emitInlineApprovalAudit } = require('../lib/markos/audit/inline-emit.cjs');

module.exports = async function handler(req, res) {
  if (req.method === 'OPTIONS') return handleCorsPreflight(req, res);

  // 1. Run business handler. It writes the row but is NOT yet finalized to the client.
  //    We use a deferred-end pattern: capture what handleApprove WOULD send, but defer res.end.
  const captured = await runWithDeferredEnd(req, res, handleApprove);
  // captured = { status, headers, body }

  // 2. Only emit audit if business write succeeded.
  if (captured.status >= 200 && captured.status < 300) {
    try {
      await emitInlineApprovalAudit(req, captured, { action: 'approve' });
    } catch (err) {
      // Fail-CLOSED: business write happened on the server, but client did NOT yet receive 200.
      // Return 500 + carry the error so the caller knows the audit row is missing.
      res.statusCode = 500;
      res.setHeader('Content-Type', 'application/json');
      return res.end(JSON.stringify({
        error: 'audit_emit_failed',
        detail: err.message,
        compensation: 'business_write_succeeded_audit_pending',
      }));
    }
  }

  // 3. Replay captured response to the client.
  res.statusCode = captured.status;
  for (const [k, v] of captured.headers) res.setHeader(k, v);
  return res.end(captured.body);
};
```

'use strict';

// Phase 201.1 D-101 (closes H1): inline audit emit for approve/submit flows.
// Replaces the post-res.end wrapper in api/approve.js + api/submit.js.
// Fail-CLOSED: errors propagate; the caller decides the response code.

const { enqueueAuditStaging } = require('./writer.cjs');

function runWithDeferredEnd(req, res, businessHandler) {
  return new Promise((resolve, reject) => {
    const headers = [];
    let bodyChunks = [];
    let status = 200;
    let ended = false;

    const wrappedRes = Object.create(res);
    Object.defineProperty(wrappedRes, 'statusCode', {
      get() { return status; },
      set(v) { status = v; },
    });
    wrappedRes.setHeader = (k, v) => { headers.push([k, v]); };
    wrappedRes.getHeader = (k) => {
      const found = headers.find(([h]) => String(h).toLowerCase() === String(k).toLowerCase());
      return found ? found[1] : undefined;
    };
    wrappedRes.removeHeader = (k) => {
      for (let i = headers.length - 1; i >= 0; i--) {
        if (String(headers[i][0]).toLowerCase() === String(k).toLowerCase()) headers.splice(i, 1);
      }
    };
    wrappedRes.write = (chunk) => {
      if (chunk == null) return true;
      bodyChunks.push(typeof chunk === 'string' ? chunk : Buffer.isBuffer(chunk) ? chunk.toString('utf8') : String(chunk));
      return true;
    };
    wrappedRes.end = (chunk, ...rest) => {
      if (ended) return wrappedRes;
      ended = true;
      if (chunk != null) wrappedRes.write(chunk);
      const body = bodyChunks.join('');
      resolve({ status, headers, body });
      return wrappedRes;
    };

    Promise.resolve()
      .then(() => businessHandler(req, wrappedRes))
      .catch((err) => {
        if (!ended) reject(err);
      });
  });
}

function loadSupabaseClient(opts) {
  if (opts && opts.client) return opts.client;
  const { createClient } = require('@supabase/supabase-js');
  return createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL || 'https://example.supabase.co',
    process.env.SUPABASE_SERVICE_ROLE_KEY || 'anon-key',
    { auth: { persistSession: false } },
  );
}

async function emitInlineApprovalAudit(req, captured, opts) {
  if (!opts || (opts.action !== 'approve' && opts.action !== 'submit')) {
    throw new Error("emitInlineApprovalAudit: opts.action must be 'approve' or 'submit'");
  }
  const headers = req.headers || {};
  const tenant_id = headers['x-markos-tenant-id'] || 'unknown';
  const actor_id = headers['x-markos-user-id'] || 'system';

  let parsed = {};
  try { parsed = captured.body ? JSON.parse(captured.body) : {}; } catch { parsed = {}; }

  let action;
  let payload;
  if (opts.action === 'approve') {
    const decision = parsed && parsed.decision && String(parsed.decision).toLowerCase() === 'rejected'
      ? 'rejected' : 'approved';
    action = decision === 'rejected' ? 'approval.rejected' : 'approval.approved';
    payload = { approval_id: parsed.approval_id || parsed.run_id || null, decision };
  } else { // submit
    action = 'approval.submitted';
    payload = {
      run_id: parsed.run_id || parsed.runId || null,
      project_slug: parsed.project_slug || parsed.slug || null,
    };
  }

  const client = loadSupabaseClient(opts);
  // enqueueAuditStaging THROWS on failure — that propagates up to the handler.
  await enqueueAuditStaging(client, {
    tenant_id, org_id: null,
    source_domain: 'approvals',
    action, actor_id, actor_role: 'owner',
    payload,
  });
}

module.exports = { runWithDeferredEnd, emitInlineApprovalAudit };
```
