<objective>
Lock the canonical-JSON serialization spec for the per-tenant SHA-256 hash chain so that Node-side `lib/markos/audit/canonical.cjs` and Postgres-side `markos_canonicalize_audit_payload(jsonb)` produce byte-identical output for ALL well-formed audit payloads. Closes review concern **H4** (canonicalization under-specified — single byte divergence breaks tamper detection across the chain).

Purpose: Phase 201 shipped a working hash chain whose canonicalization is implicitly defined by `JSON.stringify` (Node) and inline string concatenation (Postgres `append_markos_audit_row`, migration 82 lines 130-137). Cross-AI review (H4) flagged that number formatting, Unicode normalization, control-char escaping, and key-ordering rules are NOT pinned. A future payload with `1e21`, `é` (é decomposed vs composed), `/`, or a tab character could produce different bytes on Node vs Postgres → false-positive tamper detection on the verifier.

Output: (1) Standalone Postgres function `markos_canonicalize_audit_payload(jsonb) returns text` that implements the locked spec; (2) hardened `lib/markos/audit/canonical.cjs` with NFC normalization + ECMA-262 ToString(Number) + lowercase `\uXXXX` escapes + binary-lex key sort; (3) 10k-iteration property-based fuzzer comparing Node vs Postgres outputs; (4) human-readable spec doc `docs/canonical-audit-spec.md`; (5) backward-compat assertion that pre-D-103 rows still verify (no chain-break from spec lock).
</objective>

```sql
-- Phase 201.1 D-103: lock canonical-JSON spec for the audit hash chain.
-- Closes review concern H4: published byte-parity spec for Node ↔ Postgres canonicalization.
-- See docs/canonical-audit-spec.md for the full specification.

-- ============================================================================
-- 1. markos_canonicalize_audit_payload(jsonb) — standalone canonicalizer
--
-- Mirrors lib/markos/audit/canonical.cjs::canonicalJson exactly:
--   - NFC normalization on string VALUES (keys preserved as bytes)
--   - ECMA-262 §7.1.13 ToString(Number) (Postgres jsonb numeric formatting)
--   - lowercase \uXXXX escapes for control chars, no forward-slash escape
--   - binary-lexicographic key sort (UTF-8 bytes, equivalent to default jsonb_object_keys order)
-- ============================================================================
create or replace function markos_canonicalize_audit_payload(p_payload jsonb)
  returns text
  language plpgsql
  immutable
  parallel safe
as $$
declare
  v_result text;
begin
  if p_payload is null then
    return 'null';
  end if;

  -- Postgres jsonb already stores keys sorted by length-then-bytewise (rfc7159 violation
  -- but stable). To get binary-lex on UTF-8 bytes we re-serialize via a recursive walker.
  -- For Phase 201.1 simplicity we lean on jsonb_strip_nulls + the existing jsonb output
  -- and add an explicit recursive normalization helper.
  v_result := markos_canonicalize_jsonb_recursive(p_payload);
  return v_result;
end;
$$;

-- ============================================================================
-- 2. Recursive walker (private helper) — handles arrays, objects, primitives.
-- ============================================================================
create or replace function markos_canonicalize_jsonb_recursive(p_value jsonb)
  returns text
  language plpgsql
  immutable
  parallel safe
as $$
declare
  v_typ text := jsonb_typeof(p_value);
  v_keys text[];
  v_key text;
  v_parts text[] := '{}';
  v_idx int;
begin
  if v_typ = 'null' then
    return 'null';
  elsif v_typ = 'boolean' or v_typ = 'number' then
    return p_value::text;
  elsif v_typ = 'string' then
    -- NFC normalize the value string, then re-encode as JSON string.
    return to_jsonb(normalize(p_value #>> '{}', NFC))::text;
  elsif v_typ = 'array' then
    for v_idx in 0 .. (jsonb_array_length(p_value) - 1) loop
      v_parts := array_append(v_parts, markos_canonicalize_jsonb_recursive(p_value -> v_idx));
    end loop;
    return '[' || array_to_string(v_parts, ',') || ']';
  elsif v_typ = 'object' then
    select array_agg(k order by convert_to(k, 'UTF8')) into v_keys
      from jsonb_object_keys(p_value) k;
    if v_keys is null then
      return '{}';
    end if;
    foreach v_key in array v_keys loop
      v_parts := array_append(v_parts, to_jsonb(v_key)::text || ':' || markos_canonicalize_jsonb_recursive(p_payload -> v_key));
    end loop;
    return '{' || array_to_string(v_parts, ',') || '}';
  end if;
  return 'null';
end;
$$;

-- 3. Re-wire append_markos_audit_row to use the new canonicalizer for the
--    payload sub-object. The OUTER 6-key envelope (action, actor_id, actor_role,
--    occurred_at, payload, tenant_id) is built the same way as in migration 82
--    BUT the inner payload string now comes from markos_canonicalize_audit_payload(p_payload)
--    instead of `coalesce(p_payload::text, 'null')`.
--
-- Backward compatibility note: existing pre-D-103 rows have row_hash computed
-- against the migration-82 inline canonicalizer. For payloads that are PURE
-- ascii without controls and without compound Unicode, the new canonicalizer
-- produces byte-identical output. Migration 90 therefore preserves chain
-- continuity for existing rows. The accompanying recanonicalize-on-mismatch
-- procedure documented in 201.1-CONTEXT.md §phase_201_substrate_constraints
-- and Plan 06 (D-106) handles any pre-existing rows that fail the new spec.
-- ============================================================================
create or replace function append_markos_audit_row(
  p_tenant_id text,
  p_org_id text,
  p_source_domain text,
  p_action text,
  p_actor_id text,
  p_actor_role text,
  p_payload jsonb,
  p_occurred_at timestamptz default now()
) returns table (id bigint, row_hash text, prev_hash text)
language plpgsql
as $$
declare
  v_prev_hash text;
  v_canonical text;
  v_row_hash text;
  v_id bigint;
  v_payload_canonical text;
begin
  perform pg_advisory_xact_lock(hashtext('audit:' || p_tenant_id));

  select markos_audit_log.row_hash into v_prev_hash
    from markos_audit_log
    where markos_audit_log.tenant_id = p_tenant_id
    order by markos_audit_log.id desc
    limit 1;

  if v_prev_hash is null or v_prev_hash = '' then
    v_prev_hash := encode(digest('genesis:' || p_tenant_id, 'sha256'), 'hex');
  end if;

  -- D-103: payload is canonicalized via the locked spec.
  v_payload_canonical := markos_canonicalize_audit_payload(coalesce(p_payload, '{}'::jsonb));

  v_canonical :=
    '{"action":'      || to_jsonb(p_action)::text
    || ',"actor_id":' || to_jsonb(p_actor_id)::text
    || ',"actor_role":' || to_jsonb(p_actor_role)::text
    || ',"occurred_at":' || to_jsonb(to_char(p_occurred_at at time zone 'utc', 'YYYY-MM-DD"T"HH24:MI:SS.MS"Z"'))::text
    || ',"payload":'  || v_payload_canonical
    || ',"tenant_id":' || to_jsonb(p_tenant_id)::text
    || '}';

  v_row_hash := encode(digest(v_prev_hash || v_canonical, 'sha256'), 'hex');

  insert into markos_audit_log (
    tenant_id, org_id, source_domain, action, actor_id, actor_role,
    payload, prev_hash, row_hash, occurred_at,
    workspace_id, actor, entity_type, entity_id, details
  ) values (
    p_tenant_id, p_org_id, p_source_domain, p_action, p_actor_id, p_actor_role,
    coalesce(p_payload, '{}'::jsonb), v_prev_hash, v_row_hash, p_occurred_at,
    (select workspace_id from markos_tenants where markos_tenants.id = p_tenant_id),
    p_actor_id, p_source_domain, p_action, coalesce(p_payload, '{}'::jsonb)
  )
  returning markos_audit_log.id into v_id;

  id := v_id;
  row_hash := v_row_hash;
  prev_hash := v_prev_hash;
  return next;
end;
$$;
```
