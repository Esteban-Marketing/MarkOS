<objective>
Harden the demo sandbox with BotID gate + ephemeral signed token + restricted tool subset + per-token total-cost cap. Closes review concern **H5**: "Demo sandbox abuse vector — Guest rate-limit per IP trivially defeated by residential proxy pools. Demo proxies into MCP server which has no auth (H4). Net: unauthenticated, IP-rate-limited LLM endpoint published on markos.dev. LLM-cost DoS waiting to happen."

Purpose: Phase 200-08 shipped a demo sandbox at `app/(marketing)/integrations/claude/demo/` that proxies into the MCP server with only IP-rate-limit. With Plan 200.1-07 closing H4 (MCP now requires per-tenant bearer auth), the demo becomes structurally broken — anonymous demo traffic doesn't have a tenant. D-205 closes H5 with a 4-layer defense:

1. **BotID gate at token issue:** `lib/markos/auth/botid.cjs` (fail-closed, Phase 201) verifies the BotID token before any demo session begins.
2. **Ephemeral signed token:** server issues a 15-min JWT bound to a synthetic sandbox tenant, with `aud='demo-sandbox'`.
3. **Restricted tool subset:** token's `allowed_tools` claim hardcodes `['draft_message','audit_claim']` (read/propose-only — matches Decision 9 tiered autonomy). UI disables disallowed tool buttons; server-side rejects regardless of UI.
4. **Per-token total-cost cap $0.50 (50 cents):** every invoke increments a per-token cost counter; when sum exceeds 50, server returns 402.

Output: 1 lib module + 2 Next.js route handlers (issue-token + invoke) + 1 demo page update + 1 abuse test.
</objective>

    ```javascript
    'use strict';
    const crypto = require('node:crypto');
    const { verifyBotIdToken } = require('../auth/botid.cjs');
    const { enqueueAuditStaging } = require('../audit/writer.cjs');

    const DEMO_TOKEN_TTL_SECONDS = 900;
    const DEMO_COST_CAP_CENTS = 50;
    const DEMO_ALLOWED_TOOLS = Object.freeze(['draft_message', 'audit_claim']);
    const DEMO_SYNTHETIC_TENANT_ID = '00000000-0000-0000-0000-000000000001';
    const DEMO_AUDIENCE = 'demo-sandbox';
    const DEMO_ISSUER = 'markos.dev';

    function getSecret() {
      const s = process.env.MARKOS_DEMO_TOKEN_SECRET;
      if (!s || s.length < 32) throw new Error('MARKOS_DEMO_TOKEN_SECRET missing or <32 chars');
      return s;
    }

    function b64url(input) {
      return Buffer.from(typeof input === 'string' ? input : JSON.stringify(input)).toString('base64url');
    }
    function b64urlDecode(s) {
      return JSON.parse(Buffer.from(s, 'base64url').toString('utf8'));
    }

    function signJwt(payload) {
      const header = { alg: 'HS256', typ: 'JWT' };
      const head = b64url(header);
      const body = b64url(payload);
      const sig = crypto.createHmac('sha256', getSecret()).update(head + '.' + body).digest('base64url');
      return head + '.' + body + '.' + sig;
    }

    function verifyJwt(token) {
      if (typeof token !== 'string') return { ok: false, reason: 'malformed' };
      const parts = token.split('.');
      if (parts.length !== 3) return { ok: false, reason: 'malformed' };
      const [head, body, sig] = parts;
      const expected = crypto.createHmac('sha256', getSecret()).update(head + '.' + body).digest('base64url');
      const a = Buffer.from(sig);
      const b = Buffer.from(expected);
      if (a.length !== b.length || !crypto.timingSafeEqual(a, b)) return { ok: false, reason: 'signature_invalid' };
      let claims;
      try { claims = b64urlDecode(body); } catch { return { ok: false, reason: 'malformed' }; }
      if (claims.aud !== DEMO_AUDIENCE) return { ok: false, reason: 'aud_mismatch' };
      const now = Math.floor(Date.now() / 1000);
      if (typeof claims.exp !== 'number' || claims.exp < now) return { ok: false, reason: 'token_expired' };
      return { ok: true, claims };
    }

    /** @returns {Promise<{ok:true, demo_session_token:string, expires_at:string, cost_cap_cents:number, allowed_tools:string[]} | {ok:false, reason:string}>} */
    exports.issueDemoSessionToken = async function issueDemoSessionToken({ botid_token, ip, ua, supabaseClient }) {
      const botid = await verifyBotIdToken(botid_token);
      if (!botid.ok) return { ok: false, reason: 'botid_failed', detail: botid.reason };

      const sub = 'demo:' + crypto.randomUUID();
      const iat = Math.floor(Date.now() / 1000);
      const exp = iat + DEMO_TOKEN_TTL_SECONDS;
      const claims = {
        iss: DEMO_ISSUER, aud: DEMO_AUDIENCE, sub, iat, exp,
        allowed_tools: [...DEMO_ALLOWED_TOOLS],
        cost_cap_cents: DEMO_COST_CAP_CENTS,
        synthetic_tenant_id: DEMO_SYNTHETIC_TENANT_ID,
      };
      const token = signJwt(claims);

      try {
        if (supabaseClient) {
          await enqueueAuditStaging(supabaseClient, {
            source_domain: 'marketing',
            action: 'demo.session_started',
            actor_type: 'anon',
            actor_id: sub,
            tenant_id: DEMO_SYNTHETIC_TENANT_ID,
            payload: {
              ip_hash: ip ? crypto.createHash('sha256').update(ip).digest('hex').slice(0, 16) : null,
              ua_hash: ua ? crypto.createHash('sha256').update(ua).digest('hex').slice(0, 16) : null,
              sub, exp,
            },
          });
        }
      } catch { /* fail-soft for audit */ }

      return {
        ok: true,
        demo_session_token: token,
        expires_at: new Date(exp * 1000).toISOString(),
        cost_cap_cents: DEMO_COST_CAP_CENTS,
        allowed_tools: [...DEMO_ALLOWED_TOOLS],
      };
    };

    exports.verifyDemoSessionToken = verifyJwt;

    /** @returns {{ok:true}|{ok:false, reason:string}} */
    exports.assertToolAllowed = function assertToolAllowed(claims, toolName) {
      if (!claims || !Array.isArray(claims.allowed_tools)) return { ok: false, reason: 'malformed' };
      if (!claims.allowed_tools.includes(toolName)) return { ok: false, reason: 'tool_not_allowed', detail: toolName };
      return { ok: true };
    };

    /** @returns {Promise<{ok:true, total_cents:number}|{ok:false, reason:string, total_cents?:number}>} */
    exports.recordDemoCost = async function recordDemoCost(supabaseClient, tokenSub, syntheticTenantId, costCents) {
      // 1. Read current total for this token's session.
      const { data: existing, error: readErr } = await supabaseClient
        .from('markos_mcp_cost_events')
        .select('cost_cents')
        .eq('tenant_id', syntheticTenantId)
        .eq('mcp_session_id', tokenSub);
      if (readErr) return { ok: false, reason: 'db_error', detail: readErr.message };
      const prevTotal = (existing || []).reduce((s, r) => s + (r.cost_cents || 0), 0);
      const newTotal = prevTotal + costCents;

      if (newTotal > DEMO_COST_CAP_CENTS) {
        return { ok: false, reason: 'cost_cap_exceeded', total_cents: newTotal };
      }

      const { error: insErr } = await supabaseClient
        .from('markos_mcp_cost_events')
        .insert({
          tenant_id: syntheticTenantId,
          mcp_session_id: tokenSub,
          tool_name: 'demo',
          cost_cents: costCents,
        });
      if (insErr) return { ok: false, reason: 'db_error', detail: insErr.message };
      return { ok: true, total_cents: newTotal };
    };

    exports.DEMO_TOKEN_TTL_SECONDS = DEMO_TOKEN_TTL_SECONDS;
    exports.DEMO_COST_CAP_CENTS = DEMO_COST_CAP_CENTS;
    exports.DEMO_ALLOWED_TOOLS = DEMO_ALLOWED_TOOLS;
    exports.DEMO_SYNTHETIC_TENANT_ID = DEMO_SYNTHETIC_TENANT_ID;
    exports.DEMO_AUDIENCE = DEMO_AUDIENCE;
    ```

    ```typescript
    export async function POST(req: NextRequest) {

        span.setAttribute('status_code', 200);
        span.setAttribute('demo_token_sub', 'issued');
        return NextResponse.json({
          demo_session_token: result.demo_session_token,
          expires_at: result.expires_at,
          cost_cap_cents: result.cost_cap_cents,
    ```
