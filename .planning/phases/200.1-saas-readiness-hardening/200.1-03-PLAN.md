Parity definition for this plan:
1. **Set equality on filenames.** The set of `*.json` filenames in `bin/lib/presets/` MUST equal the set in `.agent/markos/templates/presets/`.
2. **Canonical-JSON byte equality on contents.** For every shared filename, after parsing and re-serializing both files with `JSON.stringify(value, null, 2)` + sorted keys (deep), the resulting strings MUST be byte-identical.

Rationale for canonical re-serialization: tolerates incidental whitespace / key-ordering differences without false-positive failing the gate. If operators want strict byte-equality, the script also exposes `--strict` flag that compares raw `fs.readFileSync` strings.

CLI surface (REQUIRED — used by the negative-drift Test 9):
- `--bin-dir <path>` overrides the default `bin/lib/presets` path.
- `--agent-dir <path>` overrides the default `.agent/markos/templates/presets` path.
- `--strict` skips canonical normalization and compares raw bytes.
</interfaces>

    Implement `scripts/ci/check-preset-parity.cjs`:

    ```javascript
    #!/usr/bin/env node
    'use strict';
    // Phase 200.1 D-210: byte-parity gate between bin/lib/presets and .agent/markos/templates/presets.

    const fs = require('node:fs');
    const path = require('node:path');

    function parseArgs(argv) {
      const args = { binDir: 'bin/lib/presets', agentDir: '.agent/markos/templates/presets', strict: false };
      for (let i = 2; i < argv.length; i++) {
        const a = argv[i];
        if (a === '--strict') args.strict = true;
        else if (a === '--bin-dir') args.binDir = argv[++i];
        else if (a === '--agent-dir') args.agentDir = argv[++i];
      }
      return args;
    }

    function listJsonFiles(dir) {
      if (!fs.existsSync(dir)) return [];
      return fs.readdirSync(dir).filter(f => f.endsWith('.json')).sort();
    }

    function sortKeysDeep(value) {
      if (Array.isArray(value)) return value.map(sortKeysDeep);
      if (value && typeof value === 'object') {
        const out = {};
        for (const k of Object.keys(value).sort()) out[k] = sortKeysDeep(value[k]);
        return out;
      }
      return value;
    }

    function canonical(jsonString) {
      return JSON.stringify(sortKeysDeep(JSON.parse(jsonString)), null, 2);
    }

    function main() {
      const args = parseArgs(process.argv);
      const binFiles = listJsonFiles(args.binDir);
      const agentFiles = listJsonFiles(args.agentDir);
      const binSet = new Set(binFiles);
      const agentSet = new Set(agentFiles);

      const errors = [];
      for (const f of binFiles) if (!agentSet.has(f)) errors.push(`missing in ${args.agentDir}: ${f}`);
      for (const f of agentFiles) if (!binSet.has(f)) errors.push(`extra in ${args.agentDir} (not in ${args.binDir}): ${f}`);

      const shared = binFiles.filter(f => agentSet.has(f));
      let mismatches = 0;
      for (const f of shared) {
        const binRaw = fs.readFileSync(path.join(args.binDir, f), 'utf8');
        const agentRaw = fs.readFileSync(path.join(args.agentDir, f), 'utf8');
        const binCmp = args.strict ? binRaw : canonical(binRaw);
        const agentCmp = args.strict ? agentRaw : canonical(agentRaw);
        if (binCmp !== agentCmp) {
          mismatches += 1;
          // Emit a small unified-style diff snippet (first 5 differing lines).
          const binLines = binCmp.split('\n');
          const agentLines = agentCmp.split('\n');
          const diffLines = [];
          for (let i = 0; i < Math.max(binLines.length, agentLines.length) && diffLines.length < 10; i++) {
            if (binLines[i] !== agentLines[i]) {
              diffLines.push(`-${binLines[i] ?? ''}`);
              diffLines.push(`+${agentLines[i] ?? ''}`);
            }
          }
          errors.push(`${f}: content diverges\n${diffLines.join('\n')}`);
        }
      }

      if (errors.length > 0) {
        console.error('preset parity FAIL:');
        for (const e of errors) console.error('  ' + e);
        process.exit(1);
      }

      console.log(`preset parity OK: ${shared.length}/${binFiles.length} files match`);
      process.exit(0);
    }

    if (require.main === module) main();
    module.exports = { parseArgs, listJsonFiles, sortKeysDeep, canonical, main };
    ```
