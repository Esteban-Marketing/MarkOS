<interfaces>
Supabase Vault — built-in pgsodium-backed encrypted secret storage. Operations:
- `select vault.create_secret(value text, name text, description text default '')` → returns secret_id (uuid)
- `select * from vault.decrypted_secrets where name = $1` → returns decrypted value (service-role only)
- `select vault.update_secret(secret_id, value)` → rotates a secret in place
- `delete from vault.secrets where name = $1` → revokes

Important Supabase Vault constraints (D-203 specifics):
- Reads require service-role JWT — webhook delivery worker MUST use service-role context (already does, per Phase 200-03 background-worker shape).
- Vault entries are global to the project, not per-tenant — naming convention `markos:webhook:secret:<subscription_uuid>` keeps them logically scoped.
- Vault entries cannot be enumerated by non-service-role; this is the encryption-at-rest guarantee.

Migration 70.2 must run as service-role inside a transaction. Backfill plan:
1. ALTER TABLE markos_webhook_subscriptions ADD COLUMN secret_vault_ref text.
2. DO block: FOR each existing row WHERE secret IS NOT NULL: call vault.create_secret(secret, 'markos:webhook:secret:'||id::text); UPDATE the row to set secret_vault_ref = name.
3. ALTER TABLE markos_webhook_subscriptions DROP COLUMN secret.
4. Note: a NULL secret_vault_ref row will fail at delivery-time — this is intentional. Operator must rotate to re-enable.

Rollback adds the column back as nullable but cannot recover the original secrets — they live only in Vault, and the operator's choice to rollback is destructive for delivery. Rollback documents this explicitly.

    ```javascript
    'use strict';
    // Phase 200.1 D-203: Supabase Vault wrapper for webhook signing secrets.
    // Fail-closed: any Vault error throws — caller MUST treat as terminal.

    const VAULT_NAME_PREFIX = 'markos:webhook:secret:';
    exports.VAULT_NAME_PREFIX = VAULT_NAME_PREFIX;

    function vaultName(subscriptionId) {
      return `${VAULT_NAME_PREFIX}${subscriptionId}`;
    }
    exports.vaultName = vaultName;

    /**
     * @param {SupabaseServiceClient} client
     * @param {string} subscriptionId
     * @param {string} secretValue
     * @returns {Promise<string>} vault entry name (= secret_vault_ref column value)
     */
    exports.storeSecret = async function storeSecret(client, subscriptionId, secretValue) {
      const name = vaultName(subscriptionId);
      const { error } = await client.rpc('vault_create_or_update_secret', {
        p_value: secretValue,
        p_name: name,
        p_description: 'markos webhook signing secret',
      });
      if (error) {
        throw new Error(`vault_unavailable:store:${error.message}`);
      }
      return name;
    };

    /**
     * @param {SupabaseServiceClient} client
     * @param {string} vaultRef
     * @returns {Promise<string>} decrypted secret
     */
    exports.readSecret = async function readSecret(client, vaultRef) {
      if (!vaultRef || typeof vaultRef !== 'string') {
        throw new Error('vault_secret_not_found:invalid_ref');
      }
      const { data, error } = await client
        .from('vault.decrypted_secrets')
        .select('decrypted_secret')
        .eq('name', vaultRef)
        .maybeSingle();
      if (error) throw new Error(`vault_unavailable:read:${error.message}`);
      if (!data || !data.decrypted_secret) throw new Error(`vault_secret_not_found:${vaultRef}`);
      return data.decrypted_secret;
    };

    /**
     * @returns {Promise<{ok:true}|{ok:false,reason:string}>}
     */
    exports.deleteSecret = async function deleteSecret(client, vaultRef) {
      if (!vaultRef) return { ok: true };
      const { error } = await client
        .from('vault.secrets')
        .delete()
        .eq('name', vaultRef);
      if (error) return { ok: false, reason: error.message };
      return { ok: true };
    };

    /**
     * @returns {Promise<string>} new vault ref
     */
    exports.rotateSecret = async function rotateSecret(client, subscriptionId, newSecretValue) {
      const name = vaultName(subscriptionId);
      // Delete-then-store is the simplest portable pattern. vault.create_secret throws on
      // duplicate name; the wrapper RPC `vault_create_or_update_secret` handles upsert.
      const ref = await exports.storeSecret(client, subscriptionId, newSecretValue);
      return ref;
    };
    ```

    'use strict';
    // Phase 200.1 D-203: webhook secret rotation endpoint.
    // F-72.1 contract. Returns the new secret in the response body ONCE; never log it.
    // Audit emit follows Phase 201.1 D-101 inline-fail-closed pattern.

    const crypto = require('node:crypto');
    const { createClient } = require('@supabase/supabase-js');
    const { rotateSecret } = require('../../lib/markos/webhooks/secret-vault.cjs');
    const { enqueueAuditStaging } = require('../../lib/markos/audit/writer.cjs');
    const { requireHostedSupabaseAuth } = require('../../lib/auth/require-hosted-supabase-auth.cjs');

    module.exports = async function handler(req, res) {
      if (req.method !== 'POST') {
        res.setHeader('Allow', 'POST');
        return res.status(405).json({ error: 'method_not_allowed' });
      }

      const auth = await requireHostedSupabaseAuth(req, res);
      if (!auth.ok) return; // requireHostedSupabaseAuth has already written the response

      const { user, tenant_id } = auth;
      const { subscription_id } = (req.body || {});
      if (!subscription_id || typeof subscription_id !== 'string') {
        return res.status(400).json({ error: 'missing_subscription_id' });
      }

      const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_ROLE_KEY, { auth: { persistSession: false } });

      // Verify subscription belongs to tenant.
      const { data: sub, error: lookupErr } = await supabase
        .from('markos_webhook_subscriptions')
        .select('id, tenant_id, secret_vault_ref')
        .eq('id', subscription_id)
        .maybeSingle();
      if (lookupErr) return res.status(500).json({ error: 'lookup_failed', detail: lookupErr.message });
      if (!sub) return res.status(404).json({ error: 'not_found' });
      if (sub.tenant_id !== tenant_id) return res.status(403).json({ error: 'forbidden' });

      // Mint + store new secret.
      const newSecret = crypto.randomBytes(32).toString('base64url');
      let newVaultRef;
      try {
        newVaultRef = await rotateSecret(supabase, subscription_id, newSecret);
      } catch (err) {
        return res.status(500).json({ error: 'vault_error', detail: err.message });
      }

      // Update the subscription row's secret_vault_ref (idempotent — same name, but explicit
      // for clarity if the naming convention ever changes).
      const { error: updErr } = await supabase
        .from('markos_webhook_subscriptions')
        .update({ secret_vault_ref: newVaultRef, updated_at: new Date().toISOString() })
        .eq('id', subscription_id);
      if (updErr) return res.status(500).json({ error: 'update_failed', detail: updErr.message });

      // Inline audit emit BEFORE res.end (Phase 201.1 D-101 pattern). Fail-closed.
      try {
        await enqueueAuditStaging(supabase, {
          source_domain: 'webhooks',
          action: 'secret.rotated',
          actor_type: 'user',
          actor_id: user.id,
          tenant_id,
          target_id: subscription_id,
          payload: { subscription_id, vault_ref: newVaultRef },
        });
      } catch (err) {
        // Fail-closed: audit emit failure → 500 with explicit reason.
        // The rotation already succeeded; operator sees this and can re-emit manually,
        // but we surface the gap rather than silently logging.
        return res.status(500).json({ error: 'audit_emit_failed', detail: err.message });
      }

      // Return the new secret ONCE in the body. Never log it.
      return res.status(200).json({
        ok: true,
        subscription_id,
        plaintext_secret_show_once: newSecret,
        rotated_at: new Date().toISOString(),
      });
    };
