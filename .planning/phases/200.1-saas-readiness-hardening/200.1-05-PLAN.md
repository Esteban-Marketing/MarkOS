<objective>
Ship migration 70.1 (nonce store), `lib/markos/webhooks/replay-protection.cjs`, signing.cjs amendments (timestamp + nonce in signature header), the periodic purge cron, and tests. Closes review concern **H2**: "HMAC replay window undefined — captured deliveries can be replayed indefinitely."

Purpose: Phase 200-03 signing.cjs emits HMAC-SHA256 with constant-time verify, but no freshness window and no idempotency-key store. An attacker who captured a delivery's body + signature could replay it indefinitely against the receiver. D-202 closes this with two layers: (1) signing now embeds `t=<unix-seconds>` in the header and verify rejects when `|now - t| > 300`s; (2) every payload also carries a random 16-byte nonce `n=<hex>` and verify INSERTs (subscription_id, nonce) into `markos_webhook_delivery_nonces` — the PK constraint makes replay raise a duplicate-key error which verify converts to `{ ok:false, reason:'replay' }`. The nonce table cleans up via 5-minute cron `DELETE WHERE created_at < now() - interval '10 minutes'`.

Output: 1 forward migration + 1 rollback + 1 lib module (replay-protection) + signing.cjs amendments + 1 cron handler + 1 test file + 1 vercel.ts cron registration.
</objective>

    ```javascript
    exports.FRESHNESS_WINDOW_SECONDS = FRESHNESS_WINDOW_SECONDS;

    /** @typedef {{ok: true}|{ok: false, reason: string, detail?: string}} RecordResult */

    /**
     * @param {{from: (table: string) => any}} client - Supabase client (service-role)
     * @param {string} subscriptionId
     * @param {string} nonce
     * @returns {Promise<RecordResult>}
     */
    exports.recordNonce = async function recordNonce(client, subscriptionId, nonce) {
      try {
        const { error } = await client
          .from('markos_webhook_delivery_nonces')
          .insert({ subscription_id: subscriptionId, nonce });
        if (error) {
          // Postgres unique-violation = 23505. Treat as replay regardless of message.
          if (error.code === '23505' || /duplicate key|unique constraint/i.test(error.message || '')) {
            return { ok: false, reason: 'replay', detail: nonce };
          }
          return { ok: false, reason: 'db_error', detail: error.message };
        }
        return { ok: true };
      } catch (err) {
        return { ok: false, reason: 'db_error', detail: err && err.message ? err.message : String(err) };
      }
    };
    ```

    ```javascript
    'use strict';
    const crypto = require('node:crypto');
    const { isStaleTimestamp } = require('./replay-protection.cjs');

    /** @returns {string} */
    exports.sign = function sign(secret, body, opts = {}) {
      const t = opts.timestamp ?? Math.floor(Date.now() / 1000);
      const n = opts.nonce ?? crypto.randomBytes(16).toString('hex');
      const signedPayload = `${t}.${n}.${body}`;
      const hmac = crypto.createHmac('sha256', secret).update(signedPayload).digest('hex');
      return `t=${t},n=${n},sha256=${hmac}`;
    };

    function parseSignatureHeader(header) {
      // Tolerate any order. Reject if any of the 3 parts missing or duplicated.
      if (typeof header !== 'string') return null;
      const parts = header.split(',').map(s => s.trim());
      const map = {};
      for (const p of parts) {
        const eq = p.indexOf('=');
        if (eq <= 0) return null;
        const k = p.slice(0, eq);
        const v = p.slice(eq + 1);
        if (k in map) return null; // duplicate
        map[k] = v;
      }
      if (!map.t || !map.n || !map.sha256) return null;
      const t = Number(map.t);
      if (!Number.isFinite(t)) return null;
      return { t, n: map.n, sha256: map.sha256 };
    }

    /**
     * @param {string} secret
     * @param {string} body
     * @param {string} signatureHeader
     * @param {{ now?: number, recordNonce?: (n: string) => Promise<RecordResult>, subscriptionId?: string }} [opts]
     */
    exports.verify = async function verify(secret, body, signatureHeader, opts = {}) {
      const parsed = parseSignatureHeader(signatureHeader);
      if (!parsed) return { ok: false, reason: 'malformed_signature' };
      if (isStaleTimestamp(parsed.t, opts.now)) return { ok: false, reason: 'stale_timestamp', detail: String(parsed.t) };
      const signedPayload = `${parsed.t}.${parsed.n}.${body}`;
      const expected = crypto.createHmac('sha256', secret).update(signedPayload).digest('hex');
      const expectedBuf = Buffer.from(expected);
      const gotBuf = Buffer.from(parsed.sha256);
      if (expectedBuf.length !== gotBuf.length) return { ok: false, reason: 'signature_mismatch' };
      if (!crypto.timingSafeEqual(expectedBuf, gotBuf)) return { ok: false, reason: 'signature_mismatch' };
      if (typeof opts.recordNonce === 'function') {
        const r = await opts.recordNonce(parsed.n);
        if (!r.ok) return { ok: false, reason: r.reason, detail: r.detail };
      }
      return { ok: true, timestamp: parsed.t, nonce: parsed.n };
    };

    exports.parseSignatureHeader = parseSignatureHeader;
    ```
