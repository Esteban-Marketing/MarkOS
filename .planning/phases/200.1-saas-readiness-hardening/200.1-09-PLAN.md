    ```javascript
    'use strict';
    // Phase 200.1 D-207: shared OTEL wrapper for api/webhooks/* + api/mcp/*.
    // Closes review concern M5 (gate-9 OTEL coverage).
    // Sink precedence: Sentry > Vercel Observability > console.log JSON.

    let _sentry = null;       // Lazily required @sentry/node
    let _initialized = false;
    let _serviceName = 'markos';
    let _testSink = null;     // Optional test-mode sink injected via setTestSink()

    function tryRequireSentry() {
      try { return require('@sentry/node'); } catch { return null; }
    }

    function dropUndefined(obj) {
      const out = {};
      for (const k of Object.keys(obj || {})) {
        if (obj[k] !== undefined) out[k] = obj[k];
      }
      return out;
    }

    function emitJsonLine(payload) {
      try {
        const line = JSON.stringify({ ts: new Date().toISOString(), service: _serviceName, ...payload });
        if (_testSink) {
          _testSink(line, payload);
        } else {
          console.log(line);
        }
      } catch (err) {
        // Last-resort: never let observability emission throw.
      }
    }

    exports.initOtel = function initOtel({ serviceName = 'markos' } = {}) {
      if (_initialized) return;
      _serviceName = serviceName;
      if (process.env.SENTRY_DSN) {
        _sentry = tryRequireSentry();
        if (_sentry && typeof _sentry.init === 'function') {
          try {
            _sentry.init({ dsn: process.env.SENTRY_DSN, tracesSampleRate: parseFloat(process.env.OTEL_SAMPLE_RATE || '1.0') });
          } catch (err) {
            // Fall through to console fallback.
            _sentry = null;
          }
        }
      }
      _initialized = true;
    };

    exports.setTestSink = function setTestSink(sinkFn) { _testSink = sinkFn; };
    exports.clearTestSink = function clearTestSink() { _testSink = null; };

    exports.startSpan = function startSpan(name, attrs) {
      if (!_initialized) exports.initOtel({});
      const start = Date.now();
      const accumulated = dropUndefined(attrs || {});
      let ended = false;
      const span = {
        setAttribute(k, v) { if (v !== undefined) accumulated[k] = v; },
        end(extraAttrs) {
          if (ended) return;
          ended = true;
          const finalAttrs = { ...accumulated, ...dropUndefined(extraAttrs || {}) };
          const duration_ms = Date.now() - start;
          if (_sentry && typeof _sentry.captureMessage === 'function') {
            try {
              _sentry.captureMessage(name, { level: 'info', tags: finalAttrs, contexts: { trace: { duration_ms } } });
            } catch { /* fall-through */ }
          }
          emitJsonLine({ name, attrs: finalAttrs, duration_ms });
        },
      };
      return span;
    };

    exports.withSpan = async function withSpan(name, attrs, fn) {
      const span = exports.startSpan(name, attrs);
      try {
        const result = await fn(span);
        span.end({ status: 'ok' });
        return result;
      } catch (err) {
        span.end({ error: true, error_message: err && err.message ? err.message : String(err), status: 'error' });
        throw err;
      }
    };

    exports.recordEvent = function recordEvent(name, attrs) {
      if (!_initialized) exports.initOtel({});
      const finalAttrs = dropUndefined(attrs || {});
      if (_sentry && typeof _sentry.captureMessage === 'function') {
        try { _sentry.captureMessage(name, { level: 'info', tags: finalAttrs }); } catch { /* */ }
      }
      emitJsonLine({ event: name, attrs: finalAttrs });
    };
    ```

    ```javascript
    'use strict';
    // Phase 200.1 Plan 09 Task 2 verifier — replaces shell-escaped inline regex + awk pipeline (BLOCKER #4 fix).
    // Reads handlers via fs (no shell-glob; handles literal '[toolName]' filename).
    const fs = require('node:fs');

    const HANDLERS = [
      ['api/webhooks/subscribe.js', 'webhook.subscribe'],
      ['api/webhooks/unsubscribe.js', 'webhook.unsubscribe'],
      ['api/webhooks/list.js', 'webhook.list'],
      ['api/webhooks/test-fire.js', 'webhook.test_fire'],
      ['api/webhooks/rotate-secret.js', 'webhook.rotate_secret'],
      ['api/mcp/session.js', 'mcp.session'],
      ['api/mcp/tools/[toolName].js', 'mcp.tool'],
      ['api/mcp/keys/create.js', 'mcp.keys.create'],
      ['api/mcp/keys/list.js', 'mcp.keys.list'],
      ['api/mcp/keys/revoke.js', 'mcp.keys.revoke'],
    ];

    const failures = [];
    let totalSetAttribute = 0;

    for (const [file, spanName] of HANDLERS) {
      let s;
      try { s = fs.readFileSync(file, 'utf8'); }
      catch (err) { failures.push(`${file}: cannot read (${err.message})`); continue; }

      if (!s.includes('observability/otel')) failures.push(`${file}: missing 'observability/otel' require`);
      if (!s.includes(`withSpan('${spanName}'`) && !s.includes(`withSpan("${spanName}"`)) {
        failures.push(`${file}: missing withSpan('${spanName}')`);
      }
      // Count setAttribute('tenant_id', ...) calls across all handlers.
      const tenantMatches = s.match(/setAttribute\(\s*['"]tenant_id['"]/g);
      if (tenantMatches) totalSetAttribute += tenantMatches.length;
    }

    // Tools handler must additionally emit recordEvent('mcp.tool.invoked', ...).
    try {
      const tools = fs.readFileSync('api/mcp/tools/[toolName].js', 'utf8');
      if (!tools.includes(`recordEvent('mcp.tool.invoked'`) && !tools.includes(`recordEvent("mcp.tool.invoked"`)) {
        failures.push(`api/mcp/tools/[toolName].js: missing recordEvent('mcp.tool.invoked')`);
      }
    } catch (err) {
      failures.push(`api/mcp/tools/[toolName].js: cannot read (${err.message})`);
    }

    // Aggregate: at least 7 setAttribute('tenant_id', ...) calls across handlers (auth-error early returns

    if (totalSetAttribute < 7) {
      failures.push(`aggregate: only ${totalSetAttribute} setAttribute('tenant_id', ...) calls across 10 handlers (need ≥7)`);

    if (failures.length) {

      process.exit(1);
    }
    ```
