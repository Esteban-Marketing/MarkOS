<task id="1" name="Update package.json name and description">

  </read_first>
  <action>

    - `"name"`: `"markos"` → `"markos"`

<task id="2" name="Update package.json bin entries">
  <read_first>
    - package.json (current bin object)
  </read_first>
  <action>

    - Add: `"markos": "./bin/install.cjs"`

    ```json

      "markos": "./bin/install.cjs"
    ```

  </action>
  <acceptance_criteria>

    - `require('./package.json').bin.markos === './bin/install.cjs'`
    - `require('./package.json').bin['markos']` is `undefined`
  </acceptance_criteria>
</task>
