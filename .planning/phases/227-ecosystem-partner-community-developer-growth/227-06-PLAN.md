      ```sql
      -- D-79 / RM4: DB-level enforcement of commission_share immutability post-acceptance.
      -- App layer alone is insufficient — service-role / direct SQL must also be blocked.
      CREATE OR REPLACE FUNCTION cosell_commission_immutability_check()
      RETURNS TRIGGER AS $$
      BEGIN
        -- If commission_locked_at was set on OLD row AND commission_share is changing → REJECT
        IF OLD.commission_locked_at IS NOT NULL
           AND NEW.commission_share IS DISTINCT FROM OLD.commission_share THEN
          RAISE EXCEPTION 'COMMISSION_IMMUTABLE_POST_ACCEPTANCE: commission_share cannot be modified after commission_locked_at is set (cosell_id=%, locked_at=%)',
            OLD.cosell_id, OLD.commission_locked_at
            USING ERRCODE = '23514';
        END IF;
        -- Also lock rev_share_terms_id (Quote-as-Snapshot mirror — D-32)
        IF OLD.commission_locked_at IS NOT NULL
           AND NEW.rev_share_terms_id IS DISTINCT FROM OLD.rev_share_terms_id THEN
          RAISE EXCEPTION 'COMMISSION_IMMUTABLE_POST_ACCEPTANCE: rev_share_terms_id cannot be modified after commission_locked_at (cosell_id=%)',
            OLD.cosell_id
            USING ERRCODE = '23514';
        END IF;
        RETURN NEW;
      END;
      $$ LANGUAGE plpgsql;

      DROP TRIGGER IF EXISTS trg_cosell_commission_immutability ON co_sell_opportunities;
      CREATE TRIGGER trg_cosell_commission_immutability
        BEFORE UPDATE ON co_sell_opportunities
        FOR EACH ROW
        EXECUTE FUNCTION cosell_commission_immutability_check();
      ```
