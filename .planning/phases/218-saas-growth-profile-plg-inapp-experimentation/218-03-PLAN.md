      export async function evaluateTriggerCondition({ trigger_id, account_id }, supabase): Promise<boolean> {
        const { data: trigger } = await supabase.from('upgrade_triggers').select('*').eq('trigger_id', trigger_id).single();
        if (!trigger || trigger.status !== 'active') return false;
        const { metric, threshold, window } = trigger.trigger_condition_jsonb;
        // Branch on trigger_type
        switch (trigger.trigger_type) {
          case 'hard_limit_reached':
          case 'usage_velocity_warning':
          case 'seat_limit_approached': {
            const { data: health } = await supabase.from('saas_health_scores').select(metric).eq('tenant_id', trigger.tenant_id).eq('account_id', account_id).single();
            return Boolean(health && health[metric] >= threshold);
          }
          case 'value_recognition_moment':
            // Delegate to event-name match against P210 connector events (out-of-scope here; placeholder false)

          case 'plan_anniversary': {
            const { data: sub } = await supabase.from('tenant_billing_subscriptions').select('start_date').eq('tenant_id', trigger.tenant_id).eq('account_id', account_id).single();

          }
          default:
            return false;
        }
