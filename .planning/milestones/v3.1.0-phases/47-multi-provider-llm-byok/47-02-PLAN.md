### 1. Database Migration: `supabase/migrations/47_operator_llm_management.sql`

**Purpose:** Create encrypted key storage + preference tracking tables with RLS.

**Contents:**

```sql
-- Create operator_api_keys table for encrypted secret storage
CREATE TABLE operator_api_keys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  operator_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  provider TEXT NOT NULL CHECK (provider IN ('anthropic', 'openai', 'gemini')),
  encrypted_key TEXT NOT NULL,  -- pgp_sym_encrypt() ciphertext
  key_hash TEXT NOT NULL,       -- SHA256(key) for audit/dedup
  rotation_scheduled_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(operator_id, provider)
);

CREATE INDEX idx_operator_api_keys_operator_id ON operator_api_keys(operator_id);
CREATE INDEX idx_operator_api_keys_created_at ON operator_api_keys(created_at);

-- RLS policies: only operators can read/write their own keys
ALTER TABLE operator_api_keys ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Operators can read own keys"
  ON operator_api_keys FOR SELECT
  USING (auth.uid() = operator_id);

CREATE POLICY "Operators can insert own keys"
  ON operator_api_keys FOR INSERT
  WITH CHECK (auth.uid() = operator_id);

CREATE POLICY "Operators can update own keys"
  ON operator_api_keys FOR UPDATE
  USING (auth.uid() = operator_id)
  WITH CHECK (auth.uid() = operator_id);

CREATE POLICY "Operators can delete own keys"
  ON operator_api_keys FOR DELETE
  USING (auth.uid() = operator_id);

-- Create operator_llm_preferences table
CREATE TABLE operator_llm_preferences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  operator_id UUID NOT NULL UNIQUE REFERENCES auth.users(id) ON DELETE CASCADE,
  available_providers TEXT[] NOT NULL DEFAULT ARRAY['anthropic', 'openai', 'gemini'],
  primary_provider TEXT NOT NULL DEFAULT 'anthropic'
    CHECK (primary_provider IN ('anthropic', 'openai', 'gemini')),
  cost_budget_monthly_usd DECIMAL(10,2) NOT NULL DEFAULT 100.00,
  allow_fallback BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- RLS policies for preferences
ALTER TABLE operator_llm_preferences ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Operators can read own preferences"
  ON operator_llm_preferences FOR SELECT
  USING (auth.uid() = operator_id);

CREATE POLICY "Operators can insert own preferences"
  ON operator_llm_preferences FOR INSERT
  WITH CHECK (auth.uid() = operator_id);

CREATE POLICY "Operators can update own preferences"
  ON operator_llm_preferences FOR UPDATE
  USING (auth.uid() = operator_id)
  WITH CHECK (auth.uid() = operator_id);

-- Helper function: decrypt operator API key (server-side only)
CREATE OR REPLACE FUNCTION decrypt_operator_api_key(
  encrypted_key_text TEXT,
  vault_secret_key TEXT
) RETURNS TEXT AS $$
BEGIN
  RETURN pgp_sym_decrypt(encrypted_key_text::bytea, vault_secret_key);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Grant function access to authenticated users (Phase 48: restrict to service role)
GRANT EXECUTE ON FUNCTION decrypt_operator_api_key TO authenticated;

-- Create telemetry events table (consumed by Wave 3)
CREATE TABLE markos_llm_call_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  operator_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  provider TEXT NOT NULL CHECK (provider IN ('anthropic', 'openai', 'gemini')),
  model TEXT NOT NULL,
  input_tokens INTEGER NOT NULL,
  output_tokens INTEGER NOT NULL,
  estimated_cost_usd DECIMAL(10,6) NOT NULL,
  decision_mode TEXT CHECK (decision_mode IN ('explicit', 'default', 'fallback')),
  fallback_attempts INTEGER DEFAULT 0,
  latency_ms INTEGER,
  context JSONB,  -- task_id, flow_domain, campaign_id, etc.
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_llm_call_events_operator_created ON markos_llm_call_events(operator_id, created_at);
CREATE INDEX idx_llm_call_events_provider ON markos_llm_call_events(provider);

-- RLS: operators can only read their own events
ALTER TABLE markos_llm_call_events ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Operators can read own events"
  ON markos_llm_call_events FOR SELECT
  USING (auth.uid() = operator_id);

CREATE POLICY "Telemetry can insert events"
  ON markos_llm_call_events FOR INSERT
  WITH CHECK (auth.uid() = operator_id);
```

### 2. `lib/markos/llm/settings.ts` (Config Schema + Validation)

**Purpose:** TypeScript schema for operator LLM settings with validation and loading logic.

**Exports:**

```typescript
import { z } from "zod";

export const OperatorLLMSettingsSchema = z.object({
  available_providers: z.enum(['anthropic', 'openai', 'gemini']).array(),
  primary_provider: z.enum(['anthropic', 'openai', 'gemini']),
  cost_budget_monthly_usd: z.number().min(0),
  allow_fallback: z.boolean().default(true),
});

export type OperatorLLMSettings = z.infer<typeof OperatorLLMSettingsSchema>;

export const DEFAULT_SETTINGS: OperatorLLMSettings = {
  available_providers: ['anthropic', 'openai', 'gemini'],
  primary_provider: 'anthropic',
  cost_budget_monthly_usd: 100.00,
  allow_fallback: true,
};

export async function loadOperatorSettings(operatorId: string): Promise<OperatorLLMSettings> {
  // Try MarkOSDB first
  try {
    const supabase = createClient(supabaseUrl, supabaseKey);
    const { data, error } = await supabase
      .from('operator_llm_preferences')
      .select('*')
      .eq('operator_id', operatorId)
      .single();
    
    if (error) throw error;
    if (data) {
      return OperatorLLMSettingsSchema.parse(data);
    }
  } catch (err) {
    console.warn(`Failed to load settings from MarkOSDB: ${err.message}`);
  }
  
  // Fallback: check env vars for local development
  const envProvider = process.env.MARKOS_LLM_PRIMARY_PROVIDER;
  if (envProvider && envProvider in ['anthropic', 'openai', 'gemini']) {
    return {
      ...DEFAULT_SETTINGS,
      primary_provider: envProvider as any,
    };
  }
  
  // Final fallback: use defaults
  return DEFAULT_SETTINGS;
}

export function validateSettings(config: unknown): OperatorLLMSettings {
  return OperatorLLMSettingsSchema.parse(config);
}
```

### 3. `lib/markos/llm/encryption.ts` (Encryption/Decryption Helpers)

**Purpose:** Handle API key encryption and secure storage.

**Exports:**

```typescript
import crypto from 'crypto';
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export async function encryptOperatorKey(
  plaintextKey: string,
  operatorId: string,
  vaultSecret: string = process.env.MARKOS_VAULT_SECRET!
): Promise<{ encrypted: string; hash: string }> {
  // Hash the key for deduplication audit
  const hash = crypto.createHash('sha256').update(plaintextKey).digest('hex');
  
  // Use Supabase RLS-protected function to encrypt
  const { data, error } = await supabase.rpc('encrypt_operator_key', {
    plaintext_key: plaintextKey,
    vault_secret: vaultSecret,
  });
  
  if (error) throw error;
  return { encrypted: data, hash };
}

export async function decryptOperatorKey(
  encryptedKey: string,
  operatorId: string,
  vaultSecret: string = process.env.MARKOS_VAULT_SECRET!
): Promise<string> {
  // Call RLS-protected server function to decrypt
  const { data, error } = await supabase.rpc('decrypt_operator_api_key', {
    encrypted_key_text: encryptedKey,
    vault_secret_key: vaultSecret,
  });
  
  if (error) throw error;
  return data;
}

export function redactKeyForLogging(key: string): string {
  // Return first 8 + last 4 chars, hide middle
  if (key.length <= 12) return '***';
  return `${key.substring(0, 8)}...${key.substring(key.length - 4)}`;
}
```

### 4. Unit Tests (`test/llm-adapter/settings.test.js`)

**Coverage:** 6 tests, ~10 minutes to run

```javascript
describe("LLM Settings & Encryption", () => {
  test("encryption round-trip: encrypt(x) -> decrypt() -> x", async () => {
    const key = "test_anthropic_key_12345";
    const { encrypted } = await encryptOperatorKey(key, "test-operator-id");
    const decrypted = await decryptOperatorKey(encrypted, "test-operator-id");
    expect(decrypted).toBe(key);
  });

  test("settings validation accepts valid configs", () => {
    const valid = {
      available_providers: ['anthropic', 'openai'],
      primary_provider: 'anthropic',
      cost_budget_monthly_usd: 50.00,
      allow_fallback: true,
    };
    const result = validateSettings(valid);
    expect(result.primary_provider).toBe('anthropic');
  });

  test("settings validation rejects invalid provider", () => {
    const invalid = {
      available_providers: ['invalid_provider'],
      primary_provider: 'anthropic',
      cost_budget_monthly_usd: 50.00,
    };
    expect(() => validateSettings(invalid)).toThrow();
  });

  test("loadOperatorSettings returns defaults when DB unavailable", async () => {
    // Mock Supabase to fail
    const settings = await loadOperatorSettings("test-operator-id");
    expect(settings.primary_provider).toBeDefined();
    expect(settings.cost_budget_monthly_usd).toBeGreaterThanOrEqual(0);
  });

  test("key redaction hides bulk of secret", () => {
    const redacted = redactKeyForLogging("anthropic_abc123def456ghi789");
    expect(redacted).toMatch(/^anthropic_\/\.\.\..{4}$/);
    expect(redacted).not.toContain("def456ghi");
  });

  test("RLS policy prevents cross-operator key access", async () => {
    // This is tested via Supabase RLS verifier in Wave 5 integration tests
    // Unit test: verify function signature + error handling
    expect(() => decryptOperatorKey("bad_ciphertext", "test-id")).rejects.toThrow();
  });
});
```

## Risks & Mitigations

| Risk | Severity | Mitigation |
|------|----------|-----------|
| pgcrypto extension not enabled | HIGH | Check Supabase SQL editor: `\dx` should list pgcrypto. Enable if missing: `CREATE EXTENSION pgcrypto;` |
| RLS policies too restrictive → operators can't decrypt keys | HIGH | Test with `serviceRole` bypass in dev; verify policies in Supabase dashboard |
| Encryption key rotation in Phase 47 not supported | MEDIUM | Document as Phase 48 feature; Phase 47 uses static `MARKOS_VAULT_SECRET` |
| Key hash collisions (SHA256 dedup fails) | LOW | Extremely unlikely; SHA256 has 2^256 space |
